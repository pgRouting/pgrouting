12 *
13 * Routine to calculate the node order for a Traveling Salesman Problem
14 * from a distance matrix.
15 *
16 * $Log: tsplib.c,v $
17 * Revision 1.1 2006/05/13 23:39:56 woodbri
18 * Initial checkin of IMT::TSP module.
19 *
20 * Initial revision created from router tsp2.c
21 *
22 **********************************************************************
23 *
24 * Simulated annealing and a non symetric
25 * Euclidian Traveling Salesman Problem.
26 *
27 * Solution based on local search heuristics for
28 * non-crossing paths and nearest neighbors
29 *
30 * Storage Requirements: 4n ints
31 *
32 * Problem: given an nXn matrix of costs for n cities in the plane,
33 * find a permutation pi_1, pi_2, ..., pi_n of 1, 2, ..., n that
34 * minimizes sum for 1<=i<n D(pi_i,pi_i+1), where D(i,j) is the
35 * euclidian distance between cities i and j from the matrix
36 *
37 * This means that the ith row is the "from" city and the jth column
38 * is the cost to the "to" city.
39 *
40 * NOTE: costs are assumed to be ints, so scaling them to meters or
41 * some other int would be advisable.
42 *
43 * NOTE: the internal cost calculations ASSUME a symetric matrix. There
44 * are notes of what needs to be updated, but all D(i,j)s need to be
45 * reviewed because order is important for asymetric cases.
46 *
47 * Note: with n cities, there is (n-1)!/2 possible tours.
48 * factorial(10)=3628800 factorial(50)=3E+64 factorial(150)=5.7E+262
49 * If we could check one tour per clock cycle on a 100 MHZ computer, we
50 * would still need to wait approximately 10^236 times the age of the
51 * universe to explore all tours for 150 cities.
52 *
53 * Derivied from code 1995 by Maugis Lionel (Sofreavia) on sid1
54 * that was placed in the public domain.
55 *************************************************************************/

97 {

102  if (seed < 0) seed += PRANDMAX;
103 
104  arr[0] = last = seed;

106  for (i = 1; i < 55; i++) {

109  next = last - next;
110  if (next < 0)
111  next += PRANDMAX;
112  last = arr[ii];
113  }
114  a = 0;
115  b = 24;
116  for (i = 0; i < 165; i++)
117  last = Rand ();

169  * Prim's approximated TSP tour
170  * See also [Cristophides'92]

172 int findEulerianPath(TSP *tsp)
173 {
174  int *mst, *arc;
175  int i, j, k, l, a;
176  int n, *iorder, *jorder;
177  DTYPE d;
178  DTYPE maxd;
179  DTYPE *dist;
180  DTYPE *dis;
181 
182  jorder = tsp->jorder;
183  iorder = tsp->iorder;
184  dist = tsp->dist;
185  maxd = tsp->maxd;
186  n = tsp->n;
187 
188  if (!(mst = (int*) palloc(n * sizeof(int))) ||
189  !(arc = (int*) palloc(n * sizeof(int))) ||
190  !(dis = (DTYPE*) palloc(n * sizeof(DTYPE))) )
191  {
192  elog(ERROR, "Failed to allocate memory!");
193  return -1;
194  }
195  //DBG("findEulerianPath: 1");
196 
197  k = -1;
198  j = -1;
199  d = maxd;
200  dis[0] = -1;
201  for (i = 1; i < n; i++)
202  {
203  dis[i] = D(i,0);
204  arc[i] = 0;
205  if (d > dis[i])
206  {
207  d = dis[i];
208  j = i;
209  }
210  }
211  //DBG("findEulerianPath: j=%d", j);
212 
213  if (j == -1)
214  elog(ERROR, "Error TSP fail to findEulerianPath, check your distance matrix is valid.");
215 
216  /*
217  * O(n^2) Minimum Spanning Trees by Prim and Jarnick
218  * for graphs with adjacency matrix.
219  */
220  for (a = 0; a < n - 1; a++)
221  {
222  mst[a] = j * n + arc[j]; /* join fragment j with MST */
223  dis[j] = -1;
224  d = maxd;
225  for (i = 0; i < n; i++)
226  {
227  if (dis[i] >= 0) /* not connected yet */
228  {
229  if (dis[i] > D(i,j))
230  {
231  dis[i] = D(i,j);
232  arc[i] = j;
233  }
234  if (d > dis[i])
235  {
236  d = dis[i];
237  k = i;
238  }
239  }
240  }
241  j = k;
242  }
243  //DBG("findEulerianPath: 3");
244 
245  /*
246  * Preorder Tour of MST
247  */
248 #define VISITED(x) jorder[x]
249 #define NQ(x) arc[l++] = x
250 #define DQ() arc[--l]
251 #define EMPTY (l==0)
252 
253  for (i = 0; i < n; i++) VISITED(i) = 0;
254  k = 0; l = 0; d = 0; NQ(0);
255  while (!EMPTY)
256  {
257  i = DQ();
258  if (!VISITED(i))
259  {
260  iorder[k++] = i;
261  VISITED(i) = 1;
262  for (j = 0; j < n - 1; j++) /* push all kids of i */
263  {
264  if (i == mst[j]%n) NQ(mst[j]/n);
265  }
266  }
267  }
268  //DBG("findEulerianPath: 4");
269 
270  return 0;
271 }

298 DTYPE getThreeWayCost (TSP *tsp, Path p)
299 {
300  int a, b, c, d, e, f;
301  int *iorder = tsp->iorder;
302  DTYPE *dist = tsp->dist;
303  int n = tsp->n;
304 
305  a = iorder[MOD(p[0]-1,n)];
306  b = iorder[p[0]];
307  c = iorder[p[1]];
308  d = iorder[MOD(p[1]+1,n)];
309  e = iorder[p[2]];
310  f = iorder[MOD(p[2]+1,n)];
311 
312  return (D(a,d) + D(e,b) + D(c,f) - D(a,b) - D(c,d) - D(e,f));

316 void doThreeWay (TSP *tsp, Path p)
317 {
318  int i, count, m1, m2, m3, a, b, c, d, e, f;
319  int *iorder = tsp->iorder;
320  int *jorder = tsp->jorder;
321  int n = tsp->n;
322 
323  a = MOD(p[0]-1,n);
324  b = p[0];
325  c = p[1];
326  d = MOD(p[1]+1,n);
327  e = p[2];
328  f = MOD(p[2]+1,n);
329 
330  m1 = MOD(n+c-b,n)+1; /* num cities from b to c */
331  m2 = MOD(n+a-f,n)+1; /* num cities from f to a */
332  m3 = MOD(n+e-d,n)+1; /* num cities from d to e */
333 
334  count = 0;
335  /* [b..c] */
336  for (i = 0; i < m1; i++) jorder[count++] = iorder[MOD(i+b,n)];
337 
338  /* [f..a] */
339  for (i = 0; i < m2; i++) jorder[count++] = iorder[MOD(i+f,n)];
340 
341  /* [d..e] */
342  for (i = 0; i < m3; i++) jorder[count++] = iorder[MOD(i+d,n)];
343 
344  /* copy segment back into iorder */
345  for (i = 0; i < n; i++) iorder[i] = jorder[i];
346 }

370 void doReverse(TSP *tsp, Path p)
371 {
372  int i, nswaps, first, last, tmp;
373  int *iorder = tsp->iorder;
374  int n = tsp->n;
375 
376 
377  /* reverse path b...c */
378  nswaps = (MOD(p[1]-p[0],n)+1)/2;
379  for (i = 0; i < nswaps; i++)
380  {
381  first = MOD(p[0]+i, n);
382  last = MOD(p[1]-i, n);
383  tmp = iorder[first];
384  iorder[first] = iorder[last];
385  iorder[last] = tmp;

389 void annealing(TSP *tsp)
390 {
391  Path p;
392  int i, j, pathchg;
393  int numOnPath, numNotOnPath;
394  DTYPE pathlen;
395  int n = tsp->n;
396  double energyChange, T;
397 
398  pathlen = pathLength (tsp);
399 
400  for (T = T_INIT; T > FINAL_T; T *= COOLING) /* annealing schedule */
401  {
402  pathchg = 0;
403  for (j = 0; j < TRIES_PER_T; j++)
404  {
405  do {
406  p[0] = unifRand (n);
407  p[1] = unifRand (n);
408  /* non-empty path */
409  if (p[0] == p[1]) p[1] = MOD(p[0]+1,n);
410  numOnPath = MOD(p[1]-p[0],n) + 1;
411  numNotOnPath = n - numOnPath;
412  } while (numOnPath < 2 || numNotOnPath < 2); /* non-empty path */
413 
414  if (RANDOM() % 2) /* threeWay */
415  {
416  do {
417  p[2] = MOD(unifRand (numNotOnPath)+p[1]+1,n);
418  } while (p[0] == MOD(p[2]+1,n)); /* avoids a non-change */
419 
420  energyChange = getThreeWayCost (tsp, p);
421  if (energyChange < 0 || RREAL < exp(-energyChange/T) )
422  {
423  pathchg++;
424  pathlen += energyChange;
425  doThreeWay (tsp, p);
426  }
427  }
428  else /* path Reverse */
429  {
430  energyChange = getReverseCost (tsp, p);
431  if (energyChange < 0 || RREAL < exp(-energyChange/T))
432  {
433  pathchg++;
434  pathlen += energyChange;
435  doReverse(tsp, p);
436  }
437  }
438  // if the new length is better than best then save it as best
439  if (pathlen < tsp->bestlen) {
440  tsp->bestlen = pathlen;
441  for (i=0; i<tsp->n; i++) tsp->border[i] = tsp->iorder[i];
442  }
443  if (pathchg > IMPROVED_PATH_PER_T) break; /* finish early */
444  }
445  DBG("T:%f L:%f B:%f C:%d", T, pathlen, tsp->bestlen, pathchg);
446  if (pathchg == 0) break; /* if no change then quit */
447  }
448 }

468  int jend = -1;

471  long seed = -314159L;
472  DTYPE blength;
473 
474  DBG("sizeof(long)=%d", (int)sizeof(long));

476  initRand (seed);
477 
478 #ifdef DEBUG
479  char bufff[2048];

481  DBG("---------- Matrix[%d][%d] ---------------------\n", num, num);
482  for (i=0; i<num; i++) {
483  sprintf(bufff, "%d:", i);
484  nnn = 0;
485  for (j=0; j<num; j++) {
486  nnn += sprintf(bufff+nnn, "\t%.4f", cost[i*num+j]);
487  }
488  DBG("%s", bufff);
489  }
490 #endif
491

494  tsp.dist = NULL;
495  tsp.iorder = NULL;
496  tsp.jorder = NULL;
497  tsp.border = NULL;
498 
499  if (!(tsp.iorder = (int*) palloc (tsp.n * sizeof(int))) ||
500  !(tsp.jorder = (int*) palloc (tsp.n * sizeof(int))) ||
501  !(tsp.border = (int*) palloc (tsp.n * sizeof(int))) ) {
502  elog(FATAL, "Memory allocation failed!");
503  return -1;
504  }
505 
506  tsp.dist = cost;
507  tsp.maxd = 0;
508  for (i=0; i<tsp.n*tsp.n; i++) {

512  /* identity permutation */
513  for (i = 0; i < tsp.n; i++) tsp.iorder[i] = i;

516  for (i = 0; i < tsp.n; i++) tsp.border[i] = tsp.iorder[i];

518  DBG("Initial Path Length: %.4f", tsp.bestlen);

522  * simulated annealing.
523  */
524  if(findEulerianPath(&tsp))
525  return -1;
526 
527  blength = pathLength(&tsp);
528  if (blength < tsp.bestlen) {
529  tsp.bestlen = blength;
530  for (i = 0; i < tsp.n; i++) tsp.border[i] = tsp.iorder[i];
531  }

534 
535  annealing(&tsp);
536 
537  *total_len = pathLength(&tsp);
538  DBG("Final Path Length: %.4f", *total_len);
539 
540  *total_len = tsp.bestlen;
541  for (i=0; i<tsp.n; i++) tsp.iorder[i] = tsp.border[i];
542  DBG("Best Path Length: %.4f", *total_len);
543 
544  // reorder ids[] with start as first
545 
546 #ifdef DEBUG
547  for (i=0; i<tsp.n; i++) {
548  DBG("i: %d, ids[i]: %d, io[i]: %d, jo[i]: %d, jo[io[i]]: %d",
549  i, ids[i], tsp.iorder[i], tsp.jorder[i], tsp.jorder[tsp.iorder[i]]);
550  }
551 #endif
552 
553  // get index of start node in ids
554  for (i=0; i < tsp.n; i++) {
555  if (ids[i] == start) istart = i;
556  if (ids[i] == end) iend = i;
557  }
558  DBG("istart: %d, iend: %d", istart, iend);
559 
560  // get the idex of start in iorder
561  for (i=0; i < tsp.n; i++) {
562  if (tsp.iorder[i] == istart) jstart = i;
563  if (tsp.iorder[i] == iend) jend = i;
564  }
565  DBG("jstart: %d, jend: %d", jstart, jend);
566 
567  /*
568  * If the end is specified and the end point and it follow start
569  * then we swap start and end and extract the list backwards
570  * and later we reverse the list for the desired order.
571  */
572  if ((jend > 0 && jend == jstart+1) || (jend == 0 && jstart == tsp.n-1)) {
573  int tmp = jend;
574  jend = jstart;
575  jstart = tmp;
576  rev = 1;
577  DBG("reversed start and end: jstart: %d, jend: %d", jstart, jend);
578  }
579 
580  // copy ids to tsp.jorder so we can rewrite ids
581  memcpy(tsp.jorder, ids, tsp.n * sizeof(int));
582 
583  // write reordered ids into ids[]
584  // remember at this point jorder is our list if ids
585  for (i=jstart, j=0; i < tsp.n; i++, j++)
586  ids[j] = tsp.jorder[tsp.iorder[i]];
587 
588  for (i=0; i < jstart; i++, j++)
589  ids[j] =tsp.jorder[tsp.iorder[i]];
590 
591  // if we reversed the order above, now put it correct.
592  if (rev) {
593  int tmp = jend;
594  jend = jstart;
595  jstart = tmp;
596  reverse(tsp.n, ids);
597  }
598 
599 #ifdef DEBUG
600  DBG("ids getting returned!");
601  for (i=0; i<tsp.n; i++) {
602  DBG("i: %d, ids[i]: %d, io[i]: %d, jo[i]: %d",

604  }
605 #endif
606 
607  DBG("tsplib: jstart=%d, jend=%d, n=%d, j=%d", jstart, jend, tsp.n, j);
608 
609  return 0;
610 }
