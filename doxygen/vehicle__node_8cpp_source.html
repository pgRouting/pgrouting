36void
37Vehicle_node::evaluate(double cargoLimit) {
38 if (is_start()) {
39 /* time */
40 m_travel_time = 0;
41 m_arrival_time = opens();
42 m_wait_time = 0;
43 m_departure_time = arrival_time() + service_time();
44
45 /* time aggregates */
46 m_tot_travel_time = 0;
47 m_tot_wait_time = 0;
48 m_tot_service_time = service_time();
49
50 /* cargo aggregates */
51 m_cargo = demand();
52
53 /* violation aggregates */
54 m_twvTot = m_cvTot = 0;
55 m_cvTot = has_cv(cargoLimit) ? 1 : 0;
56 m_delta_time = 0;
57 }
58}

66void
67Vehicle_node::evaluate(
68 const Vehicle_node &pred,
69 double cargoLimit,
70 double speed) {
71 /* time */
72 m_travel_time = pred.travel_time_to(*this, speed);
73 m_arrival_time = pred.departure_time() + travel_time();
74 m_wait_time = is_early_arrival(arrival_time()) ?
75 opens() - m_arrival_time :
76 0;
77 m_departure_time = arrival_time() + wait_time() + service_time();
78
79 /* time aggregates */
80 m_tot_travel_time = pred.total_travel_time() + travel_time();
81 m_tot_wait_time = pred.total_wait_time() + wait_time();
82 m_tot_service_time = pred.total_service_time() + service_time();
83
84 /* cargo aggregates */
85 if (is_dump() && pred.cargo() >= 0) {
86 demand(-pred.cargo());
87 }
88 m_cargo = pred.cargo() + demand();
89
90 /* violations aggregates */
91
92 m_twvTot = has_twv() ? pred.twvTot() + 1 : pred.twvTot();
93 m_cvTot = has_cv(cargoLimit) ? pred.cvTot() + 1 : pred.cvTot();
94 m_delta_time = departure_time() - pred.departure_time();
95}
