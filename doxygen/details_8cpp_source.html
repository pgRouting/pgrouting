35std::vector<int64_t>
36clean_vids(std::vector<int64_t> vids) {
37 std::sort(vids.begin(), vids.end());
38 vids.erase(
39 std::unique(vids.begin(), vids.end()),
40 vids.end());
41 vids.erase(
42 std::remove(vids.begin(), vids.end(), 0),
43 vids.end());
44 return vids;
45}

47std::vector<MST_rt>
48get_no_edge_graph_result(const std::vector<int64_t> &vids) {
49 std::vector<MST_rt> results;
50 if (vids.empty()) return results;
51 for (auto const root : clean_vids(vids)) {
52 results.push_back({root, 0, root, root, -1, 0.0, 0.0});
53 }
54 return results;
55}
