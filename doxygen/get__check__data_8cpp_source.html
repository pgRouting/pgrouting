59bool
60get_column_info(const TupleDesc &tupdesc, pgrouting::Column_info_t &info) {
61 info.colNumber = SPI_fnumber(tupdesc, info.name.c_str());
62 if (info.strict && info.colNumber == SPI_ERROR_NOATTRIBUTE) {
63 throw std::string("Column '") + info.name + "' not Found";
64 }
65
66 if (info.colNumber != SPI_ERROR_NOATTRIBUTE) {
67 info.type = SPI_gettypeid(tupdesc, info.colNumber);
68 if (info.type == InvalidOid) {
69 throw std::string("Type of column '") + info.name + "' not Found";
70 }
71 return true;
72 }
73 return false;
74}

181void fetch_column_info(
182 const TupleDesc &tupdesc,
183 std::vector<pgrouting::Column_info_t> &info) {
184 for (auto &coldata : info) {
185 if (get_column_info(tupdesc, coldata)) {
186 switch (coldata.eType) {
187 case ANY_INTEGER:
188 check_any_integer_type(coldata);
189 break;
190 case ANY_NUMERICAL:
191 check_any_numerical_type(coldata);
192 break;
193 case TEXT:
194 check_text_type(coldata);
195 break;
196 case CHAR1:
197 check_char_type(coldata);
198 break;
199 case ANY_INTEGER_ARRAY:
200 check_any_integer_array_type(coldata);
201 break;
202 default:
203 throw std::string("Unexpected type of column ") + coldata.name;
204 }
205 }
206 }

224char getChar(
225 const HeapTuple tuple, const TupleDesc &tupdesc, const pgrouting::Column_info_t &info,
226 bool strict, char default_value) {
227 Datum binval;
228 bool isNull;
229 char value = default_value;
230
231 binval = SPI_getbinval(tuple, tupdesc, info.colNumber, &isNull);
232 if (!(info.type == BPCHAROID)) {
233 throw std::string("Unexpected Column type of ") + info.name + ". Expected CHAR";
234 }
235 if (!isNull) {
236 value = reinterpret_cast<char*>(binval)[1];
237 } else {
238 if (strict) {
239 throw std::string("Unexpected Null value in column ") + info.name;
240 }
241 value = default_value;
242 }
243 return value;
244}

266get_pgset(ArrayType *v) {
267 std::set<int64_t> results;
268
269 if (!v) return results;
270
271 auto element_type = ARR_ELEMTYPE(v);
272 auto dim = ARR_DIMS(v);
273 auto ndim = ARR_NDIM(v);
274 auto nitems = ArrayGetNItems(ndim, dim);
275 Datum *elements = nullptr;
276 bool *nulls = nullptr;
277 int16 typlen;
278 bool typbyval;
279 char typalign;
280
281
282 if (ndim == 0 || nitems <= 0) {
283 return results;
284 }
285
286 if (ndim != 1) {
287 throw std::string("One dimension expected");
288 }
289
290 get_typlenbyvalalign(element_type, &typlen, &typbyval, &typalign);
291
292 switch (element_type) {
293 case INT2OID:
294 case INT4OID:
295 case INT8OID:
296 break;
297 default:
298 throw std::string("Expected array of ANY-INTEGER");
299 }
300
301 deconstruct_array(v, element_type, typlen, typbyval,
302 typalign, &elements, &nulls,
303 &nitems);
304
305 int64_t data(0);
306
307 for (int i = 0; i < nitems; i++) {
308 if (nulls[i]) {
309 throw std::string("NULL value found in Array!");
310 } else {
311 switch (element_type) {
312 case INT2OID:
313 data = static_cast<int64_t>(DatumGetInt16(elements[i]));
314 break;
315 case INT4OID:
316 data = static_cast<int64_t>(DatumGetInt32(elements[i]));
317 break;
318 case INT8OID:
319 data = DatumGetInt64(elements[i]);
320 break;
321 }
322 }
323 results.insert(data);
324 }
325
326 pfree(elements);
327 pfree(nulls);
328 return results;

444int64_t*
445get_array(ArrayType *v, size_t *arrlen, bool allow_empty) {
446 int64_t *c_array = nullptr;
447
448 auto element_type = ARR_ELEMTYPE(v);
449 auto dim = ARR_DIMS(v);
450 auto ndim = ARR_NDIM(v);
451 auto nitems = ArrayGetNItems(ndim, dim);
452 Datum *elements = nullptr;
453 bool *nulls = nullptr;
454 int16 typlen;
455 bool typbyval;
456 char typalign;
457
458
459 if (allow_empty && (ndim == 0 || nitems <= 0)) {
460 return nullptr;
461 }
462
463 if (ndim != 1) {
464 throw std::string("One dimension expected");
465 }
466
467 if (nitems <= 0) {
468 throw std::string("No elements found");
469 }
470
471 get_typlenbyvalalign(element_type, &typlen, &typbyval, &typalign);
472
473 /* validate input data type */
474 switch (element_type) {
475 case INT2OID:
476 case INT4OID:
477 case INT8OID:
478 break;
479 default:
480 throw std::string("Expected array of ANY-INTEGER");
481 return nullptr;
482 }
483
484 deconstruct_array(v, element_type, typlen, typbyval,
485 typalign, &elements, &nulls,
486 &nitems);
487
488 c_array = pgr_alloc(static_cast<size_t>(nitems), (c_array));
489 if (!c_array) {
490 throw std::string("Out of memory!");
491 }
492
493
494 for (int i = 0; i < nitems; i++) {
495 if (nulls[i]) {
496 pfree(c_array);
497 throw std::string("NULL value found in Array!");
498 return nullptr;
499 } else {
500 switch (element_type) {
501 case INT2OID:
502 c_array[i] = static_cast<int64_t>(DatumGetInt16(elements[i]));
503 break;
504 case INT4OID:
505 c_array[i] = static_cast<int64_t>(DatumGetInt32(elements[i]));
506 break;
507 case INT8OID:
508 c_array[i] = DatumGetInt64(elements[i]);
509 break;
510 }
511 }
512 }
513 (*arrlen) = static_cast<size_t>(nitems);
514
515 pfree(elements);
516 pfree(nulls);
517 return c_array;
518}

590double getFloat8(
591 const HeapTuple tuple, const TupleDesc &tupdesc, const pgrouting::Column_info_t &info) {
592 Datum binval;
593 bool isnull = false;
594 binval = SPI_getbinval(tuple, tupdesc, info.colNumber, &isnull);
595 if (isnull) {
596 throw std::string("Unexpected Null value in column ") + info.name;
597 }
598
599 switch (info.type) {
600 case INT2OID:
601 return static_cast<double>(DatumGetInt16(binval));
602 break;
603 case INT4OID:
604 return static_cast<double>(DatumGetInt32(binval));
605 break;
606 case INT8OID:
607 return static_cast<double>(DatumGetInt64(binval));
608 break;
609 case FLOAT4OID:
610 return static_cast<double>(DatumGetFloat4(binval));
611 break;
612 case FLOAT8OID:
613 return static_cast<double>(DatumGetFloat8(binval));
614 break;
615 case NUMERICOID:
616 /* Note: out-of-range values will be clamped to +-HUGE_VAL */
617 return static_cast<double>(DatumGetFloat8(DirectFunctionCall1(numeric_float8_no_overflow, binval)));
618 break;
619 default:
620 throw std::string("Unexpected Column type of ") + info.name + ". Expected ANY-NUMERICAL";
621 }
622 return 0.0;
623}
