56template <class G>
57size_t
58count_rows(
59 const G &graph,
60 const std::vector<std::vector<double>> &matrix) {
61 int64_t count = 0;
62 for (size_t i = 0; i < graph.num_vertices(); i++) {
63 count += std::count_if(
64 matrix[i].begin(), matrix[i].end(),
65 [](double value) {
66 return value != (std::numeric_limits<double>::max)();
67 });
68 }
69 return static_cast<size_t>(count) - graph.num_vertices();
70}

85template <class G>
86void vector_to_tuple(
87 const G &graph,
88 const std::vector<double> results,
89 size_t &result_count,
90 IID_t_rt **result_tuples) {
91 result_count = results.size();
92 *result_tuples = pgrouting::pgr_alloc(result_count, (*result_tuples));
93
94 size_t seq = 0;
95 for (typename G::V v_i = 0; v_i < graph.num_vertices(); ++v_i) {
96 (*result_tuples)[seq].from_vid = graph[v_i].id;
97 /*
98 * These 2 lines are specifically for pgr_betweennessCentrality
99 */
100 (*result_tuples)[seq].to_vid = 0;
101 (*result_tuples)[seq].cost = graph.is_directed()? results[v_i] / 2.0 : results[v_i];
102 seq++;
103 }
104}

119template <class G>
120void matrix_to_tuple(
121 const G &graph,
122 const std::vector<std::vector<double>> &matrix,
123 size_t &result_tuple_count,
124 IID_t_rt **postgres_rows) {
125 result_tuple_count = detail::count_rows(graph, matrix);
126 *postgres_rows = pgr_alloc(result_tuple_count, (*postgres_rows));
127
128 size_t seq = 0;
129 for (typename G::V v_i = 0; v_i < graph.num_vertices(); v_i++) {
130 for (typename G::V v_j = 0; v_j < graph.num_vertices(); v_j++) {
131 if (v_i == v_j) continue;
132 if (matrix[v_i][v_j] != (std::numeric_limits<double>::max)()) {
133 (*postgres_rows)[seq].from_vid = graph[v_i].id;
134 (*postgres_rows)[seq].to_vid = graph[v_j].id;
135 (*postgres_rows)[seq].cost = matrix[v_i][v_j];
136 seq++;
137 }
138 }
139 }

149template <class G>
150void graph_to_tuple(
151 G &graph,
152 contractionHierarchies_rt **return_tuples,
153 size_t *count) {
154 using pgrouting::pgr_alloc;
155 auto modified_vertices(graph.get_modified_vertices());
156 auto shortcut_edges(graph.get_shortcuts());
157
158 (*count) = modified_vertices.size() + shortcut_edges.size();
159 (*return_tuples) = pgr_alloc((*count), (*return_tuples));
160 size_t sequence = 0;
161
162 for (const auto id : modified_vertices) {
163 auto v = graph.get_V(id);
164 int64_t* contracted_vertices = NULL;
165 auto vids = graph[v].contracted_vertices();
166 int64_t o = graph.get_vertex_order(id);
167 int64_t m = graph.get_vertex_metric(id);
168
169 contracted_vertices = pgr_alloc(vids.size(), contracted_vertices);
170
171 int count = 0;
172 for (const auto id : vids) {
173 contracted_vertices[count++] = id;
174 }
175 (*return_tuples)[sequence] = {
176 id,
177 const_cast<char*>("v"),
178 -1, -1, -1.00,
179 contracted_vertices,
180 count,
181 o, m};
182
183 ++sequence;
184 }
185
186 int64_t eid = 0;
187 for (auto e : shortcut_edges) {
188 auto edge = graph[e];
189 int64_t* contracted_vertices = NULL;
190 const auto vids(edge.contracted_vertices());
191 pgassert(!vids.empty());
192
193 contracted_vertices = pgr_alloc(vids.size(), contracted_vertices);
194 int count = 0;
195 for (const auto vid : vids) {
196 contracted_vertices[count++] = vid;
197 }
198 (*return_tuples)[sequence] = {
199 --eid,
200 const_cast<char*>("e"),
201 edge.source, edge.target, edge.cost,
202 contracted_vertices, count, -1, -1};
203 ++sequence;
204 }
205}
