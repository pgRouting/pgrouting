134 std::deque<Path> Yen(G &graph,
135 int64_t start_vertex,
136 int64_t end_vertex,
137 size_t K) {
138 /*
139 * No path: already in destination
140 */
141 if (start_vertex == end_vertex) {
142 return std::deque<Path>();
143 }
144
145 /*
146 * no path: disconnected vertices
147 */
148 if (!graph.has_vertex(start_vertex)
149 || !graph.has_vertex(end_vertex)) {
150 return std::deque<Path>();
151 }
152
153 /*
154 * clean up containers
155 */
156 clear();
157
158 this->v_source = graph.get_V(start_vertex);
159 this->v_target = graph.get_V(end_vertex);
160 this->m_start = start_vertex;
161 this->m_end = end_vertex;
162 this->m_K = K;
163 Pgr_ksp<G>::m_heap_paths = true;
164 delete this->m_vis;
165 this->m_vis = new Myvisitor(
166 m_solutions,
167 m_restrictions,
168 m_stop_on_first);
169
170
171
172 try {
173 Pgr_ksp< G >::executeYen(graph);
174 } catch(found_goals &) {
175 pgassert(!m_solutions.empty());
176 std::deque<Path> solutions(m_solutions.begin(), m_solutions.end());
177 return solutions;
178 } catch (boost::exception const& ex) {
179 (void)ex;
180 throw;
181 } catch (std::exception &e) {
182 (void)e;
183 throw;
184 } catch (...) {
185 throw;
186 }
187
188
189 if (!m_solutions.empty()) {
190 std::deque<Path> solutions(m_solutions.begin(), m_solutions.end());
191 return solutions;
192 }
193
194 auto solutions = Pgr_ksp<G>::get_results();
195
196 return get_results(solutions);
197 }

200 std::deque<Path> get_results(
201 std::deque<Path> &paths) {
202 if (paths.empty()) return paths;
203 if (m_strict) return std::deque<Path>();
204
205 paths = inf_cost_on_restriction(paths);
206
207 std::stable_sort(paths.begin(), paths.end(),
208 [](const Path &left, const Path &right) -> bool {
209 return (left.countInfinityCost() < right.countInfinityCost());
210 });
211
212 auto count = paths.begin()->countInfinityCost();
213
214
215 if (m_heap_paths) return paths;
216
217 paths.erase(
218 std::remove_if(
219 paths.begin(), paths.end(),
220 [&count](const Path &p){
221 return count != p.countInfinityCost();
222 }),
223 paths.end());
224
225 return paths;
226 }
