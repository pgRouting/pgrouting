55 Pgr_contract(
56 G &graph,
57 Identifiers<V> forbidden_vertices,
58 std::vector<int64_t> contraction_order,
59 int64_t max_cycles
60 ) {
61 std::deque<int64_t> contract_order;
62 // push -1 to indicate the start of the queue
63 contract_order.push_back(-1);
64 contract_order.insert(
65 contract_order.end(),
66 contraction_order.begin(), contraction_order.end());
67 for (int64_t i = 0; i < max_cycles; ++i) {
68 int64_t front = contract_order.front();
69 contract_order.pop_front();
70 contract_order.push_back(front);
71 auto kind = contract_order.front();
72 while (kind != -1) {
73 one_cycle(graph, kind, forbidden_vertices);
74 contract_order.pop_front();
75 contract_order.push_back(front);
76 kind = contract_order.front();
77 }
78 }
79 }
