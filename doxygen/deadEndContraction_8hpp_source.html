64 void doContraction(G &graph) {
65 calculateVertices(graph);
66
67 while (!deadendVertices.empty()) {
68 V v = deadendVertices.front();
69 deadendVertices -= v;
70 pgassert(graph.is_dead_end(v));
71
72 Identifiers<V> local;
73 for (auto u : graph.find_adjacent_vertices(v)) {
74 /*
75 * u{v1} --{v2}-> v{v3}
76 *
77 * u{v1 + v + v2 + v3} v{}
78 */
79 const auto& e = graph.get_min_cost_edge(u, v);
80 graph[u].contracted_vertices() += std::get<0>(e).contracted_vertices();
81 graph[u].add_contracted_vertex(graph[v]);
82
83 deadendVertices -= v;
84 local += u;
85 }
86
87 graph[v].contracted_vertices().clear();
88 boost::clear_vertex(v, graph.graph);
89
90 /* abort in case of an interruption occurs (e.g. the query is being cancelled) */
91 CHECK_FOR_INTERRUPTS();
92
93 for (const auto u : local) {
94 if (graph.is_dead_end(u) && !graph.is_forbidden(u)) {
95 deadendVertices += u;
96 } else {
97 deadendVertices -= u;
98 }
99 }
100 }
101 }
