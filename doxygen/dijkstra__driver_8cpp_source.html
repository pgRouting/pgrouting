55post_process(std::deque<pgrouting::Path> &paths, bool only_cost, bool normal, size_t n_goals, bool global) {
56 using pgrouting::Path;
57 paths.erase(std::remove_if(paths.begin(), paths.end(),
58 [](const Path &p) {
59 return p.size() == 0;
60 }),
61 paths.end());
62 using difference_type = std::deque<double>::difference_type;
63
64 if (!normal) {
65 for (auto &path : paths) path.reverse();
66 }
67
68 if (!only_cost) {
69 for (auto &p : paths) {
70 p.recalculate_agg_cost();
71 }
72 }
73
74 if (n_goals != (std::numeric_limits<size_t>::max)()) {
75 std::sort(paths.begin(), paths.end(),
76 [](const Path &e1, const Path &e2)->bool {
77 return e1.end_id() < e2.end_id();
78 });
79 std::stable_sort(paths.begin(), paths.end(),
80 [](const Path &e1, const Path &e2)->bool {
81 return e1.start_id() < e2.start_id();
82 });
83 std::stable_sort(paths.begin(), paths.end(),
84 [](const Path &e1, const Path &e2)->bool {
85 return e1.tot_cost() < e2.tot_cost();
86 });
87 if (global && n_goals < paths.size()) {
88 paths.erase(paths.begin() + static_cast<difference_type>(n_goals), paths.end());
89 }
90 } else {
91 std::sort(paths.begin(), paths.end(),
92 [](const Path &e1, const Path &e2)->bool {
93 return e1.end_id() < e2.end_id();
94 });
95 std::stable_sort(paths.begin(), paths.end(),
96 [](const Path &e1, const Path &e2)->bool {
97 return e1.start_id() < e2.start_id();
98 });
99 }
