47std::vector<II_t_rt>
48pgr_connectedComponents(pgrouting::UndirectedGraph &graph) {
49 typedef pgrouting::UndirectedGraph::V V;
50 // perform the algorithm
51 std::vector<V> components(num_vertices(graph.graph));
52 size_t num_comps;
53 /* abort in case of an interruption occurs (e.g. the query is being cancelled) */
54 CHECK_FOR_INTERRUPTS();
55 try {
56 num_comps = boost::connected_components(graph.graph, &components[0]);
57 } catch (...) {
58 throw;
59 }
60
61 // get the results
62 std::vector< std::vector< int64_t > > results(num_comps);
63 for (auto vd : boost::make_iterator_range(vertices(graph.graph))) {
64 results[components[vd]].push_back(graph[vd].id);
65 }
66
67 return detail::componentsResult(results);
68}

102biconnectedComponents(
103 pgrouting::UndirectedGraph &graph) {
104 using G = pgrouting::UndirectedGraph;
105 using E = G::E;
106 using Edge_map = std::map< E, size_t >;
107
108 // perform the algorithm
109 Edge_map bicmp_map;
110 boost::associative_property_map<Edge_map> bimap(bicmp_map);
111 size_t num_comps;
112 try {
113 // NOLINTNEXTLINE(clang-analyzer-cplusplus.NewDelete)
114 num_comps = biconnected_components(graph.graph, bimap);
115 } catch (...) {
116 throw;
117 }
118
119 std::vector< std::vector< int64_t > > results(num_comps);
120 for (auto ed : boost::make_iterator_range(edges(graph.graph))) {
121 results[bimap[ed]].push_back(graph[ed].id);
122 }
123
124 return detail::componentsResult(results);

163Identifiers<int64_t>
164bridges(pgrouting::UndirectedGraph &graph) {
165 using G = pgrouting::UndirectedGraph;
166 using V = G::V;
167 using EO_i = G::EO_i;
168
169 Identifiers<int64_t> bridge_edges;
170 Identifiers<int64_t> processed_edges;
171 std::vector<V> components(num_vertices(graph.graph));
172 size_t ini_comps;
173 /* abort in case of an interruption occurs (e.g. the query is being cancelled) */
174 CHECK_FOR_INTERRUPTS();
175 try {
176 ini_comps = boost::connected_components(graph.graph, &components[0]);
177 } catch (...) {
178 throw;
179 }
180 /* abort in case of an interruption occurs (e.g. the query is being cancelled) */
181 CHECK_FOR_INTERRUPTS();
182 std::vector<V> art_points;
183 try {
184 boost::articulation_points(graph.graph, std::back_inserter(art_points));
185 } catch (...) {
186 throw;
187 }
188
189 for (auto v : boost::make_iterator_range(vertices(graph.graph))) {
190 if (graph.out_degree(v) == 1) {
191 art_points.push_back(v);
192 }
193 }
194
195 for (const auto u : art_points) {
196 for (const auto v : art_points) {
197 /*
198 * skip when the vertices are the same and do half the work
199 */
200 if (u < v) continue;
201 auto p = boost::edge(u, v, graph.graph);
202
203 /*
204 * skip when there is no edge (u, v) on the graph
205 */
206 if (!p.second) continue;
207 auto edge = p.first;
208 auto id = graph[edge].id;
209
210 /*
211 * Skip when the edge has already being processed
212 */
213 if (processed_edges.has(id)) continue;
214
215 /*
216 * Processing edge
217 */
218 processed_edges += id;
219
220
221 /*
222 * At least one edge between articulation points u & v
223 */
224 int parallel_count = 0;
225 EO_i ei, ei_end;
226 boost::tie(ei, ei_end) = out_edges(u, graph.graph);
227
228 for ( ; ei != ei_end; ++ei) {
229 if (target(*ei, graph.graph) == v) ++parallel_count;
230 }
231
232 if (parallel_count == 1) {
233 // TODO(vicky) filter graph instead of removing edges
234 size_t now_comps;
235 try {
236 boost::remove_edge(edge, graph.graph);
237
238 now_comps = boost::connected_components(graph.graph, &components[0]);
239
240 boost::add_edge(boost::source(edge, graph.graph),
241 boost::target(edge, graph.graph),
242 graph.graph);
243 } catch (...) {
244 throw;
245 }
246
247 if (now_comps > ini_comps) {
248 bridge_edges += id;
249 }
250 }
251 }
252 }
253
254 return bridge_edges;
255}
