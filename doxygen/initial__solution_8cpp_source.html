38Initial_solution::invariant() const {

45Initial_solution::Initial_solution(
46 Initials_code kind,
47 size_t number_of_orders) :
48 Solution(),
49 m_all_orders(number_of_orders),
50 m_unassigned(number_of_orders),
51 m_assigned() {
52 invariant();
53 pgassert(kind >= 0 && kind <= OneDepot);
54
55 switch (kind) {
56 case OneTruck:
57 one_truck_all_orders();
58 break;
59 case OnePerTruck:
60 case FrontTruck:
61 case BackTruck:
62 case BestInsert:
63 case BestBack:
64 case BestFront:
65 case OneDepot:
66 do_while_foo(kind);
67 break;
68 default: pgassert(false);
69 }
70
71 invariant();
72}

76void
77Initial_solution::do_while_foo(int kind) {
78 invariant();
79 pgassert(kind > 0 && kind <= OneDepot);
80
81 Identifiers<size_t> notused;
82
83 while (!m_unassigned.empty()) {
84 auto current = m_unassigned.size();
85 auto truck = trucks.get_truck(m_unassigned.front());
86 /*
87 * kind 1 to 7 work with the same code structure
88 */
89 truck.do_while_feasable((Initials_code)kind, m_unassigned, m_assigned);
90 pgassertwm(current > m_unassigned.size(), msg().get_log().c_str());
91
92 fleet.push_back(truck);
93 invariant();
94 }
95
96 pgassertwm(true, msg().get_log().c_str());
97 pgassert(is_feasable());
98 invariant();
99}

104void
105Initial_solution::one_truck_all_orders() {
106 invariant();
107 msg().log << "\nInitial_solution::one_truck_all_orders\n";
108 auto truck = trucks.get_truck();
109 while (!m_unassigned.empty()) {
110 auto order(truck.orders()[*m_unassigned.begin()]);
111
112 truck.insert(order);
113
114 m_assigned += m_unassigned.front();
115 m_unassigned.pop_front();
116
117 invariant();
118 }
119 fleet.push_back(truck);
120 invariant();
121}
