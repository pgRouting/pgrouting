102/*
103 * Initializing the set of nodes that can be placed
104 * immediately after \bthis node
105 *
106 * (*this) -> J
107 *
108 */
109void
110Order::set_compatibles(const Order J, double speed) {
111 if (J.idx() == idx()) return;
112 if (J.isCompatibleIJ(*this, speed)) {
113 /*
114 * this -> {J}
115 */
116 m_compatibleJ += J.idx();
117 }
118 if (this->isCompatibleIJ(J, speed)) {
119 /*
120 * {J} -> this
121 */
122 m_compatibleI += J.idx();
123 }
124}

127/*
128 * True when
129 *
130 * I -> (*this)
131 *
132 */
133bool
134Order::isCompatibleIJ(const Order &I, double speed) const {
135 /* this is true in all cases */
136 auto all_cases(
137 pickup().is_compatible_IJ(I.pickup(), speed)
138 && delivery().is_compatible_IJ(I.pickup(), speed));
139
140 /* case other(P) other(D) this(P) this(D) */
141 auto case1(pickup().is_compatible_IJ(I.delivery(), speed)
142 && delivery().is_compatible_IJ(I.delivery(), speed));
143
144 /* case other(P) this(P) other(D) this(D) */
145 auto case2(I.delivery().is_compatible_IJ(pickup(), speed)
146 && delivery().is_compatible_IJ(I.delivery(), speed));
147
148 /* case other(P) this(P) this(D) other(D) */
149 auto case3(I.delivery().is_compatible_IJ(pickup(), speed)
150 && I.delivery().is_compatible_IJ(delivery(), speed));
151
152 return all_cases && (case1 || case2 || case3);
153}
