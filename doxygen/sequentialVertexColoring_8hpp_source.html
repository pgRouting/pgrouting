51class Pgr_sequentialVertexColoring {
52 public:
53 typedef typename G::V V;
54 typedef typename G::E E;
55 typedef boost::adjacency_list < boost::listS, boost::vecS, boost::undirectedS > Graph;
56 typedef boost::graph_traits < Graph > ::vertices_size_type vertices_size_type;

74 std::vector <II_t_rt> sequentialVertexColoring(G &graph) {
75 std::vector <II_t_rt> results;
76
77 auto i_map = boost::get(boost::vertex_index, graph.graph);
78
79 // vector which will store the color of all the vertices in the graph
80 std::vector < vertices_size_type > colors(boost::num_vertices(graph.graph));
81
82 // An iterator property map which records the color of each vertex
83 auto color_map = boost::make_iterator_property_map(colors.begin(), i_map);
84
85 /* abort in case of an interruption occurs (e.g. the query is being cancelled) */
86 CHECK_FOR_INTERRUPTS();
87
88 try {
89 boost::sequential_vertex_coloring(graph.graph, color_map);
90 } catch (boost::exception const& ex) {
91 (void)ex;
92 throw;
93 } catch (std::exception &e) {
94 (void)e;
95 throw;
96 } catch (...) {
97 throw;
98 }
99
100 results = get_results(colors, graph);
101
102 return results;
103 }
104
106
107 private:
117 std::vector <II_t_rt> get_results(
118 std::vector < vertices_size_type > &colors,
119 const G &graph) {
120 std::vector <II_t_rt> results;
121
122 typename boost::graph_traits < Graph > ::vertex_iterator v, vend;
123
124 for (boost::tie(v, vend) = vertices(graph.graph); v != vend; ++v) {
125 int64_t node = graph[*v].id;
126 auto color = colors[*v];
127 results.push_back({{node}, {static_cast<int64_t>(color + 1)}});
128 }
129
130 // ordering the results in an increasing order of the node id
131 std::sort(results.begin(), results.end(),
132 [](const II_t_rt row1, const II_t_rt row2) {
133 return row1.d1.id < row2.d1.id;
134 });
135
136 return results;
137 }
138};
