56std::deque<std::pair<int64_t, double>>
57start_vid_end_vid_are_fixed(
58 std::deque<std::pair<int64_t, double>> tsp_path,
59 int64_t start_vid,
60 int64_t end_vid) {
61 pgassert(tsp_path[0].first == start_vid);
62 /*
63 * Where is 0?
64 */
65 auto where = std::find_if(tsp_path.begin(), tsp_path.end(),
66 [&](std::pair<int64_t, double>& row){return row.first == 0;});
67
68 /*
69 * Move the 0 to be the second value
70 * 5,2,1,8,7,10,11,16,6,9,12,17,15,13,14,0,3,4,5
71 * to
72 * 5 0 3 4 2 1 8 7 10 11 16 6 9 12 17 15 13 14 5
73 */
74 std::rotate(tsp_path.begin() + 1, where, tsp_path.end() - 1);
75
76 /*
77 * delete the 0
78 * to
79 * 5 3 4 2 1 8 7 10 11 16 6 9 12 17 15 13 14 5
80 */
81 tsp_path.erase(tsp_path.begin() + 1);
82
83 /*
84 * when the second value is not the end_vid, reverse the path
85 * - it is the second to last, like the 14 above
86 */
87 if (tsp_path[0].first != end_vid) std::reverse(tsp_path.begin(), tsp_path.end());
88 return tsp_path;
89}

91double
92get_min_cost(
93 V u,
94 V v,
95 TSP_graph &graph) {
96 auto the_edge = boost::edge(u, v, graph.graph());
97 /*
98 * The edge exists, then return the weight it has
99 */
100 if (the_edge.second) {
101 return get(boost::edge_weight, graph.graph())[the_edge.first];
102 }
103
104 std::vector<V> predecessors(num_vertices(graph.graph()));
105 std::vector<double> distances(num_vertices(graph.graph()), std::numeric_limits<double>::infinity());
106 bool found = false;
107
108 CHECK_FOR_INTERRUPTS();
109 try {
110 boost::dijkstra_shortest_paths(
111 graph.graph(), u,
112 boost::predecessor_map(&predecessors[0])
113 .distance_map(&distances[0])
114 .visitor(pgrouting::visitors::dijkstra_one_goal_visitor<V>(v)));
115 } catch(pgrouting::found_goals &) {
116 found = true;
117 }
118 if (!found) {
119 pgassert(false);
120 throw std::make_pair(std::string("INTERNAL: graph is incomplete 1"), std::string("Check graph before calling"));
121 }
122
123 double agg_cost(0);
124 while (v != u) {
125 agg_cost += distances[predecessors[v]];
126 v = predecessors[v];
127 }
128 return agg_cost;
129}

131std::deque<std::pair<int64_t, double>>
132eval_tour(TSP_graph& graph, const std::vector<V> &tsp_tour) {
133 std::deque<std::pair<int64_t, double>> results;
134 auto u = graph.graph().null_vertex();
135 for (auto v : tsp_tour) {
136 auto cost = (u == graph.graph().null_vertex()) ?
137 0 :
138 get_min_cost(u, v, graph);
139 u = v;
140 results.push_back(std::make_pair(graph.get_vertex_id(v), cost));
141 }
142 return results;
143}

145double
146eval_tour(TSP_graph& graph, std::deque<std::pair<int64_t, double>>& tsp_tour) {
147 std::deque<std::pair<int64_t, double>> results;
148 auto u = graph.graph().null_vertex();
149 double agg_cost(0);
150 for (auto &node : tsp_tour) {
151 auto v = graph.get_boost_vertex(node.first);
152 auto cost = (u == graph.graph().null_vertex()) ?
153 0 :
154 get_min_cost(u, v, graph);
155 u = v;
156 node.second = cost;
157 agg_cost += cost;
158 }
159 return agg_cost;
160}

163std::deque<std::pair<int64_t, double>>
164tsp(TSP_graph& graph) {
165 std::vector<V> tsp_path;
166
167 CHECK_FOR_INTERRUPTS();
168 try {
169 boost::metric_tsp_approx_tour(
170 graph.graph(),
171 back_inserter(tsp_path));
172 } catch (...) {
173 throw std::make_pair(
174 std::string("INTERNAL: something went wrong while calling boost::metric_tsp_approx_tour"),
175 std::string(__PGR_PRETTY_FUNCTION__));
176 }
177
178 return eval_tour(graph, tsp_path);
179}

181std::deque<std::pair<int64_t, double>>
182tsp(TSP_graph& graph, int64_t start_vid) {
183 std::vector<V> tsp_path;
184 /*
185 * check that the start_vid
186 */
187 if (!graph.has_vertex(start_vid)) {
188 throw std::make_pair(
189 std::string("INTERNAL: Verify start_vid before calling"),
190 std::string(__PGR_PRETTY_FUNCTION__));
191 }
192
193 auto v = graph.get_boost_vertex(start_vid);
194
195 CHECK_FOR_INTERRUPTS();
196 try {
197 boost::metric_tsp_approx_tour_from_vertex(
198 graph.graph(),
199 v,
200 back_inserter(tsp_path));
201 } catch (...) {
202 throw std::make_pair(
203 std::string("INTERNAL: something went wrong while calling boost::metric_tsp_approx_tour_from_vertex"),
204 std::string(__PGR_PRETTY_FUNCTION__));
205 }
206
207 return eval_tour(graph, tsp_path);
208}

215std::deque<std::pair<int64_t, double>>
216crossover_optimize(TSP_graph& graph, std::deque<std::pair<int64_t, double>> result, size_t limit) {
217 auto best_cost = eval_tour(graph, result);
218 for (size_t i = 1; i < result.size() - limit; ++i) {
219 for (size_t j = result.size() - limit; j >= i + 1; --j) {
220 auto tmp = result;
221 std::reverse(tmp.begin() + static_cast<ptrdiff_t>(i), tmp.begin() + static_cast<ptrdiff_t>(j));
222 auto new_cost = eval_tour(graph, tmp);
223 if (new_cost < best_cost) {
224 result = tmp;
225 best_cost = new_cost;
226 }
227 }
228 }
229 return result;
230}

238std::deque<std::pair<int64_t, double>>
239do_tsp(
240 TSP_graph& graph,
241 int64_t start_vid,
242 int64_t end_vid) {
243 /*
244 * check that the start_vid, and end_vid exist on the data
245 */
246 if (start_vid == 0) std::swap(start_vid, end_vid);
247
248 if (start_vid != 0 && !graph.has_vertex(start_vid)) {
249 throw std::make_pair(
250 std::string("INTERNAL: start_id not found on data"),
251 std::string(__PGR_PRETTY_FUNCTION__));
252 }
253
254 if (end_vid !=0 && !graph.has_vertex(end_vid)) {
255 throw std::make_pair(
256 std::string("INTERNAL: end_id not found on data"),
257 std::string(__PGR_PRETTY_FUNCTION__));
258 }
259
260 std::deque<std::pair<int64_t, double>> result;
261
262 /*
263 * just get a tsp result
264 */
265 if (start_vid == 0) {
266 return crossover_optimize(graph, tsp(graph), 1);
267 }
268
269 /*
270 * the has a start value same as end_vid ignore end_vid
271 */
272 if ((end_vid == 0) || (start_vid == end_vid)) {
273 return crossover_optimize(graph, tsp(graph, start_vid), 1);
274 }
275
276 /*
277 * have start and end, a dummy node is needed
278 */
279 auto u = graph.get_boost_vertex(start_vid);
280 auto v = graph.get_boost_vertex(end_vid);
281
282 graph.insert_vertex(0);
283 auto dummy_node = graph.get_boost_vertex(0);
284 boost::add_edge(u, dummy_node, 0, graph.graph());
285 boost::add_edge(v, dummy_node, 0, graph.graph());
286
287 for (auto v1 : boost::make_iterator_range(boost::vertices(graph.graph()))) {
288 if (v1 == u || v1 == v) continue;
289 boost::add_edge(v1, dummy_node, std::numeric_limits<double>::infinity(), graph.graph());
290 }
291
292 result = tsp(graph, start_vid);
293 result = start_vid_end_vid_are_fixed(result, start_vid, end_vid);
294
295 return crossover_optimize(graph, result, 2);
296}
