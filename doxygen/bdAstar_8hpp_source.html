101 void explore_forward(const Cost_Vertex_pair &node) {
102 typename G::EO_i out, out_end;
103
104 auto current_node = node.second;
105 auto current_node_cost = forward_cost[current_node];
106
107 for (boost::tie(out, out_end) = out_edges(current_node, graph.graph);
108 out != out_end; ++out) {
109 auto edge_cost = graph[*out].cost;
110 auto next_node = graph.adjacent(current_node, *out);
111
112 if (forward_finished[next_node]) continue;
113
114 if (edge_cost + current_node_cost < forward_cost[next_node]) {
115 forward_cost[next_node] = edge_cost + current_node_cost;
116 forward_predecessor[next_node] = current_node;
117 forward_edge[next_node] = graph[*out].id;
118 forward_queue.push({
119 forward_cost[next_node]
120 + heuristic(next_node, v_target),
121 next_node});
122 }
123 }
124 forward_finished[current_node] = true;
125 }
126
127 void explore_backward(const Cost_Vertex_pair &node) {
128 typename G::EI_i in, in_end;
129
130 auto current_cost = node.first;
131 auto current_node = node.second;
132
133 for (boost::tie(in, in_end) = in_edges(current_node, graph.graph);
134 in != in_end; ++in) {
135 auto edge_cost = graph[*in].cost;
136 auto next_node = graph.adjacent(current_node, *in);
137
138 if (backward_finished[next_node]) continue;
139
140 if (edge_cost + current_cost < backward_cost[next_node]) {
141 backward_cost[next_node] = edge_cost + current_cost;
142 backward_predecessor[next_node] = current_node;
143 backward_edge[next_node] = graph[*in].id;
144 backward_queue.push({
145 backward_cost[next_node]
146 + heuristic(next_node, v_source),
147 next_node});
148 }
149 }
150 backward_finished[current_node] = true;
151 }
152
153
154 double heuristic(V v, V u) {
155 if (m_heuristic == 0) return 0;
156
157 double dx = graph[v].x() - graph[u].x();
158 double dy = graph[v].y() - graph[u].y();
159 double current;
160
161 switch (m_heuristic) {
162 case 0:
163 current = 0;
164 break;
165 case 1:
166 current = std::fabs((std::max)(dx, dy)) * m_factor;
167 break;
168 case 2:
169 current = std::fabs((std::min)(dx, dy)) * m_factor;
170 break;
171 case 3:
172 current = (dx * dx + dy * dy) * m_factor * m_factor;
173 break;
174 case 4:
175 current = std::sqrt(dx * dx + dy * dy) * m_factor;
176 break;
177 case 5:
178 current = (std::fabs(dx) + std::fabs(dy)) * m_factor;
179 break;
180 default:
181 current = 0;
182 }
183 return current;
184 }

196template <class G>
197std::deque<Path> bdastar(
198 G &graph,
199 const std::map<int64_t, std::set<int64_t>> &combinations,
200 int heuristic,
201 double factor,
202 double epsilon,
203 bool only_cost) {
204 std::deque<Path> paths;
205 pgrouting::bidirectional::Pgr_bdAstar<G> fn_bdAstar(graph);
206
207 for (const auto &c : combinations) {
208 if (!graph.has_vertex(c.first)) continue;
209
210 for (const auto &destination : c.second) {
211 if (!graph.has_vertex(destination)) continue;
212
213 fn_bdAstar.clear();
214
215 paths.push_back(fn_bdAstar.pgr_bdAstar(
216 graph.get_V(c.first), graph.get_V(destination),
217 heuristic, factor, epsilon, only_cost));
218 }
219 }
220
221 return paths;
222}
