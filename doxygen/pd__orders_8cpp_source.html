62void
63PD_Orders::build_orders(
64 const std::vector<Orders_t> &pd_orders
65 ) {
66 for (const auto &order : pd_orders) {
67 /*
68 * SAMPLE CORRECT INFORMATION
69 *
70 * id | demand | pick_x | pick_y | pick_open_t | pick_close_t | pick_service_t | deliver_x | deliver_y | deliver_open_t | deliver_open_t | deliver_close_t | deliver_service_t
71 * 1 | 10 | 35 | 69 | 448 | 505 | 90 | 45 | 68 | 912 | 967 | 90 | 35
72 */
73
74 /*
75 * matrix version
76 */
77
78 if (!problem->get_cost_matrix().has_id(order.pick_node_id)) {
79 throw std::make_pair(std::string("Unable to find node on matrix"), order.pick_node_id);
80 }
81 if (!problem->get_cost_matrix().has_id(order.deliver_node_id)) {
82 throw std::make_pair(std::string("Unable to find node on matrix"), order.deliver_node_id);
83 }
84
85 Vehicle_node pickup({problem->get_nodes().size(), order, Tw_node::NodeType::kPickup});
86 problem->add_node(pickup);
87 Vehicle_node delivery({problem->get_nodes().size(), order, Tw_node::NodeType::kDelivery});
88 problem->add_node(delivery);
89
90 add_order(order, pickup, delivery);
91 } // for (creating orders)
92}

116size_t
117PD_Orders::find_best_J(
118 Identifiers<size_t> &within_this_set) const {
119 pgassert(!within_this_set.empty());
120 auto best_order = within_this_set.front();
121 size_t max_size = 0;
122
123
124 for (auto o : within_this_set) {
125 auto size_J = m_orders[o].subsetJ(within_this_set).size();
126 if (max_size < size_J) {
127 max_size = size_J;
128 best_order = o;
129 }
130 }
131 return best_order;
132}
