154 bool
155 PD_Orders::is_valid(double speed) const {
156  for (const auto &o : m_orders) {
157  if (!o.is_valid(speed)) {
158  return false;
159  }
160  pgassert(o.pickup().is_pickup());
161  pgassert(o.delivery().is_delivery());
162  /* P -> D */
163  pgassert(o.delivery().is_compatible_IJ(o.pickup(), speed));
164  }
165  return true;
166 }
