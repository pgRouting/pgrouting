160  double heuristic(V v, V u) {
161  if (m_heuristic == 0) return 0;
162 
163  double dx = graph[v].x() - graph[u].x();
164  double dy = graph[v].y() - graph[u].y();
165  double current;
166 
167  switch (m_heuristic) {
168  case 0:
169  current = 0;
170  break;
171  case 1:
172  current = std::fabs((std::max)(dx, dy)) * m_factor;
173  break;
174  case 2:
175  current = std::fabs((std::min)(dx, dy)) * m_factor;
176  break;
177  case 3:
178  current = (dx * dx + dy * dy) * m_factor * m_factor;
179  break;
180  case 4:
181  current = std::sqrt(dx * dx + dy * dy) * m_factor;
182  break;
183  case 5:
184  current = (std::fabs(dx) + std::fabs(dy)) * m_factor;
185  break;
186  default:
187  current = 0;
188  }
189  return current;
190  }
