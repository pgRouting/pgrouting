66Vehicle_pickDeliver
67Fleet::get_truck() {
68 ENTERING(msg());
69 auto idx = m_un_used.front();
70 msg().log << "Available vehicles: " << m_un_used << "\n";
71 msg().log << "NOT Available vehicles: " << m_used << "\n";
72 msg().log << "getting idx" << idx << "\n";
73 pgassertwm(idx < m_trucks.size(), msg().log.str());
74 m_used += idx;
75 if (m_un_used.size() > 1) m_un_used -= idx;
76 EXITING(msg());
77 return m_trucks[idx];

81Vehicle_pickDeliver
82Fleet::get_truck(size_t order) {
83 size_t idx{0};
84
85 for (const auto &i : m_un_used) {
86 if (m_trucks[i].feasable_orders().has(order)) {
87 idx = i;
88 msg().log << "getting idx" << idx << "\n";
89 m_used += idx;
90 if (m_un_used.size() > 1) m_un_used -= idx;
91 return m_trucks[idx];
92 }
93 }
94
95 /*
96 * using phoney truck
97 */
98 pgassert(false);
99 return m_trucks.back();
100
101 for (auto truck : m_trucks) {
102 if (truck.feasable_orders().has(order)) {
103 idx = truck.idx();
104 msg().log << "idx" << idx << "size" << m_trucks.size();
105 pgassertwm(idx < m_trucks.size(), msg().get_log());
106 m_used += idx;
107 if (m_un_used.size() > 1) m_un_used -= idx;
108 break;
109 }
110 }
111 return m_trucks[idx];
112}

117void
118Fleet::add_vehicle(
119 Vehicle_t vehicle,
120 double factor,
121 const Vehicle_node &starting_site,
122 const Vehicle_node &ending_site) {
123 pgassert(starting_site.is_start() && ending_site.is_end());
124 pgassert(starting_site.opens() <= starting_site.closes());
125 pgassert(ending_site.opens() <= ending_site.closes());
126
127
128 for (int i = 0; i < vehicle.cant_v; ++i) {
129 m_trucks.push_back(Vehicle_pickDeliver(
130 m_trucks.size(),
131 vehicle.id,
132 starting_site,
133 ending_site,
134 vehicle.capacity,
135 vehicle.speed,
136 factor));
137 pgassert((m_trucks.back().idx() + 1) == m_trucks.size());
138 pgassert(m_trucks.back().is_ok());
139 }
140}

152bool
153Fleet::build_fleet(
154 std::vector<Vehicle_t> vehicles,
155 double factor) {
156 /*
157 * creating a phoney truck with max capacity and max window
158 * with the start & end points of the first vehicle given
159 */
160 vehicles.push_back({
161 /*
162 * id, capacity
163 */
164 -1,
165 std::numeric_limits<double>::infinity(),
166
167 vehicles[0].speed,
168 vehicles[0].start_x,
169 vehicles[0].start_y,
170 vehicles[0].start_node_id,
171
172 /*
173 * cant_v, start_open_t, start_close_t, start_service_t
174 */
175 1,
176 0,
177 std::numeric_limits<double>::infinity(),
178 0,
179
180 vehicles[0].end_x,
181 vehicles[0].end_y,
182 vehicles[0].end_node_id,
183 /*
184 * end_open_t, end_close_t, end_service_t
185 */
186 0,
187 std::numeric_limits<double>::infinity(),
188 0});
189
190
191 for (auto vehicle : vehicles) {
192 if (vehicle.cant_v < 0) {
193 throw std::make_pair(std::string("Illegal number of vehicles found"), vehicle.cant_v);
194 }
195
196 if (vehicle.capacity < 0) {
197 throw std::make_pair(std::string("Illegal value for capacity found"), vehicle.capacity);
198 }
199
200 if (!problem->get_cost_matrix().empty()) {
201 if (!problem->get_cost_matrix().has_id(vehicle.start_node_id)) {
202 throw std::make_pair(std::string("Unable to find node on matrix"), vehicle.start_node_id);
203 }
204 if (!problem->get_cost_matrix().has_id(vehicle.end_node_id)) {
205 throw std::make_pair(std::string("Unable to find node on matrix"), vehicle.end_node_id);
206 }
207 }
208
209 if (!(vehicle.start_open_t <= vehicle.start_close_t
210 && vehicle.end_open_t <= vehicle.end_close_t
211 && vehicle.start_open_t <= vehicle.end_close_t)) {
212 msg().error << "Illegal values found on vehicle";
213 msg().log << "On vehicle " << vehicle.id
214 << " a condition is not met, verify that:"
215 << "\nvehicle.start_open_t <= vehicle.start_close_t\t"
216 << vehicle.start_open_t << " <= " << vehicle.start_close_t
217 << "\nvehicle.end_open_t <= vehicle.end_close_t\t"
218 << vehicle.end_open_t << " <= " << vehicle.end_close_t
219 << "\nvehicle.start_open_t <= vehicle.end_close_t\t"
220 << vehicle.start_open_t << " <= " << vehicle.end_close_t;
221
222 throw std::make_pair(msg().get_error(), msg().get_log());
223 }
224
225 /*
226 * Matrix version
227 */
228 auto starting_site = Vehicle_node({problem->get_nodes().size(), vehicle, Tw_node::NodeType::kStart});
229 problem->add_node(starting_site);
230 auto ending_site = Vehicle_node({problem->get_nodes().size(), vehicle, Tw_node::NodeType::kEnd});
231 problem->add_node(ending_site);
232
233 pgassert(starting_site.opens() <= starting_site.closes());
234 pgassert(ending_site.opens() <= ending_site.closes());
235 pgassert(starting_site.is_start() && ending_site.is_end());
236
237 add_vehicle(vehicle, factor, starting_site, ending_site);
238 }
239 Identifiers<size_t> unused(m_trucks.size());
240 m_un_used = unused;
241 return true;
242}

245bool
246Fleet::is_fleet_ok() const {
247 ENTERING(msg());
248 if (!msg().get_error().empty()) return false;
249 for (auto truck : m_trucks) {
250 if (!truck.is_ok()) {
251 msg().error << "Illegal values found on vehicle";
252 msg().log << "On vehicle " << truck.id()
253 << " a condition is not met, verify that:\n"
254 << "- start_open <= start_close\n"
255 << "- end_open <= end_close\n"
256 << "- capacity > 0\n";
257 return false;
258 }
259
260 if (!(truck.start_site().is_start()
261 && truck.end_site().is_end())) {
262 pgassertwm(false, "should never pass through here");
263 msg().error << "Illegal values found on vehicle";
264 return false;
265 }
266 if (!truck.is_feasable()) {
267 msg().error << "Truck is not feasible";
268 return false;
269 }
270 }
271 EXITING(msg());
272 return true;
273}
