82 void doContraction(G &graph, const Identifiers<V> &forbidden_vertices) {
83 graph.set_forbidden_vertices(forbidden_vertices);
84 calculateVertices(graph);

94 void one_cycle(G &graph, V v) {
95 pgassert(is_contractible(graph, v));
96
97 Identifiers<V> adjacent_vertices =
98 graph.find_adjacent_vertices(v);
99 pgassert(adjacent_vertices.size() == 2);
100
101 V u = adjacent_vertices.front();
102 adjacent_vertices.pop_front();
103 V w = adjacent_vertices.front();
104 adjacent_vertices.pop_front();
105
106 pgassert(v != u);
107 pgassert(v != w);
108 pgassert(u != w);
109
110 E e, f;
111 bool found_e, found_f;
112
113 if (graph.is_directed()) {
114 /*
115 * u --> v --> w
116 */
117 boost::tie(e, found_e) = boost::edge(u, v, graph.graph);
118 boost::tie(f, found_f) = boost::edge(v, w, graph.graph);
119 if (found_e && found_f) {
120 graph.process_shortcut(u, v, w);
121 }
122 /*
123 * w --> v --> u
124 */
125 boost::tie(e, found_e) = boost::edge(w, v, graph.graph);
126 boost::tie(f, found_f) = boost::edge(v, u, graph.graph);
127 if (found_e && found_f) {
128 graph.process_shortcut(w, v, u);
129 }
130
131 } else {
132 pgassert(graph.is_undirected());
133 /*
134 * u - v - w
135 */
136 graph.process_shortcut(u, v, w);
137 }
138
139 graph[v].contracted_vertices().clear();
140 boost::clear_vertex(v, graph.graph);
141 m_linearVertices -= v;
142
143 if (is_contractible(graph, u)) {
144 one_cycle(graph, u);
145 } else {
146 m_linearVertices -= u;
147 }
148 if (is_contractible(graph, w)) {
149 one_cycle(graph, w);
150 } else {
151 m_linearVertices -= w;
152 }
153 }
