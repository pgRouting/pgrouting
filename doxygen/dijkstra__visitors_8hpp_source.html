81template <typename V>
82class dijkstra_many_goal_visitor : public boost::default_dijkstra_visitor {
83 public:
84 dijkstra_many_goal_visitor(
85 const std::set<V> &goals,
86 size_t n_goals,
87 std::set<V> &f_goals) :
88 m_goals(goals),
89 m_n_goals(n_goals),
90 m_found_goals(f_goals) {
91 }
92 template <class B_G>
93 void examine_vertex(V u, B_G &) {
94 auto s_it = m_goals.find(u);
95
96 /* examined vertex is not a goal */
97 if (s_it == m_goals.end()) return;
98
99 // found one more goal
100 m_found_goals.insert(*s_it);
101 m_goals.erase(s_it);
102
103 // all goals found
104 if (m_goals.size() == 0) throw found_goals();
105
106 // number of requested goals found
107 --m_n_goals;
108 if (m_n_goals == 0) throw found_goals();
109 }
110
111 private:
112 std::set<V> m_goals;
113 size_t m_n_goals;
114 std::set<V> &m_found_goals;
115};

139template <typename V, typename E>
140class dijkstra_distance_visitor_no_init : public boost::default_dijkstra_visitor {
141 public:
142 explicit dijkstra_distance_visitor_no_init(
143 V root,
144 double distance_goal,
145 std::vector<V> &predecessors,
146 std::vector<double> &distances,
147 std::vector<boost::default_color_type> &color_map) :
148 m_root(root),
149 m_distance_goal(distance_goal),
150 m_num_examined(0),
151 m_predecessors(predecessors),
152 m_dist(distances),
153 m_color(color_map) {
154 pgassert(m_distance_goal > 0);
155 }
156
157 template <class B_G>
158 void examine_vertex(V u, B_G &) {
159 if ( 0 == m_num_examined++) m_root = u;
160 if (m_dist[u] > m_distance_goal) {
161 throw found_goals();
162 }
163 if (u != m_root && m_predecessors[u] == u) {
164 m_color[u] = boost::black_color;
165 }
166 }
167
168 template <class B_G>
169 void examine_edge(E e, B_G &g) {
170 if (source(e, g) != m_root && m_predecessors[source(e, g)] == source(e, g)) {
171 m_color[target(e, g)] = boost::black_color;
172 }
173 }
174
175
176 template <class B_G>
177 void edge_not_relaxed(E e, B_G &g) {
178 if (source(e, g) != m_root && m_predecessors[source(e, g)] == source(e, g)) {
179 m_color[target(e, g)] = boost::black_color;
180 }
181 }
182
183 template <class B_G>
184 void discover_vertex(V u, B_G &) {
185 if (u != m_root && m_predecessors[u] == u) {
186 m_color[u] = boost::black_color;
187 }
188 }
189
190 private:
191 V m_root;
192 double m_distance_goal;
193 size_t m_num_examined;
194 std::vector<V > &m_predecessors;
195 std::vector<double> &m_dist;
196 std::vector<boost::default_color_type> &m_color;
197};
