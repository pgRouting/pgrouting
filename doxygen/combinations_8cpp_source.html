41std::map<int64_t, std::set<int64_t>>
42get_combinations(
43 std::deque<Path> &paths,
44 const std::vector<pgrouting::trsp::Rule> &ruleList) {
45 std::map<int64_t, std::set<int64_t>> new_combinations;
46
47 for (auto &p : paths) {
48 if (p.empty()) {
49 new_combinations[p.start_id()].insert(p.end_id());
50 continue;
51 }
52 std::deque<int64_t> edgesList(p.size());
53 for (const auto &row : p) {
54 edgesList.push_back(row.edge);
55 }
56 for (const auto &r : ruleList) {
57 auto ptr = std::find(edgesList.begin(), edgesList.end(), r.precedences().front());
58 if (ptr == edgesList.end()) continue;
59 /*
60 * And edge on the beginning of a rule list was found
61 * Checking if the complete rule applies
62 */
63
64 /*
65 * Suppose the rule is used
66 */
67 bool rule_breaker = true;
68 for (const auto &e : r.precedences()) {
69 if (*ptr != e) {
70 rule_breaker = false; break;
71 }
72 ++ptr;
73 }
74 if (rule_breaker) {
75 new_combinations[p.start_id()].insert(p.end_id());
76 p.clear();
77 }
78 }
79 }
80 paths.erase(std::remove_if(paths.begin(), paths.end(),
81 [](const Path &p) {
82 return p.size() == 0;
83 }),
84 paths.end());
85 return new_combinations;
86}

107std::map<int64_t , std::set<int64_t>>
108get_combinations(
109 const char *combinations_sql,
110 ArrayType* startsArr, ArrayType* endsArr, bool normal) {
111 using pgrouting::pgget::get_intSet;
112 std::map<int64_t, std::set<int64_t>> result;
113
114 std::set<int64_t> starts;
115 std::set<int64_t> ends;
116
117 if (startsArr && endsArr) {
118 starts = normal? get_intSet(startsArr) : get_intSet(endsArr);
119 ends = normal? get_intSet(endsArr) : get_intSet(startsArr);
120 }
121
122 /* TODO Read query storing like std::map<int64_t , std::set<int64_t>> */
123 /* queries are stored in vectors */
124 auto combinations = combinations_sql?
125 pgrouting::pgget::get_combinations(std::string(combinations_sql)) : std::vector<II_t_rt>();
126
127 /* data comes from a combinations */
128 for (const auto &row : combinations) {
129 result[row.d1.source].insert(row.d2.target);
130 }
131
132 /* data comes from many to many */
133 for (const auto &s : starts) {
134 result[s] = ends;
135 }
136 return result;
137}

153std::map<int64_t , std::set<int64_t>>
154get_combinations(
155 const std::string &combinations_sql,
156 ArrayType* startsArr, ArrayType* endsArr, bool normal, bool &is_matrix) {
157 using pgrouting::pgget::get_intSet;
158 using pgrouting::pgget::get_combinations;
159 std::map<int64_t, std::set<int64_t>> result;
160
161 auto starts = normal? get_intSet(startsArr) : get_intSet(endsArr);
162 auto ends = endsArr? (normal? get_intSet(endsArr) : get_intSet(startsArr)) : std::set<int64_t>();
163
164 /* TODO Read query storing like std::map<int64_t , std::set<int64_t>> */
165 /* queries are stored in vectors */
166 auto combinations = !combinations_sql.empty()? get_combinations(combinations_sql) : std::vector<II_t_rt>();
167
168 /* data comes from CostMatrix */
169 if (combinations.empty() && !starts.empty() && ends.empty()) {
170 is_matrix = true;
171 ends = starts;
172 }
173
174 /* data comes from a combinations */
175 for (const auto &row : combinations) {
176 result[row.d1.source].insert(row.d2.target);
177 }
178
179 /* data comes from many to many */
180 for (const auto &s : starts) {
181 result[s] = ends;
182 }
183 return result;
