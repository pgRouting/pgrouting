43void
44Dmatrix::set_ids(const std::vector < IID_t_rt > &data_costs) {
45 ids.reserve(data_costs.size() * 2);
46 for (const auto &cost : data_costs) {
47 ids.push_back(cost.from_vid);
48 ids.push_back(cost.to_vid);
49 }
50 std::sort(ids.begin(), ids.end());
51 ids.erase(std::unique(ids.begin(), ids.end()), ids.end());
52 /*
53 * freeing up unused space
54 */
55 ids.shrink_to_fit();
56}

85/*
86 * Transforms the input data to a matrix
87 */
88Dmatrix::Dmatrix(const std::vector < IID_t_rt > &data_costs) {
89 set_ids(data_costs);
90 costs.resize(
91 ids.size(),
92 std::vector<double>(
93 ids.size(),
94 (std::numeric_limits<double>::max)()));
95
96 for (const auto &data : data_costs) {
97 costs[get_index(data.from_vid)][get_index(data.to_vid)] = data.cost;
98 }
99
100 for (size_t i = 0; i < costs.size(); ++i) {
101 costs[i][i] = 0;
102 }
103}

106double
107get_distance(std::pair<double, double> p1 , std::pair<double, double> p2) {
108 auto dx = p1.first - p2.first;
109 auto dy = p1.second - p2.second;
110 return std::sqrt(dx * dx + dy * dy);
111}

116Dmatrix::Dmatrix(const std::map<std::pair<double, double>, int64_t> &euclidean_data) {
117 ids.reserve(euclidean_data.size());
118 for (const auto &e : euclidean_data) {
119 ids.push_back(e.second);
120 }
121 costs.resize(
122 ids.size(),
123 std::vector<double>(
124 ids.size(),
125 (std::numeric_limits<double>::max)()));
126
127 for (const auto &from : euclidean_data) {
128 for (const auto &to : euclidean_data) {
129 auto from_id = get_index(from.second);
130 auto to_id = get_index(to.second);
131 costs[from_id][to_id] = get_distance(from.first, to.first);
132 costs[to_id][from_id] = costs[from_id][to_id];
133 }
134 }
135
136 for (size_t i = 0; i < costs.size(); ++i) {
137 costs[i][i] = 0;
138 }
139}

160bool
161Dmatrix::obeys_triangle_inequality() const {
162 for (size_t i = 0; i < costs.size(); ++i) {
163 for (size_t j = 0; j < costs.size(); ++j) {
164 for (size_t k = 0; k < costs.size(); ++k) {
165 if (!(costs[i][k] <= (costs[i][j] + costs[j][k]))) return false;
166 }
167 }
168 }
169 return true;
170}

172bool
173Dmatrix::is_symmetric() const {
174 for (size_t i = 0; i < costs.size(); ++i) {
175 for (size_t j = 0; j < costs.size(); ++j) {
176 if (0.000001 < std::fabs(costs[i][j] - costs[j][i])) {
177 std::ostringstream log;
178 log << "i \t" << i
179 << "j \t" << j
180 << "costs[i][j] \t" << costs[i][j]
181 << "costs[j][i] \t" << costs[j][i]
182 << "\n";
183 log << (*this);
184 pgassertwm(false, log.str());
185 return false;
186 }
187 }
188 }
189 return true;
190}
