63size_t
64Vehicle::insert(std::pair<POS, POS> position_limits, const Vehicle_node &node) {
65 invariant();
66 pgassert(position_limits.first <= m_path.size());
67 pgassert(position_limits.second <= m_path.size());
68
69 auto low = position_limits.first;
70 auto high = position_limits.second;
71 auto best = low;
72
73
74 insert(low, node);
75
76 Vehicle::Cost best_cost(cost());
77
78
79 while (low < high) {
80 swap(low, low + 1);
81 ++low;
82 if (cost_compare(best_cost, cost())) {
83 best_cost = cost();
84 best = low;
85 }
86 }
87 return best;
88
89 pgassert(best < m_path.size());
90 pgassert(m_path[best].idx() == node.idx());
91 invariant();
92}

97bool
98Vehicle::cost_compare(const Cost &lhs, const Cost &rhs) const {
99 /*
100 * capacity violations
101 */
102 if (std::get<1>(lhs) < std::get<1>(rhs)) return true;
103 if (std::get<1>(lhs) > std::get<1>(rhs)) return false;
104
105 /*
106 * time window violations
107 */
108 if (std::get<0>(lhs) < std::get<0>(rhs)) return true;
109 if (std::get<0>(lhs) > std::get<0>(rhs)) return false;
110
111 /*
112 * waiting time
113 */
114 if (std::get<3>(lhs) < std::get<3>(rhs)) return true;
115 if (std::get<3>(lhs) > std::get<3>(rhs)) return false;
116
117 /*
118 * duration
119 */
120 if (std::get<4>(lhs) < std::get<4>(rhs)) return true;
121 if (std::get<4>(lhs) > std::get<4>(rhs)) return false;
122
123 /*
124 * truck size
125 */
126 if (std::get<2>(lhs) < std::get<2>(rhs)) return true;
127 if (std::get<2>(lhs) > std::get<2>(rhs)) return false;
128
129 return false;
130}

134std::vector<Schedule_rt>
135Vehicle::get_postgres_result(
136 int vid) const {
137 std::vector<Schedule_rt> result;
138 /* postgres numbering starts with 1 */
139 int stop_seq(1);
140 msg().log << "getting solution: " << tau() << "\n";
141 for (const auto &p_stop : m_path) {
142 Schedule_rt data = {
143 vid,
144 id(),
145 stop_seq,
146 /* order_id
147 * The order_id is invalid for stops type 0 and 5
148 */
149 (p_stop.type() == 0 || p_stop.type() == 5)? -1 : p_stop.order(),
150 p_stop.id(),
151 p_stop.type(),
152 p_stop.cargo(),
153 p_stop.travel_time(),
154 p_stop.arrival_time(),
155 p_stop.wait_time(),
156 p_stop.service_time(),
157 p_stop.departure_time()};
158 result.push_back(data);
159 ++stop_seq;
160 }
161 return result;
162}

224Vehicle::swap(POS i, POS j) {
225 invariant();
226 pgassert(m_path.size() > 3);
227 pgassert(!m_path[i].is_start());
228 pgassert(!m_path[i].is_end());
229 pgassert(!m_path[j].is_start());
230 pgassert(!m_path[j].is_end());
231
232 std::swap(m_path[i], m_path[j]);
233 i < j ? evaluate(i) : evaluate(j);
234
235 invariant();
236}

260void
261Vehicle::evaluate(POS from) {
262 invariant();
263 // preconditions
264 pgassert(from < m_path.size());
265
266
267 auto node = m_path.begin() + static_cast<difference_type>(from);
268
269 while (node != m_path.end()) {
270 if (node == m_path.begin()) {
271 node->evaluate(m_capacity);
272 } else {
273 node->evaluate(*(node - 1), m_capacity, speed());
274 }
275
276 ++node;
277 }
278 invariant();
279}

316size_t
317Vehicle::getDropPosLowLimit(const Vehicle_node &nodeI) const {
318 invariant();
319
320 POS low = 0;
321 POS high = m_path.size();
322 POS low_limit = high;
323
324 /* J == m_path[low_limit - 1] */
325 while (low_limit > low
326 && m_path[low_limit - 1].is_compatible_IJ(nodeI, speed())
327 && !m_path[low_limit - 1].is_pickup()) {
328 --low_limit;
329 }
330
331 invariant();
332 return low_limit;
333}

382Vehicle::getPosHighLimit(const Vehicle_node &nodeJ) const {
383 invariant();

385 POS low = 0;

389 /* I == m_path[high_limit] */

392 ++high_limit;
393 }

395 invariant();

480bool
481operator<(const Vehicle &lhs, const Vehicle &rhs) {
482 lhs.invariant();
483 rhs.invariant();
484
485 if (lhs.m_path.size() < rhs.m_path.size()) return true;
486
487 /* here because sizes are equal */
488
489 if (lhs.m_path.back().total_travel_time()
490 < rhs.m_path.back().total_travel_time()) return true;
491
492 return false;
493}
