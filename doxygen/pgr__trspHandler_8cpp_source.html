62 int64_t Pgr_trspHandler::renumber_edges(
63  pgr_edge_t *edges,
64  size_t total_edges) const {
65  int64_t v_min_id = UINT64_MAX;
66  size_t z;
67  for (z = 0; z < total_edges; z++) {
68  if (edges[z].source < v_min_id)
69  v_min_id = edges[z].source;
70 
71  if (edges[z].target < v_min_id)
72  v_min_id = edges[z].target;
73  }
74 
75  for (z = 0; z < total_edges; z++) {
76  edges[z].source -= v_min_id;
77  edges[z].target -= v_min_id;
78  }
79 
80  return v_min_id;
81 }

95 double Pgr_trspHandler::construct_path(int64_t ed_id, Position pos) {
96  pgassert(pos != ILLEGAL);
97 
98  if (m_parent[ed_id].isIllegal(pos)) {
99  Path_t pelement;
100  auto cur_edge = &m_edges[ed_id];
101  if (pos == RC_EDGE) {
102  pelement.node = cur_edge->startNode();
103  pelement.cost = cur_edge->cost();
104  } else {
105  pelement.node = cur_edge->endNode();
106  pelement.cost = cur_edge->r_cost();
107  }
108  pelement.edge = cur_edge->edgeID();
109 
110  m_path.push_back(pelement);
111  pgassert(m_path.start_id() == m_start_vertex);
112  return pelement.cost;
113  }
114 
115  double ret = construct_path(m_parent[ed_id].e_idx[pos],
116  m_parent[ed_id].v_pos[pos]);
117  Path_t pelement;
118  auto cur_edge = &m_edges[ed_id];
119  if (pos == RC_EDGE) {
120  pelement.node = cur_edge->startNode();
121  pelement.cost = m_dCost[ed_id].endCost - ret;
122  ret = m_dCost[ed_id].endCost;
123  } else {
124  pelement.node = cur_edge->endNode();
125  pelement.cost = m_dCost[ed_id].startCost - ret;
126  ret = m_dCost[ed_id].startCost;
127  }
128  pelement.edge = cur_edge->edgeID();
129 
130  m_path.push_back(pelement);
131 
132  return ret;
133 }

137 double Pgr_trspHandler::getRestrictionCost(
138  int64_t edge_ind,
139  const EdgeInfo &edge,
140  bool isStart) {
141  double cost = 0.0;
142  int64_t edge_id = edge.edgeID();
143  if (m_ruleTable.find(edge_id) == m_ruleTable.end()) {
144  return(0.0);
145  }
146  auto vecRules = m_ruleTable[edge_id];
147  int64_t st_edge_ind = edge_ind;
148  for (const auto &rule : vecRules) {
149  bool flag = true;
150  int64_t v_pos = (isStart? C_EDGE : RC_EDGE);
151  edge_ind = st_edge_ind;
152 
153  pgassert(!(edge_ind == -1));
154  for (auto const &precedence : rule.precedencelist()) {
155  if (precedence != m_edges[edge_ind].edgeID()) {
156  flag = false;
157  break;
158  }
159  auto m_parent_ind = m_parent[edge_ind].e_idx[v_pos];
160  v_pos = m_parent[edge_ind].v_pos[v_pos];
161  edge_ind = m_parent_ind;
162  }
163  if (flag)
164  cost += rule.cost();
165  }
166  return cost;
167 }

184 void Pgr_trspHandler::explore(
185  int64_t cur_node,
186  const EdgeInfo cur_edge,
187  bool isStart) {
188  double totalCost;
189 
190  auto vecIndex = cur_edge.get_idx(isStart);
191 
192  for (const auto &index : vecIndex) {
193  auto edge = m_edges[index];
194 
195  auto extra_cost = getRestrictionCost(
196  cur_edge.idx(),
197  edge, isStart);
198 
199  if ((edge.startNode() == cur_node) && (edge.cost() >= 0.0)) {
200  totalCost = get_tot_cost(
201  edge.cost() + extra_cost,
202  cur_edge.idx(),
203  isStart);
204 
205  if (totalCost < m_dCost[index].endCost) {
206  m_dCost[index].endCost = totalCost;
207  m_parent[edge.idx()].v_pos[RC_EDGE] = isStart? C_EDGE : RC_EDGE;
208  m_parent[edge.idx()].e_idx[RC_EDGE] =
209  cur_edge.idx();
210 
211  add_to_que(totalCost, edge.idx(), true);
212  }
213  }
214 
215  if ((edge.endNode() == cur_node) && (edge.r_cost() >= 0.0)) {
216  totalCost = get_tot_cost(
217  edge.r_cost() + extra_cost,
218  cur_edge.idx(),
219  isStart);
220 
221  if (totalCost < m_dCost[index].startCost) {
222  m_dCost[index].startCost = totalCost;
223  m_parent[edge.idx()].v_pos[C_EDGE] = isStart? C_EDGE : RC_EDGE;
224  m_parent[edge.idx()].e_idx[C_EDGE] = cur_edge.idx();
225 
226  add_to_que(totalCost, edge.idx(), false);
227  }
228  }
229  } // for
230 }

289 std::deque<Path>
290 Pgr_trspHandler::process(
291  const std::vector<int64_t> sources,
292  const std::vector<int64_t> targets) {
293  std::deque<Path> paths;
294  for (const auto &s : sources) {
295  for (const auto &t : targets) {
296  paths.push_back(process(s, t));
297  }
298  }
299 
300  std::sort(paths.begin(), paths.end(),
301  [](const Path &e1, const Path &e2)->bool {
302  return e1.end_id() < e2.end_id();
303  });
304  std::stable_sort(paths.begin(), paths.end(),
305  [](const Path &e1, const Path &e2)->bool {
306  return e1.start_id() < e2.start_id();
307  });
308  return paths;
309 }

327 void Pgr_trspHandler::initialize_que() {
328  /*
329  * For each adjacent edge to the start_vertex
330  */
331  for (const auto source : m_adjacency[m_start_vertex]) {
332  EdgeInfo &cur_edge = m_edges[source];
333 
334  if (cur_edge.startNode() == m_start_vertex
335  && cur_edge.cost() >= 0.0) {
336  m_dCost[cur_edge.idx()].endCost = cur_edge.cost();
337  m_parent[cur_edge.idx()].v_pos[0] = ILLEGAL;
338  add_to_que(cur_edge.cost(), cur_edge.idx(), true);
339  }
340 
341  if (cur_edge.endNode() == m_start_vertex
342  && cur_edge.r_cost() >= 0.0) {
343  m_dCost[cur_edge.idx()].startCost =
344  cur_edge.r_cost();
345  m_parent[cur_edge.idx()].v_pos[1] = ILLEGAL;
346  add_to_que(cur_edge.r_cost(), cur_edge.idx(), false);
347  }

351 EdgeInfo Pgr_trspHandler::dijkstra_exploration() {
352  EdgeInfo cur_edge;
353  pgassert(current_node == m_start_vertex);
354 
355  while (!que.empty()) {
356  auto cur_pos = que.top();
357  que.pop();
358 
359  auto cure_idxex = cur_pos.second.first;
360  cur_edge = m_edges[cure_idxex];
361 
362  if (cur_pos.second.second) {
363  /*
364  * explore edges connected to end node
365  */
366  current_node = cur_edge.endNode();
367  if (cur_edge.cost() < 0.0) continue;
368  if (current_node == m_end_vertex) break;
369  explore(current_node, cur_edge, false);
370  } else {
371  /*
372  * explore edges connected to start node
373  */
374  current_node = cur_edge.startNode();
375  if (cur_edge.r_cost() < 0.0) continue;
376  if (current_node == m_end_vertex) break;
377  explore(current_node, cur_edge, true);
378  }
379  }
380  return cur_edge;
381 }

387 Pgr_trspHandler::process_trsp(
388  size_t edge_count) {
389  pgassert(m_path.start_id() == m_start_vertex);
390  pgassert(m_path.end_id() == m_end_vertex);
391  pgassert(m_parent.empty());
392 
393  m_parent.resize(edge_count + 1);
394  m_dCost.resize(edge_count + 1);
395 
396  initialize_que();
397 
398  current_node = m_start_vertex;
399 
400  pgassert(m_path.start_id() == m_start_vertex);
401 
402  auto cur_edge = dijkstra_exploration();
403 
404  pgassert(m_path.start_id() == m_start_vertex);
405  if (current_node != m_end_vertex) {
406  Path result(m_start_vertex, m_end_vertex);
407  return result.renumber_vertices(m_min_id);;
408  }
409 
410  pgassert(m_path.start_id() == m_start_vertex);
411 
412  if (current_node == cur_edge.startNode()) {
413  construct_path(cur_edge.idx(), C_EDGE);
414  } else {
415  construct_path(cur_edge.idx(), RC_EDGE);
416  }
417 
418  Path_t pelement;
419  pelement.node = m_end_vertex;
420  pelement.edge = -1;
421  pelement.cost = 0.0;
422  m_path.push_back(pelement);
423 
424  m_path.Path::recalculate_agg_cost();
425  return m_path.renumber_vertices(m_min_id);
426 }

433 void Pgr_trspHandler::construct_graph(
434  pgr_edge_t* edges,
435  const size_t edge_count,
436  const bool directed) {
437  for (size_t i = 0; i < edge_count; i++) {
438  auto current_edge = &edges[i];
439 
440  /*
441  * making all costs > 0
442  */
443  if (current_edge->cost < 0 && current_edge->reverse_cost > 0) {
444  std::swap(current_edge->cost, current_edge->reverse_cost);
445  std::swap(current_edge->source, current_edge->target);
446  }
447 
448  if (!directed) {
449  if (current_edge->reverse_cost < 0) {
450  current_edge->reverse_cost = current_edge->cost;
451  }
452  }
453  addEdge(*current_edge);
454  }
455  m_mapEdgeId2Index.clear();
456 }

506 bool Pgr_trspHandler::addEdge(const pgr_edge_t edgeIn) {
507  /*
508  * The edge was already inserted
509  *
510  * If the user has rows with repeated edge id, the subsequent edges will be ignored
511  *
512  * When changing to boost graph, the additional edges are to be added
513  */
514  if (m_mapEdgeId2Index.find(edgeIn.id) != m_mapEdgeId2Index.end()) {
515  return false;
516  }
517 
518 
519  /*
520  * the index of this new edge in the edges container is
521  * m_edges.size()
522  */
523  EdgeInfo edge(edgeIn, m_edges.size());
524 
525  // Adding edge to the list
526  m_mapEdgeId2Index.insert(
527  std::make_pair(
528  edge.edgeID(),
529  m_edges.size()));
530 
531  m_edges.push_back(edge);
532 
533  EdgeInfo &newEdge = m_edges[m_edges.size()-1];
534 
535 
536 
537  /*
538  * Searching the start node for connectivity
539  */
540  auto itNodeMap = m_adjacency.find(edgeIn.source);
541 
542  if (itNodeMap != m_adjacency.end()) {
543  for (const auto e_idx : itNodeMap->second) {
544  connectStartEdge(edge.idx(), e_idx);
545  }
546  }
547 
548 
549  /*
550  * Searching the end node for connectivity
551  */
552  itNodeMap = m_adjacency.find(edgeIn.target);
553  if (itNodeMap != m_adjacency.end()) {
554  for (const auto e_idx : itNodeMap->second) {
555  connectEndEdge(edge.idx(), e_idx);
556  }
557  }
558 
559 
560  /*
561  * Add the edges to the adjacency list
562  */
563  m_adjacency[edgeIn.source].push_back(newEdge.idx());
564  m_adjacency[edgeIn.target].push_back(newEdge.idx());
565 
566 
567 
568  return true;
569 }
