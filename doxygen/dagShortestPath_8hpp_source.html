63 std::deque<Path> dag(
64 G &graph,
65 int64_t start_vertex,
66 const std::set<int64_t> &end_vertex,
67 bool only_cost) {
68 std::deque<Path> paths;
69
70 /* precondition */
71 if (!graph.has_vertex(start_vertex)) return paths;
72
73 /* adjust predecessors and distances vectors */
74 clear();
75 size_t n_goals = (std::numeric_limits<size_t>::max)();
76 predecessors.resize(graph.num_vertices());
77 distances.resize(
78 graph.num_vertices(),
79 std::numeric_limits<double>::infinity());
80
81
82 auto v_source(graph.get_V(start_vertex));
83
84 std::set<V> v_targets;
85 for (const auto &vertex : end_vertex) {
86 if (graph.has_vertex(vertex)) v_targets.insert(graph.get_V(vertex));
87 }
88 if (v_targets.empty()) return paths;
89
90 dag_1_to_many(graph, v_source, v_targets, n_goals);
91 paths = get_paths(graph, v_source, v_targets, only_cost);
92
93 std::stable_sort(paths.begin(), paths.end(),
94 [](const Path &e1, const Path &e2)->bool {
95 return e1.end_id() < e2.end_id();
96 });
97
98 return paths;
99 }

103 std::deque<Path> dag(
104 G &graph,
105 const std::map<int64_t, std::set<int64_t>> &combinations,
106 bool only_cost) {
107 std::deque<Path> paths;
108
109 for (const auto &c : combinations) {
110 auto result_paths = dag(graph, c.first, c.second, only_cost);
111 paths.insert(
112 paths.end(),
113 std::make_move_iterator(result_paths.begin()),
114 std::make_move_iterator(result_paths.end()));
115 }
116
117 return paths;
118 }

122 private:
124 bool dag_1_to_many(
125 G &graph,
126 V source,
127 const std::set<V> &targets,
128 size_t n_goals = (std::numeric_limits<size_t>::max)()) {
129 CHECK_FOR_INTERRUPTS();
130 try {
131 boost::dag_shortest_paths(graph.graph, source,
132 boost::predecessor_map(&predecessors[0])
133 .weight_map(get(&G::G_T_E::cost, graph.graph))
134 .distance_map(&distances[0])
135 .distance_inf(std::numeric_limits<double>::infinity())
136 .visitor(dijkstra_many_goal_visitor(targets, n_goals)));
137 } catch(found_goals &) {
138 return true;
139 } catch (boost::exception const& ex) {
140 (void)ex;
141 throw;
142 } catch (std::exception &e) {
143 (void)e;
144 throw;
145 } catch (...) {
146 throw;
147 }
148 return true;
149 }

192 class dijkstra_many_goal_visitor : public boost::default_dijkstra_visitor {
193 public:
194 explicit dijkstra_many_goal_visitor(
195 const std::set<V> &goals,
196 size_t n_goals) :
197 m_goals(goals),
198 m_n_goals(n_goals) {}
199 template <class B_G>
200 void examine_vertex(V u, B_G &) {
201 auto s_it = m_goals.find(u);
202 if (s_it == m_goals.end()) return;
203
204 // found one more goal
205 m_goals.erase(s_it);
206
207 // all goals found
208 if (m_goals.size() == 0) throw found_goals();
209
210 // number of requested goals found
211 --m_n_goals;
212 if (m_n_goals == 0) throw found_goals();
213 }
214
215 private:
216 std::set<V> m_goals;
217 size_t m_n_goals;
218 };
