60 private:
61 std::vector< II_t_rt > generatemakeConnected(G &graph ) {
62 std::vector<size_t> component(boost::num_vertices(graph.graph));
63 auto comp = boost::connected_components(graph.graph, &component[0]);
64 comp--;
65 auto edgeCount = boost::num_edges(graph.graph);
66 size_t newEdge = 0;
67 log << "Number of Components before: " << boost::connected_components(graph.graph, &component[0]) << "\n";
68 size_t i = 0;
69
70 /* abort in case of an interruption occurs (e.g. the query is being cancelled) */
71 CHECK_FOR_INTERRUPTS();
72 try {
73 boost::make_connected(graph.graph);
74 } catch (boost::exception const& ex) {
75 (void)ex;
76 throw;
77 } catch (std::exception &e) {
78 (void)e;
79 throw;
80 } catch (...) {
81 throw;
82 }
83
84 log << "Number of Components after: " << boost::connected_components(graph.graph, &component[0]) << "\n";
85 E_i ei, ei_end;
86 std::vector< II_t_rt > results(comp);
87 for (boost::tie(ei, ei_end) = edges(graph.graph); ei != ei_end; ++ei) {
88 int64_t src = graph[graph.source(*ei)].id;
89 int64_t tgt = graph[graph.target(*ei)].id;
90 log<< "src:" << src<< "tgt:" << tgt <<"\n";
91 if (newEdge >= edgeCount) {
92 results[i] = {{src}, {tgt}};
93 i++;
94 }
95 newEdge++;
96 }
97 return results;
98 }
99};
