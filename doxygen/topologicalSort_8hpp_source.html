54template < class G >
55class Pgr_topologicalSort {
56 public:
57 typedef typename G::V V;
58
59 std::vector<I_rt> topologicalSort(
60 G &graph);
61
62 private:
63 std::vector< I_rt >
64 generatetopologicalSort(
65 const G &graph ) {
66 std::vector< I_rt > results;
67
68 typedef typename std::vector< V > container;
69 container c;
70
71 /* abort in case of an interruption occurs (e.g. the query is being cancelled) */
72 CHECK_FOR_INTERRUPTS();
73
74 try {
75 boost::topological_sort(graph.graph, std::back_inserter(c));
76 } catch (...) {
77 throw std::string("Graph is not DAG");
78 }
79
80 typename std::vector< V >::reverse_iterator ii;
81 for (ii = c.rbegin(); ii != c.rend(); ++ii) {
82 auto t = *ii;
83 I_rt tmp;
84 tmp.id = graph.graph[t].id;
85 results.push_back(tmp);
86 }
87
88 return results;
89 }
90};

92template < class G >
93std::vector<I_rt>
94Pgr_topologicalSort< G >::topologicalSort(
95 G &graph) {
96 return generatetopologicalSort(
97 graph);
98}
