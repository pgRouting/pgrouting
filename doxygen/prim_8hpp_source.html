95template <class G>
96void
97Pgr_prim<G>::primTree(
98 const G &graph,
99 int64_t root_vertex) {
100 clear();
101
102 predecessors.resize(graph.num_vertices());
103 distances.resize(graph.num_vertices());
104
105 auto v_root(graph.get_V(root_vertex));
106
107 using prim_visitor = visitors::Prim_dijkstra_visitor<V>;
108
109 /* abort in case of an interruption occurs (e.g. the query is being cancelled) */
110 CHECK_FOR_INTERRUPTS();
111
112 boost::prim_minimum_spanning_tree(
113 graph.graph,
114 &predecessors[0],
115 boost::distance_map(&distances[0]).
116 weight_map(get(&G::G_T_E::cost, graph.graph))
117 .root_vertex(v_root)
118 .visitor(prim_visitor(data)));
119
120 for (const auto v : data) {
121 /*
122 * its not a tree, its a forest
123 * - v is not on current tree
124 */
125 if (std::isinf(distances[v])) continue;
126 m_unassigned.erase(v);
127
128
129 auto u = predecessors[v];
130
131 /*
132 * Not a valid edge
133 */
134 if (u == v) continue;
135
136 auto cost = distances[u] - distances[v];
137 auto edge = graph.get_edge(u, v, cost);
138 this->m_spanning_tree.edges.insert(edge);
139 }
140}

143template <class G>
144void
145Pgr_prim<G>::generate_mst(const G &graph) {
146 this->clear();
147
148 size_t totalNodes = num_vertices(graph.graph);
149
150 m_unassigned.clear();
151 for (V v = 0; v < totalNodes; ++v) {
152 m_unassigned.insert(m_unassigned.end(), v);
153 }
154
155 while (!m_unassigned.empty()) {
156 auto root = *m_unassigned.begin();
157 m_unassigned.erase(m_unassigned.begin());
158 primTree(
159 graph,
160 graph.graph[root].id);
161 }
162}
