55void
56post_process_trsp(std::deque<pgrouting::Path> &paths) {
57 using pgrouting::Path;
58 paths.erase(std::remove_if(paths.begin(), paths.end(),
59 [](const Path &p) {
60 return p.size() == 0;
61 }),
62 paths.end());
63 for (auto &p : paths) {
64 p.recalculate_agg_cost();
65 }
66
67 std::sort(paths.begin(), paths.end(),
68 [](const Path &e1, const Path &e2)->bool {
69 return e1.end_id() < e2.end_id();
70 });
71 std::stable_sort(paths.begin(), paths.end(),
72 [](const Path &e1, const Path &e2)->bool {
73 return e1.start_id() < e2.start_id();
74 });
75}
