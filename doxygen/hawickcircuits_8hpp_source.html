56template <typename G, typename E>
57class circuit_detector{
58 public:
65 circuit_detector(
66 G &graph,
67 std::deque<circuits_rt> &data) :
68 m_graph(graph),
69 m_data(data) {}
70 template <typename P, typename Gr>
71 void cycle(P const &p, Gr const& Grap) {
72 if (p.empty()) {
73 return;
74 }
75 int seq = 0;
76 typename P::const_iterator i, before_end = boost::prior(p.end());
77 auto start_vid = m_graph[*p.begin()].id;
78 auto end_vid = start_vid;
79 double agg_cost = 0;
80 E edge;
81 for (i = p.begin(); i != before_end; ++i, ++seq) {
82 auto node = m_graph[*i].id;
83 edge = boost::edge(*i, *(i+1), Grap).first;
84 auto cost = m_graph[edge].cost;
85 auto id = m_graph[edge].id;
86 m_data.push_back({circuit_No, seq, start_vid, end_vid, node, id, cost, agg_cost});
87 agg_cost = agg_cost + cost;
88 }
89 auto node = m_graph[*i].id;
90 edge = boost::edge(*(boost::prior(p.end())), *(p.begin()), Grap).first;
91 auto cost = m_graph[edge].cost;
92 auto id = m_graph[edge].id;
93 m_data.push_back({circuit_No, seq, start_vid, end_vid, node, id, cost, agg_cost});
94 agg_cost = agg_cost + cost;
95 ++seq;
96 m_data.push_back({circuit_No, seq, start_vid, end_vid, start_vid, -1, 0, agg_cost});
97 circuit_No++;
98 }
99
100 private:
101 G &m_graph;
102 std::deque<circuits_rt> &m_data;
103 int circuit_No = 1;
104};

106template <class G>
107class pgr_hawickCircuits{
108 public:
109 typedef typename G::E E;
110
128 std::deque<circuits_rt> hawickCircuits(G & graph) {
129 // results storing the output
130 std::deque<circuits_rt> results;
131
132 // a circuit detector to provide the mechanism to store the circuit
133 circuit_detector <G, E> detector(graph, results);
134
135 /* abort in case of an interruption occurs (e.g. the query is being cancelled) */
136 CHECK_FOR_INTERRUPTS();
137 try {
138 boost::hawick_unique_circuits(graph.graph, detector);
139 } catch (boost::exception const& ex) {
140 (void)ex;
141 throw;
142 } catch (std::exception &e) {
143 (void)e;
144 throw;
145 } catch (...) {
146 throw;
147 }
148 return results;
150}
151};
