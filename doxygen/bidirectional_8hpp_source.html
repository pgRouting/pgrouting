126 forward_cost[v_source] = 0;
127 forward_queue.push(std::make_pair(0.0, v_source));

130 backward_cost[v_target] = 0;
131 backward_queue.push(std::make_pair(0.0, v_target));
132
133 while (!forward_queue.empty() && !backward_queue.empty()) {
134 auto forward_node = forward_queue.top();
135 auto backward_node = backward_queue.top();
136 /*
137 * done: there is no path with lower cost
138 */
139 if (forward_node.first == INF || backward_node.first == INF) {
140 break;
141 }
142
143 /*
144 * Explore from the cheapest side
145 */
146 if (backward_node.first < forward_node.first) {
147 backward_queue.pop();
148 if (!backward_finished[backward_node.second]) {
149 explore_backward(backward_node);
150 }
151 if (found(backward_node.second)) {
152 break;
153 }
154 } else {
155 forward_queue.pop();
156 if (!forward_finished[forward_node.second]) {
157 explore_forward(forward_node);
158 }
159 if (found(forward_node.second)) {
160 break;
161 }
162 }
163 }
164
165 if (best_cost == INF) return Path();
