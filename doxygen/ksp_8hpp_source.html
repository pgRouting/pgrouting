55template < class G >
56class Pgr_ksp : public Pgr_messages {
57 typedef typename G::V V;
58 typedef std::set<Path, compPathsLess> pSet;
59
60 public:
61 Pgr_ksp() :
62 m_start(0),
63 m_end(0),
64 m_K(0),
65 m_heap_paths(false) {
66 m_vis = new Visitor;
67 }
68 ~Pgr_ksp() {
69 delete m_vis;
70 }
71
72 std::deque<Path> Yen(
73 G &graph,
74 int64_t start_vertex,
75 int64_t end_vertex,
76 size_t K,
77 bool heap_paths) {
78 /*
79 * No path: already in destination
80 */
81 if ((start_vertex == end_vertex) || (K == 0)) {
82 return std::deque<Path>();
83 }
84
85 /*
86 * no path: disconnected vertices
87 */
88 if (!graph.has_vertex(start_vertex)
89 || !graph.has_vertex(end_vertex)) {
90 return std::deque<Path>();
91 }
92
93 /*
94 * clean up containers
95 */
96 clear();
97
98
99 v_source = graph.get_V(start_vertex);
100 v_target = graph.get_V(end_vertex);
101 m_start = start_vertex;
102 m_end = end_vertex;
103 m_K = K;
104 m_heap_paths = heap_paths;
105
106
107 executeYen(graph);
108
109 auto paths = get_results();
110 if (!m_heap_paths && paths.size() > m_K) paths.resize(m_K);
111
112 return paths;
113 }
114
115 void clear() {
116 m_Heap.clear();
117 m_ResultSet.clear();
118 }

122 class Visitor {
123 public:
124 virtual ~Visitor() {}
125
126 virtual void on_insert_first_solution(const Path) const {
127 /* noop */
128 }
129
130 virtual void on_insert_to_heap(const Path) const {
131 /* noop */
132 }
133 };
134
135 protected:
137 void executeYen(G &graph) {
138 clear();
139 curr_result_path = getFirstSolution(graph);
140 m_vis->on_insert_first_solution(curr_result_path);
141
142 if (m_ResultSet.size() == 0) return; // no path found
143
144 while (m_ResultSet.size() < m_K) {
145 doNextCycle(graph);
146 if (m_Heap.empty()) break;
147 curr_result_path = *m_Heap.begin();
148 curr_result_path.recalculate_agg_cost();
149 m_ResultSet.insert(curr_result_path);
150 m_Heap.erase(m_Heap.begin());
151 }
152 }
153
156
158 protected:
159 Path getFirstSolution(G &graph) {
160 Path path;
161
162 path = algorithms::dijkstra(graph, m_start, m_end);
163 path.recalculate_agg_cost();
164
165 if (path.empty()) return path;
166 m_ResultSet.insert(path);
167 return path;
168 }
169
170
171 protected:
173 void doNextCycle(G &graph) {
174 for (unsigned int i = 0; i < curr_result_path.size(); ++i) {
175 int64_t spurNodeId = curr_result_path[i].node;
176
177 auto rootPath = curr_result_path.getSubpath(i);
178
179 for (const auto &path : m_ResultSet) {
180 if (path.isEqual(rootPath) && spurNodeId == path[i].node) {
181 if (path.size() > i + 1) {
182 graph.disconnect_edge(path[i].node, // from
183 path[i + 1].node); // to
184 }
185 }
186 }
187
188 removeVertices(graph, rootPath);
189
190 auto spurPath = algorithms::dijkstra(graph, spurNodeId, m_end);
191
192 if (spurPath.size() > 0) {
193 rootPath.appendPath(spurPath);
194 m_Heap.insert(rootPath);
195 m_vis->on_insert_to_heap(rootPath);
196 }
197
198 graph.restore_graph();
199 }
200 }
201
202 protected:
204 void removeVertices(G &graph, const Path &subpath) {
205 for (const auto &e : subpath) graph.disconnect_vertex(e.node);
206 }
207
208 std::deque<Path> get_results() {
209 if (this->m_ResultSet.empty()) {
210 return std::deque<Path>();
211 }
212
213 std::deque<Path> paths(m_ResultSet.begin(), m_ResultSet.end());
214
215 if (m_heap_paths && !m_Heap.empty()) {
216 paths.insert(paths.end(), m_Heap.begin(), m_Heap.end());
217 }
218 pgassert(!paths.empty());
219
220 std::sort(paths.begin(), paths.end(), compPathsLess());
221
222 return paths;
223 }
224
226
227 protected:
230 V v_source;
231 V v_target;
232 int64_t m_start;

235 bool m_heap_paths;
236
237 Path curr_result_path;

239 pSet m_ResultSet;
240 pSet m_Heap;
241
242 Visitor *m_vis;
243};

250 template <class G>
251 std::deque<Path> Yen(
252 G &graph,
253 const std::map<int64_t, std::set<int64_t>> &combinations,
254 size_t k,
255 bool heap_paths) {
256 std::deque<Path> paths;
257 pgrouting::yen::Pgr_ksp<G> fn_yen;
258
259 for (const auto &c : combinations) {
260 if (!graph.has_vertex(c.first)) continue;
261
262 for (const auto &destination : c.second) {
263 if (!graph.has_vertex(destination)) continue;
264
265 fn_yen.clear();
266 auto result_path = fn_yen.Yen(graph, c.first, destination, k, heap_paths);
267 paths.insert(paths.end(), result_path.begin(), result_path.end());
268 }
269 }
270
271 return paths;
272 }
