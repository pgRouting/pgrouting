72template <class G>
73void
74Pgr_kruskal<G>::generate_mst(const G &graph) {
75 this->clear();
76 /* abort in case of an interruption occurs (e.g. the query is being cancelled) */
77 CHECK_FOR_INTERRUPTS();
78 boost::kruskal_minimum_spanning_tree(
79 graph.graph,
80 std::inserter(this->m_spanning_tree.edges, this->m_spanning_tree.edges.begin()),
81 boost::weight_map(get(&G::G_T_E::cost, graph.graph)));
82}
