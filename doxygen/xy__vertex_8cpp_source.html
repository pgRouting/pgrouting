39template<typename T>
40typename std::enable_if<!std::numeric_limits<T>::is_integer, bool>::type
41 almost_equal(T x, T y, int ulp) {
42 // the machine epsilon has to be scaled to the magnitude of the values used
43 // and multiplied by the desired precision in ULPs (units in the last place)
44 return std::abs(x-y) <= std::numeric_limits<T>::epsilon() * std::abs(x+y) * ulp
45 // unless the result is subnormal
46 || std::abs(x-y) < std::numeric_limits<T>::min();
47}

59bool
60XY_vertex::operator==(const XY_vertex &rhs) const {
61 if (&rhs == this) return true;
62 return this->id == rhs.id &&
63 almost_equal(point.x(), rhs.point.x(), 2) && almost_equal(point.y(), rhs.point.y(), 2);
64}

67size_t
68check_vertices(
69 std::vector < XY_vertex > vertices) {
70 auto count(vertices.size());
71 std::stable_sort(
72 vertices.begin(), vertices.end(),
73 [](const XY_vertex &lhs, const XY_vertex &rhs)
74 {return lhs.id < rhs.id;});
75 vertices.erase(
76 std::unique(
77 vertices.begin(), vertices.end(),
78 [](const XY_vertex &lhs, const XY_vertex &rhs)
79 {return lhs.id == rhs.id;}), vertices.end());
80
81 return count - vertices.size();
82}

84std::vector < XY_vertex >
85extract_vertices(
86 const std::vector <Edge_xy_t > &data_edges) {
87
88 std::vector< XY_vertex > vertices;
89 if (data_edges.empty()) return vertices;
90
91 vertices.reserve(data_edges.size() * 2);
92
93 for (const auto &edge : data_edges) {
94 XY_vertex v_source(edge, true);
95 vertices.push_back(v_source);
96
97 XY_vertex v_target(edge, false);
98 vertices.push_back(v_target);
99 }
100
101 /*
102 * sort and delete duplicates
103 */
104 std::stable_sort(
105 vertices.begin(), vertices.end(),
106 [](const XY_vertex &lhs, const XY_vertex &rhs)
107 {return lhs.id < rhs.id;});
108 vertices.erase(
109 std::unique(
110 vertices.begin(), vertices.end(),
111 [](const XY_vertex &lhs, const XY_vertex &rhs)
112 {return lhs.id == rhs.id;}), vertices.end());
113 return vertices;
114}
