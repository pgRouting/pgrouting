113Edge_bool_t fetch_basic_edge(
114 const HeapTuple tuple,
115 const TupleDesc &tupdesc,
116 const std::vector<Column_info_t> &info,
117 int64_t *default_id,
118 size_t *valid_edges,
119 bool) {
120 Edge_bool_t edge;
121 if (column_found(info[0].colNumber)) {
122 edge.id = getBigInt(tuple, tupdesc, info[0]);
123 } else {
124 edge.id = *default_id;
125 ++(*default_id);
126 }
127 bool new_columns = column_found(info[5].colNumber);
128
129 edge.source = getBigInt(tuple, tupdesc, info[1]);
130 edge.target = getBigInt(tuple, tupdesc, info[2]);
131
132 if (new_columns) {
133 edge.going = getFloat8(tuple, tupdesc, info[5]) > 0
134 || (column_found(info[6].colNumber)
135 && getFloat8(tuple, tupdesc, info[6]) > 0);
136 } else {
137 edge.going = getFloat8(tuple, tupdesc, info[3]) > 0
138 || (column_found(info[4].colNumber)
139 && getFloat8(tuple, tupdesc, info[4]) > 0);
140 }
141
142 (*valid_edges)++;
143 return edge;
144}

146Edge_t fetch_edge(
147 const HeapTuple tuple,
148 const TupleDesc &tupdesc,
149 const std::vector<Column_info_t> &info,
150 int64_t *default_id,
151 size_t *valid_edges,
152 bool normal) {
153 Edge_t edge;
154 if (column_found(info[0].colNumber)) {
155 edge.id = getBigInt(tuple, tupdesc, info[0]);
156 } else {
157 edge.id = *default_id;
158 ++(*default_id);
159 }
160
161 if (normal) {
162 edge.source = getBigInt(tuple, tupdesc, info[1]);
163 edge.target = getBigInt(tuple, tupdesc, info[2]);
164 } else {
165 edge.target = getBigInt(tuple, tupdesc, info[1]);
166 edge.source = getBigInt(tuple, tupdesc, info[2]);
167 }
168
169 edge.cost = getFloat8(tuple, tupdesc, info[3]);
170
171 if (column_found(info[4].colNumber)) {
172 edge.reverse_cost = getFloat8(tuple, tupdesc, info[4]);
173 } else {
174 edge.reverse_cost = -1;
175 }
176
177 edge.cost = std::isinf(edge.cost)?
178 std::numeric_limits<double>::max() : edge.cost;
179
180 edge.reverse_cost = std::isinf(edge.reverse_cost)?
181 std::numeric_limits<double>::max() : edge.reverse_cost;
182
183 *valid_edges = edge.cost < 0? *valid_edges: *valid_edges + 1;
184 *valid_edges = edge.reverse_cost < 0? *valid_edges: *valid_edges + 1;
185
186 return edge;
187}

190CostFlow_t fetch_costFlow_edge(
191 const HeapTuple tuple,
192 const TupleDesc &tupdesc,
193 const std::vector<Column_info_t> &info,
194 int64_t *default_id,
195 size_t *valid_edges,
196 bool normal) {
197 CostFlow_t edge;
198 if (column_found(info[0].colNumber)) {
199 edge.edge_id = getBigInt(tuple, tupdesc, info[0]);
200 } else {
201 edge.edge_id = *default_id;
202 ++(*default_id);
203 }
204
205 if (normal) {
206 edge.source = getBigInt(tuple, tupdesc, info[1]);
207 edge.target = getBigInt(tuple, tupdesc, info[2]);
208 } else {
209 edge.target = getBigInt(tuple, tupdesc, info[1]);
210 edge.source = getBigInt(tuple, tupdesc, info[2]);
211 }
212
213 edge.capacity = getBigInt(tuple, tupdesc, info[3]);
214 if (column_found(info[4].colNumber)) {
215 edge.reverse_capacity = getBigInt(tuple, tupdesc, info[4]);
216 } else {
217 edge.reverse_capacity = -1;
218 }
219
220 edge.cost = getFloat8(tuple, tupdesc, info[5]);
221 if (column_found(info[6].colNumber)) {
222 edge.reverse_cost = getFloat8(tuple, tupdesc, info[6]);
223 } else {
224 edge.reverse_cost = 0;
225 }
226
227 *valid_edges = edge.capacity < 0? *valid_edges: *valid_edges + 1;
228 *valid_edges = edge.reverse_capacity < 0? *valid_edges: *valid_edges + 1;
229 return edge;
230}

290 const std::vector<Column_info_t> &info,
291 int64_t*,
292 size_t*,
293 bool with_id) {
294 Orders_t pd_order;
295 pd_order.id = getBigInt(tuple, tupdesc, info[0]);
296 pd_order.demand = getFloat8(tuple, tupdesc, info[1]);
297
298 /*
299 * the pickups
300 */
301 pd_order.pick_x = with_id ?
302 0 : getFloat8(tuple, tupdesc, info[2]);
303 pd_order.pick_y = with_id ?
304 0 : getFloat8(tuple, tupdesc, info[3]);
305 pd_order.pick_open_t = getFloat8(tuple, tupdesc, info[4]);
306 pd_order.pick_close_t = getFloat8(tuple, tupdesc, info[5]);
307 pd_order.pick_service_t = column_found(info[6].colNumber) ?
308 getFloat8(tuple, tupdesc, info[6]) : 0;
309
310 /*
311 * the deliveries
312 */
313 pd_order.deliver_x = with_id ?
314 0 : getFloat8(tuple, tupdesc, info[7]);
315 pd_order.deliver_y = with_id ?
316 0 : getFloat8(tuple, tupdesc, info[8]);
317 pd_order.deliver_open_t = getFloat8(tuple, tupdesc, info[9]);
318 pd_order.deliver_close_t = getFloat8(tuple, tupdesc, info[10]);
319 pd_order.deliver_service_t = column_found(info[11].colNumber) ?
320 getFloat8(tuple, tupdesc, info[11]) : 0;
321
322 pd_order.pick_node_id = with_id ?
323 getBigInt(tuple, tupdesc, info[12]) : 0;
324 pd_order.deliver_node_id = with_id ?
325 getBigInt(tuple, tupdesc, info[13]) : 0;
326 return pd_order;
327}

373Vehicle_t fetch_vehicle(
374 const HeapTuple tuple,
375 const TupleDesc &tupdesc,
376 const std::vector<Column_info_t> &info,
377 int64_t*,
378 size_t*,
379 bool with_id) {
380 Vehicle_t vehicle;
381 vehicle.id = getBigInt(tuple, tupdesc, info[0]);
382 vehicle.capacity = getFloat8(tuple, tupdesc, info[1]);
383
384 vehicle.start_x = with_id ?
385 0 :
386 getFloat8(tuple, tupdesc, info[2]);
387 vehicle.start_y = with_id ?
388 0 :
389 getFloat8(tuple, tupdesc, info[3]);
390
391 vehicle.speed = column_found(info[13].colNumber) ?
392 getFloat8(tuple, tupdesc, info[13]) :
393 1;
394 vehicle.cant_v = column_found(info[4].colNumber) ?
395 getBigInt(tuple, tupdesc, info[4]) :
396 1;
397 vehicle.start_open_t = column_found(info[5].colNumber) ?
398 getFloat8(tuple, tupdesc, info[5]) :
399 0;
400 vehicle.start_close_t = column_found(info[6].colNumber) ?
401 getFloat8(tuple, tupdesc, info[6]) :
402 std::numeric_limits<double>::max();
403 vehicle.start_service_t = column_found(info[7].colNumber) ?
404 getFloat8(tuple, tupdesc, info[7]) :
405 0;
406
407
408 if (!(column_found(info[8].colNumber))
409 && column_found(info[9].colNumber)) {
410 ereport(ERROR,
411 (errmsg("Column \'%s\' not Found", info[8].name.c_str()),
412 errhint("%s was found, also column is expected %s ",
413 info[9].name.c_str(), info[8].name.c_str())));
414 }
415 if (column_found(info[8].colNumber)
416 && !(column_found(info[9].colNumber))) {
417 ereport(ERROR,
418 (errmsg("Column \'%s\' not Found", info[9].name.c_str()),
419 errhint("%s was found, also column is expected %s ",
420 info[8].name.c_str(), info[9].name.c_str())));
421 }
422
423 vehicle.end_x = column_found(info[8].colNumber) ?
424 getFloat8(tuple, tupdesc, info[8]) :
425 vehicle.start_x;
426 vehicle.end_y = column_found(info[9].colNumber) ?
427 getFloat8(tuple, tupdesc, info[9]) :
428 vehicle.start_y;
429
430 if (!(column_found(info[10].colNumber))
431 && column_found(info[11].colNumber)) {
432 ereport(ERROR,
433 (errmsg("Column \'%s\' not Found", info[10].name.c_str()),
434 errhint("%s was found, also column is expected %s ",
435 info[10].name.c_str(), info[11].name.c_str())));
436 }
437
438 if (column_found(info[10].colNumber)
439 && !(column_found(info[11].colNumber))) {
440 ereport(ERROR,
441 (errmsg("Column \'%s\' not Found", info[11].name.c_str()),
442 errhint("%s was found, also column is expected %s ",
443 info[11].name.c_str(), info[10].name.c_str())));
444 }
445 vehicle.end_open_t = column_found(info[10].colNumber) ?
446 getFloat8(tuple, tupdesc, info[10]) :
447 vehicle.start_open_t;
448 vehicle.end_close_t = column_found(info[11].colNumber) ?
449 getFloat8(tuple, tupdesc, info[11]) :
450 vehicle.start_close_t;
451 vehicle.end_service_t = column_found(info[12].colNumber) ?
452 getFloat8(tuple, tupdesc, info[12]) :
453 vehicle.start_service_t;
454
455 vehicle.speed = column_found(info[13].colNumber) ?
456 getFloat8(tuple, tupdesc, info[13]) :
457 1;
458 vehicle.start_node_id = with_id ?
459 getBigInt(tuple, tupdesc, info[14]) :
460 0;
461 vehicle.end_node_id = with_id ?
462 (column_found(info[12].colNumber) ?
463 getBigInt(tuple, tupdesc, info[15]) :
464 vehicle.start_node_id) :
465 0;
466 return vehicle;
467}
