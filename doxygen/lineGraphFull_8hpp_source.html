226 void apply_transformation(
227 const pgrouting::DirectedGraph& digraph) {
228 V_i vertexIt, vertexEnd;
229 EO_i e_outIt, e_outEnd;
230 EI_i e_inIt, e_inEnd;
231
232 // For every vertex in the original graph, create a line graph
233 // using the edges connected to that vertex
234 for (boost::tie(vertexIt, vertexEnd) = boost::vertices(digraph.graph);
235 vertexIt != vertexEnd; vertexIt++) {
236 V vertex = *vertexIt;
237 auto vertex_id = digraph[vertex].id;
238 for (boost::tie(e_outIt, e_outEnd) =
239 boost::out_edges(vertex, digraph.graph);
240 e_outIt != e_outEnd; e_outIt++) {
241 auto out_edge_id = digraph.graph[*e_outIt].id;
242 insert_vertex(vertex_id, out_edge_id);
243 }
244
245 for (boost::tie(e_inIt, e_inEnd) =
246 boost::in_edges(vertex, digraph.graph);
247 e_inIt != e_inEnd; e_inIt++) {
248 auto in_edge_id = digraph.graph[*e_inIt].id;
249 insert_vertex(vertex_id, in_edge_id);
250
251 for (boost::tie(e_outIt, e_outEnd) =
252 boost::out_edges(vertex, digraph.graph);
253 e_outIt != e_outEnd; e_outIt++) {
254 auto out_edge_id = digraph.graph[*e_outIt].id;
255
256 ++m_num_edges;
257
258 graph_add_edge(
259 m_num_edges,
260 vertex_id,
261 vertex_id,
262 in_edge_id,
263 out_edge_id);
264 }
265 }
266 }
267
268 // Connect all of the line graphs that were just created using the
269 // edges from the original graph
270 for (boost::tie(vertexIt, vertexEnd) =
271 boost::vertices(digraph.graph);
272 vertexIt != vertexEnd; vertexIt++) {
273 V vertex = *vertexIt;
274 auto vertex_id = digraph[vertex].id;
275
276 for (boost::tie(e_inIt, e_inEnd) =
277 boost::in_edges(vertex, digraph.graph);
278 e_inIt != e_inEnd; e_inIt++) {
279 auto source_vertex_id = digraph[digraph.source(*e_inIt)].id;
280 auto in_edge_id = digraph.graph[*e_inIt].id;
281
282 ++m_num_edges;
283
284 graph_add_edge(
285 m_num_edges,
286 source_vertex_id,
287 vertex_id,
288 in_edge_id,
289 in_edge_id);
290 }
291 }
