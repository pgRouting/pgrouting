108PgrDirectedChPPGraph::PgrDirectedChPPGraph(const std::vector<Edge_t> &dataEdges) :
109 totalDeg(0), totalCost(0), vertices(),
110 edgeToIdx(), originalEdges(),
111 resultGraph(), VToVecid(), edgeVisited(),
112 pathStack(), resultPath(),
113 edges(), sources(), targets() {
114 pgassert(pathStack.empty());
115
116 pgassert(originalEdges.empty());
117 startPoint = dataEdges[0].source;
118 for (const auto e : dataEdges) {
119 if (e.cost > 0) {
120 auto edge = e;
121 edge.reverse_cost = -1.0;
122 totalCost += edge.cost;
123 originalEdges.push_back(edge);
124 vertices += e.source;
125 vertices += e.target;
126 }
127 if (e.reverse_cost > 0) {
128 auto edge = e;
129 std::swap(edge.source, edge.target);
130 std::swap(edge.cost, edge.reverse_cost);
131 edge.reverse_cost = -1.0;
132 totalCost += edge.cost;
133 originalEdges.push_back(edge);
134 vertices += e.source;
135 vertices += e.target;
136 pgassert(e.source == edge.target);
137 pgassert(e.target == edge.source);
138 }
139 }
140
141 // calcu deg & build part of edges
142 std::map<int64_t, int> deg;
143 for (const auto &e : originalEdges) {
144 pgassert(e.cost > 0);
145 /* has out going edge */
146 deg[e.source]++;
147 /* has out incoming edge */
148 deg[e.target]--;
149
150 auto current_edge(std::make_pair(e.source, e.target));
151 if (edgeToIdx.find(current_edge) == edgeToIdx.end()) {
152 edgeToIdx[current_edge] = &e;
153 } else {
154 if (edgeToIdx[current_edge]->cost > e.cost) {
155 edgeToIdx[current_edge] = &e;
156 }
157 }
158
159 CostFlow_t edge;
160 edge.edge_id = e.id;
161 edge.reverse_capacity = -1;
162 edge.reverse_cost = -1.0;
163 edge.source = e.source;
164 edge.target = e.target;
165 edge.capacity = (std::numeric_limits<int32_t>::max)();
166 edge.cost = e.cost;
167 edges.push_back(edge);
168 }
169
170 superSource = deg.rbegin()->first + 1;
171 superTarget = deg.rbegin()->first + 2;
172
173 sources.insert(superSource);
174 targets.insert(superTarget);
175
176 // build full edges
177 std::map<int64_t, int>::iterator iter;
178 totalDeg = 0;
179 for (iter = deg.begin(); iter != deg.end(); ++iter) {
180 int64_t p = iter->first;
181 int d = iter->second;
182 if (d == 0) continue;
183 if (d > 0) totalDeg += d;
184 CostFlow_t edge = {};
185 edge.reverse_capacity = -1;
186 edge.reverse_cost = -1.0;
187 edge.cost = 0.0;
188 edge.capacity = std::abs(d);
189 edge.edge_id = 0;
190 if (d > 0) {
191 edge.source = p, edge.target = superTarget;
192 }
193 if (d < 0) {
194 edge.source = superSource, edge.target = p;
195 }
196 edges.push_back(edge);
197 }
198
199 PgrCostFlowGraph flowGraph(edges, sources, targets);
200 pgassert(pathStack.empty());
201 try {
202 pgrouting::graph::PgrCostFlowGraph digraph1(edges, sources, targets);
203 auto minAddedCost = digraph1.MinCostMaxFlow();
204 auto maxFlow = digraph1.GetMaxFlow();
205 m_cost = (maxFlow == totalDeg)? minAddedCost + totalCost : -1.0;
206 } catch (...) {
207 m_cost = -1;
208 }
209 setPathEdges(flowGraph);
210}

213void
214PgrDirectedChPPGraph::setPathEdges(graph::PgrCostFlowGraph &flowGraph) {
215 pgassert(pathStack.empty());
216 resultPath.clear();
217 if (m_cost == -1) return;
218 // catch new edges
219 try {
220 flowGraph.MinCostMaxFlow();
221 flowGraph.GetMaxFlow();
222 std::vector<Flow_t> addedEdges = flowGraph.GetFlowEdges();
223 resultEdges = originalEdges;
224 for (auto &flow_t : addedEdges) {
225 if (flow_t.source != superSource && flow_t.source != superTarget
226 && flow_t.target != superSource && flow_t.target != superTarget) {
227 auto current_edge(std::make_pair(flow_t.source, flow_t.target));
228 Edge_t newEdge = *edgeToIdx[current_edge];
229 /* adding edges that need to be traversed twice */
230 while (flow_t.flow--) resultEdges.push_back(newEdge);
231 }
232 }
233 } catch (...) {
234 resultPath.clear();
235 return;
236 }
237
238 pgassert(pathStack.empty());
239 BuildResultGraph();
240
241 pgassert(pathStack.empty());
242 pgassert(edgeVisited.empty());
243
244
245 EulerCircuitDFS(startPoint);
246
247 if (!(vertices - vertexVisited).empty()) {
248 resultPath.clear();
249 return;
250 }
251 pgassert(!pathStack.empty());
252
253 BuildResultPath();
254
255
256 return;
257}

259void
260PgrDirectedChPPGraph::BuildResultPath() {
261 if (pathStack.empty()) return;
262 pgassert(resultPath.empty());
263
264 int64_t preNode = pathStack.top();
265 pathStack.pop();
266
267 Path_rt newElement = {};
268 while (!pathStack.empty()) {
269 int64_t nowNode = pathStack.top();
270 pathStack.pop();
271
272 auto edge_t = *edgeToIdx[std::make_pair(preNode, nowNode)];
273 newElement.node = edge_t.source;
274 newElement.edge = edge_t.id;
275 newElement.cost = edge_t.cost;
276 if (resultPath.empty()) {
277 /* adding the first row because is a cycle */
278 newElement.agg_cost = 0.0;
279 } else {
280 newElement.agg_cost = resultPath.back().agg_cost + resultPath.back().cost;
281 }
282 resultPath.push_back(newElement);
283 preNode = nowNode;
284 }
285 newElement.node = preNode;
286 newElement.edge = -1;
287 newElement.cost = 0;
288 if (resultPath.empty()) {
289 newElement.agg_cost = 0.0;
290 } else {
291 newElement.agg_cost =
292 resultPath.back().agg_cost + resultPath.back().cost;
293 }
294 resultPath.push_back(newElement);
295}

297// perform DFS approach to generate Euler circuit
298// TODO(mg) find suitable API in BGL, maybe DfsVisitor will work.

300bool
301PgrDirectedChPPGraph::EulerCircuitDFS(int64_t vertex) {
302 for (const auto e : resultGraph[VToVecid[vertex]].second) {
303 if (!edgeVisited.has(e)) {
304 edgeVisited += e;
305 EulerCircuitDFS(resultEdges[e].target);
306 }
307 }
308 pathStack.push(vertex);
309 vertexVisited += vertex;
310 return true;
311}

313void
314PgrDirectedChPPGraph::BuildResultGraph() {
315 pgassert(resultGraph.empty());
316 pgassert(VToVecid.empty());
317 resultGraph.clear();
318 VToVecid.clear();
319 size_t e_id(0);
320 for (const auto &edge_t : resultEdges) {
321 if (VToVecid.find(edge_t.source) == VToVecid.end()) {
322 VToVecid[edge_t.source] = resultGraph.size();
323 resultGraph.resize(resultGraph.size() + 1);
324 }
325 size_t vid = VToVecid[edge_t.source];
326 resultGraph[vid].second.push_back(e_id);
327 resultGraph[vid].first = edge_t.source;
328 ++e_id;
329 }
330}
