46std::set<int64_t>
47maxCardinalityMatch(pgrouting::graph::UndirectedNoCostsBG &graph) {
48 using G = pgrouting::graph::UndirectedNoCostsBG::G;
49 using V = pgrouting::graph::UndirectedNoCostsBG::V;
50 using E = pgrouting::graph::UndirectedNoCostsBG::E;
51
52 std::vector<V> mate_map(boost::num_vertices(graph()));
53 std::set<int64_t> match;
54
55 CHECK_FOR_INTERRUPTS();
56 try {
57 edmonds_maximum_cardinality_matching(graph(), &mate_map[0]);
58 } catch (boost::exception const& ex) {
59 (void)ex;
60 throw;
61 } catch (std::exception &e) {
62 (void)e;
63 throw;
64 } catch (...) {
65 throw;
66 }
67
68
69 /*
70 * Check for each vertex:
71 * 1) The vertex does not have a match
72 * 2) prevent double output of the edge
73 */
74 for (const auto &v2 : mate_map) {
75 auto v1 = static_cast<V>(&v2 - &mate_map[0]);
76
77 if (v2 == boost::graph_traits<G>::null_vertex()) continue;
78 if (v1 >= v2) continue;
79
80 E e;
81 bool exists;
82 boost::tie(e, exists) = boost::edge(v1, v2, graph());
83 if (!exists) throw;
84
85 match.insert(graph.get_edge_id(e));
86 }
87
88 return match;
89}
