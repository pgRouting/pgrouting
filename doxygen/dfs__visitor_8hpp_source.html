45template <typename V, typename E, typename G>
46class Dfs_visitor : public boost::default_dfs_visitor {
47 public:
48 Dfs_visitor(
49 V root,
50 std::vector<E> &data,
51 int64_t max_depth,
52 std::vector<boost::default_color_type> &colors,
53 G &graph) :
54 m_roots(root),
55 m_data(data),
56 m_max_depth(max_depth),
57 m_colors(colors),
58 m_graph(graph) {
59 m_depth.resize(m_graph.num_vertices(), 0);
60 }
61 template <typename B_G>
62 void start_vertex(V v, const B_G&) {
63 // exception for visitor termination
64 if (v != m_roots) throw found_goals();
65 m_depth[v] = 0;
66 }
67 template <typename B_G>
68 void examine_edge(E e, const B_G&) {
69 auto source = m_graph.source(e), target = m_graph.target(e);
70 // If the target has not been visited before
71 if (m_depth[target] == 0 && target != m_roots) {
72 m_depth[target] = m_depth[source] + 1;
73 }
74
75 // If the max_depth is reached, mark the target as visited, and push the corresponding edge
76 if (m_depth[target] == m_max_depth && m_colors[target] != 4) {
77 m_colors[target] = boost::black_color;
78 m_data.push_back(e);
79 }
80 }
81 template <typename B_G>
82 void tree_edge(E e, const B_G&) {
83 m_data.push_back(e);
84 }
85
86 private:
87 V m_roots;
88 std::vector<E> &m_data;
89 int64_t m_max_depth;
90 std::vector<boost::default_color_type> &m_colors;
91 G &m_graph;
92 std::vector<int64_t> m_depth;
93};
