82TrspHandler::renumber_edges(std::vector<Edge_t> &edges) {
83 int64_t idx(0);
84 for (auto &e : edges) {
85 if (m_id_to_idx.find(e.source) == m_id_to_idx.end()) {
86 m_id_to_idx[e.source] = idx;
87 ++idx;
88 }
89 if (m_id_to_idx.find(e.target) == m_id_to_idx.end()) {
90 m_id_to_idx[e.target] = idx;
91 ++idx;
92 }
93 e.source = m_id_to_idx.at(e.source);
94 e.target = m_id_to_idx.at(e.target);
95 }

141double TrspHandler::construct_path(int64_t ed_id, Position pos) {
142 if (pos == ILLEGAL) return (std::numeric_limits<double>::max)();
143 pgassert(pos != ILLEGAL);
144
145 if (m_parent[static_cast<size_t>(ed_id)].isIllegal(pos)) {
146 Path_t pelement = {};
147 auto cur_edge = &m_edges[static_cast<size_t>(ed_id)];
148 if (pos == RC_EDGE) {
149 pelement.node = cur_edge->startNode();
150 pelement.cost = cur_edge->cost();
151 } else {
152 pelement.node = cur_edge->endNode();
153 pelement.cost = cur_edge->r_cost();
154 }
155 pelement.edge = cur_edge->edgeID();
156
157 m_path.push_back(pelement);
158 pgassert(m_path.start_id() == m_start_vertex);
159 return pelement.cost;
160 }
161
162 double ret = construct_path(
163 static_cast<int64_t>(m_parent[static_cast<size_t>(ed_id)].e_idx[static_cast<size_t>(pos)]),
164 static_cast<Position>(m_parent[static_cast<size_t>(ed_id)].v_pos[static_cast<size_t>(pos)]));
165 Path_t pelement = {};
166 auto cur_edge = &m_edges[static_cast<size_t>(ed_id)];
167 if (pos == RC_EDGE) {
168 pelement.node = cur_edge->startNode();
169 pelement.cost = m_dCost[static_cast<size_t>(ed_id)].endCost - ret;
170 ret = m_dCost[static_cast<size_t>(ed_id)].endCost;
171 } else {
172 pelement.node = cur_edge->endNode();
173 pelement.cost = m_dCost[static_cast<size_t>(ed_id)].startCost - ret;
174 ret = m_dCost[static_cast<size_t>(ed_id)].startCost;
175 }
176 pelement.edge = cur_edge->edgeID();
177
178 m_path.push_back(pelement);
179
180 return ret;
181}

184// -------------------------------------------------------------------------
185double TrspHandler::getRestrictionCost(
186 int64_t edge_ind,
187 const EdgeInfo &edge,
188 bool isStart) {
189 double cost = 0.0;
190 int64_t edge_id = edge.edgeID();
191 if (m_ruleTable.find(edge_id) == m_ruleTable.end()) {
192 return(0.0);
193 }
194 auto vecRules = m_ruleTable[edge_id];
195 int64_t st_edge_ind = edge_ind;
196 for (const auto &rule : vecRules) {
197 bool flag = true;
198 int64_t v_pos = (isStart? C_EDGE : RC_EDGE);
199 edge_ind = st_edge_ind;
200
201 pgassert(!(edge_ind == -1));
202 for (auto const &precedence : rule.precedencelist()) {
203 if (precedence != m_edges[static_cast<size_t>(edge_ind)].edgeID()) {
204 flag = false;
205 break;
206 }
207 auto m_parent_ind = m_parent[static_cast<size_t>(edge_ind)].e_idx[static_cast<size_t>(v_pos)];
208 v_pos = m_parent[static_cast<size_t>(edge_ind)].v_pos[static_cast<size_t>(v_pos)];
209 edge_ind = static_cast<int64_t>(m_parent_ind);
210 }
211 cost += flag? rule.cost() : 0;
212 }
213 return cost;
214}

230// -------------------------------------------------------------------------
231void TrspHandler::explore(
232 int64_t cur_node,
233 const EdgeInfo cur_edge,
234 bool isStart) {
235 double totalCost;
236
237 auto vecIndex = cur_edge.get_idx(isStart);
238
239 for (const auto &index : vecIndex) {
240 auto edge = m_edges[index];
241
242 auto extra_cost = getRestrictionCost(
243 static_cast<int64_t>(cur_edge.idx()),
244 edge, isStart);
245
246 if ((edge.startNode() == cur_node) && (edge.cost() >= 0.0)) {
247 totalCost = get_tot_cost(
248 edge.cost() + extra_cost,
249 cur_edge.idx(),
250 isStart);
251
252 if (totalCost < m_dCost[index].endCost) {
253 m_dCost[index].endCost = totalCost;
254 m_parent[edge.idx()].v_pos[RC_EDGE] = isStart? C_EDGE : RC_EDGE;
255 m_parent[edge.idx()].e_idx[RC_EDGE] =
256 cur_edge.idx();
257
258 add_to_que(totalCost, edge.idx(), true);
259 }
260 }
261
262 if ((edge.endNode() == cur_node) && (edge.r_cost() >= 0.0)) {
263 totalCost = get_tot_cost(
264 edge.r_cost() + extra_cost,
265 cur_edge.idx(),
266 isStart);
267
268 if (totalCost < m_dCost[index].startCost) {
269 m_dCost[index].startCost = totalCost;
270 m_parent[edge.idx()].v_pos[C_EDGE] = isStart? C_EDGE : RC_EDGE;
271 m_parent[edge.idx()].e_idx[C_EDGE] = cur_edge.idx();
272
273 add_to_que(totalCost, edge.idx(), false);
274 }
275 }
276 } // for
277}

282int TrspHandler::initialize_restrictions(
283 const std::vector<Rule> &ruleList) {
284 for (const auto &rule : ruleList) {
285 auto dest_edge_id = rule.dest_id();
286 if (m_ruleTable.find(dest_edge_id) != m_ruleTable.end()) {
287 m_ruleTable[dest_edge_id].push_back(rule);
288 } else {
289 std::vector<Rule> r;
290 r.push_back(rule);
291 m_ruleTable.insert(std::make_pair(dest_edge_id, r));
292 }
293 }
294
295 return true;

303Path
304TrspHandler::process(
305 const int64_t start_vertex,
306 const int64_t end_vertex) {
307 if (m_id_to_idx.find(start_vertex) == m_id_to_idx.end()
308 || m_id_to_idx.find(end_vertex) == m_id_to_idx.end()) {
309 return Path();
310 }
311 clear();
312 pgassert(m_id_to_idx.find(start_vertex) != m_id_to_idx.end());
313 pgassert(m_id_to_idx.find(end_vertex) != m_id_to_idx.end());
314
315 m_start_vertex = m_id_to_idx.at(start_vertex);
316 m_end_vertex = m_id_to_idx.at(end_vertex);
317 pgassert(m_idx_to_id.at(m_start_vertex) == start_vertex);
318 pgassert(m_idx_to_id.at(m_end_vertex) == end_vertex);
319
320 Path tmp(m_start_vertex, m_end_vertex);
321 m_path = tmp;
322
323 if (m_adjacency.find(m_start_vertex) == m_adjacency.end()) {
324 return Path();
325 }
326
327 if (m_adjacency.find(m_end_vertex) == m_adjacency.end()) {
328 return Path();
329 }
330
331 return process_trsp(m_edges.size());
332}

359void TrspHandler::initialize_que() {
360 /*
361 * For each adjacent edge to the start_vertex
362 */
363 for (const auto source : m_adjacency[m_start_vertex]) {
364 EdgeInfo &cur_edge = m_edges[source];
365
366 if (cur_edge.startNode() == m_start_vertex && cur_edge.cost() >= 0.0) {
367 m_dCost[cur_edge.idx()].endCost = cur_edge.cost();
368 m_parent[cur_edge.idx()].v_pos[0] = ILLEGAL;
369 add_to_que(cur_edge.cost(), cur_edge.idx(), true);
370 }
371
372 if (cur_edge.endNode() == m_start_vertex && cur_edge.r_cost() >= 0.0) {
373 m_dCost[cur_edge.idx()].startCost = cur_edge.r_cost();
374 m_parent[cur_edge.idx()].v_pos[1] = ILLEGAL;
375 add_to_que(cur_edge.r_cost(), cur_edge.idx(), false);
376 }
377 }
378}

380EdgeInfo TrspHandler::dijkstra_exploration() {
381 EdgeInfo cur_edge;
382 pgassert(m_current_node == m_start_vertex);
383
384 while (!m_que.empty()) {
385 auto cur_pos = m_que.top();
386 m_que.pop();
387
388 auto cure_idxex = cur_pos.second.first;
389 cur_edge = m_edges[static_cast<size_t>(cure_idxex)];
390
391 if (cur_pos.second.second) {
392 /*
393 * explore edges connected to end node
394 */
395 m_current_node = cur_edge.endNode();
396 if (cur_edge.cost() < 0.0) continue;
397 if (m_current_node == m_end_vertex) break;
398 explore(m_current_node, cur_edge, false);
399 } else {
400 /*
401 * explore edges connected to start node
402 */
403 m_current_node = cur_edge.startNode();
404 if (cur_edge.r_cost() < 0.0) continue;
405 if (m_current_node == m_end_vertex) break;
406 explore(m_current_node, cur_edge, true);
407 }
408 }
409 return cur_edge;
410}

415Path
416TrspHandler::process_trsp(
417 size_t edge_count) {
418 pgassert(m_path.start_id() == m_start_vertex);
419 pgassert(m_path.end_id() == m_end_vertex);
420 pgassert(m_parent.empty());
421
422 m_parent.resize(edge_count + 1);
423 m_dCost.resize(edge_count + 1);
424
425 initialize_que();
426
427 m_current_node = m_start_vertex;
428
429 pgassert(m_path.start_id() == m_start_vertex);
430
431 auto cur_edge = dijkstra_exploration();
432
433 pgassert(m_path.start_id() == m_start_vertex);
434 if (m_current_node != m_end_vertex) {
435 Path result(m_start_vertex, m_end_vertex);
436 return result.renumber_vertices(m_idx_to_id);;
437 }
438
439 pgassert(m_path.start_id() == m_start_vertex);
440
441 construct_path(static_cast<int64_t>(cur_edge.idx()), m_current_node == cur_edge.startNode()? C_EDGE : RC_EDGE);
442
443 Path_t pelement = {};
444 pelement.node = m_end_vertex;
445 pelement.edge = -1;
446 pelement.cost = 0.0;
447 m_path.push_back(pelement);
448
449 m_path.Path::recalculate_agg_cost();
450 return m_path.renumber_vertices(m_idx_to_id);
451}

470void TrspHandler::connectStartEdge(
471 size_t firstEdge_idx,
472 size_t secondEdge_idx) {
473 EdgeInfo &firstEdge = m_edges[firstEdge_idx];
474 EdgeInfo &secondEdge = m_edges[secondEdge_idx];
475
476 if (firstEdge.r_cost() >= 0.0) {
477 firstEdge.connect_startEdge(secondEdge_idx);
478 }
479
480 if (firstEdge.startNode() == secondEdge.startNode()
481 && secondEdge.r_cost() >= 0.0) {
482 secondEdge.connect_startEdge(firstEdge_idx);
483 }
484
485 if (firstEdge.startNode() == secondEdge.endNode()
486 &&secondEdge.cost() >= 0.0) {
487 secondEdge.connect_endEdge(firstEdge_idx);
488 }
489}

515// -------------------------------------------------------------------------
516bool TrspHandler::addEdge(Edge_t edgeIn, bool directed) {
517 /*
518 * making all "cost" > 0
519 */
520 if (edgeIn.cost < 0 && edgeIn.reverse_cost > 0) {
521 std::swap(edgeIn.cost, edgeIn.reverse_cost);
522 std::swap(edgeIn.source, edgeIn.target);
523 }
524
525 /* This gives a value to reverse_cost when:"
526 * graph is undirected
527 * reverse_cost is negative
528 */
529 if (!directed && edgeIn.reverse_cost < 0) {
530 edgeIn.reverse_cost = edgeIn.cost;
531 }
532
533 /*
534 * the index of this new edge in the edges container is
535 * m_edges.size()
536 */
537 EdgeInfo edge(edgeIn, m_edges.size());
538
539 /*
540 * Add to the set of edges
541 */
542 m_edges.push_back(edge);
543 EdgeInfo &newEdge = m_edges[m_edges.size()-1];
544
545 /*
546 * Searching the start node for connectivity
547 */
548 auto itNodeMap = m_adjacency.find(edgeIn.source);
549
550 if (itNodeMap != m_adjacency.end()) {
551 for (const auto &e_idx : itNodeMap->second) {
552 connectStartEdge(edge.idx(), e_idx);
553 }
554 }
555
556
557 /*
558 * Searching the end node for connectivity
559 */
560 itNodeMap = m_adjacency.find(edgeIn.target);
561 if (itNodeMap != m_adjacency.end()) {
562 for (const auto &e_idx : itNodeMap->second) {
563 connectEndEdge(edge.idx(), e_idx);
564 }
565 }
566
567
568 /*
569 * Add the edges to the adjacency list
570 */
571 m_adjacency[edgeIn.source].push_back(newEdge.idx());
572 m_adjacency[edgeIn.target].push_back(newEdge.idx());
573
574
575
576 return true;
577}
