54template <typename Data_type, typename Func>
55void get_data(
56 char *sql,
57 Data_type **pgtuples,
58 size_t *total_pgtuples,
59 bool flag,
60 std::vector<Column_info_t> info,
61 Func func) {
62 const int tuple_limit = 1000000;
63
64 size_t total_tuples;
65 size_t valid_pgtuples;
66
67 auto SPIplan = pgr_SPI_prepare(sql);
68 auto SPIportal = pgr_SPI_cursor_open(SPIplan);
69
70 bool moredata = true;
71 (*total_pgtuples) = total_tuples = valid_pgtuples = 0;
72
73 int64_t default_id = 0;
74
75 while (moredata == true) {
76 SPI_cursor_fetch(SPIportal, true, tuple_limit);
77 auto tuptable = SPI_tuptable;
78 auto tupdesc = SPI_tuptable->tupdesc;
79 if (total_tuples == 0) fetch_column_info(tupdesc, info);
80
81 size_t ntuples = SPI_processed;
82 total_tuples += ntuples;
83
84 if (ntuples > 0) {
85 (*pgtuples) = pgr_alloc(total_tuples, *pgtuples);
86 if ((*pgtuples) == NULL) {
87 throw std::string("Out of memory!");
88 }
89
90 for (size_t t = 0; t < ntuples; t++) {
91 func(tuptable->vals[t], tupdesc, info,
92 &default_id,
93 &(*pgtuples)[total_tuples - ntuples + t],
94 &valid_pgtuples, flag);
95 }
96 SPI_freetuptable(tuptable);
97 } else {
98 moredata = false;
99 }
100 }
101
102 SPI_cursor_close(SPIportal);
103 (*total_pgtuples) = total_tuples;
104}

116std::vector<Data_type> get_data(
117 const std::string& sql,
118 bool flag,
119 std::vector<Column_info_t> info,
120 Func func) {
121 const int tuple_limit = 1000000;
122
123 size_t total_tuples;
124 size_t valid_pgtuples;
125
126 auto SPIplan = pgr_SPI_prepare(sql.c_str());
127 auto SPIportal = pgr_SPI_cursor_open(SPIplan);
128
129 bool moredata = true;
130 total_tuples = valid_pgtuples = 0;
131
132 int64_t default_id = 0;
133 std::vector<Data_type> tuples;
134
135 while (moredata == true) {
136 SPI_cursor_fetch(SPIportal, true, tuple_limit);
137 auto tuptable = SPI_tuptable;
138 auto tupdesc = SPI_tuptable->tupdesc;
139 if (total_tuples == 0) fetch_column_info(tupdesc, info);
140
141 size_t ntuples = SPI_processed;
142 total_tuples += ntuples;
143
144 if (ntuples > 0) {
145 tuples.reserve(total_tuples);
146 for (size_t t = 0; t < ntuples; t++) {
147 tuples.push_back(func(tuptable->vals[t], tupdesc, info,
148 &default_id,
149 &valid_pgtuples, flag));
150 }
151 SPI_freetuptable(tuptable);
152 } else {
153 moredata = false;
154 }
155 }
156
157 SPI_cursor_close(SPIportal);
158 return tuples;
159}
