81size_t
82get_viaRoute(
83 std::deque<pgrouting::Path> &paths,
84 Routes_t **tuples) {
85 pgassert(!(*tuples));
86
87 auto count = count_tuples(paths);
88 if (count == 0) return 0;
89
90 (*tuples) = pgr_alloc(count, (*tuples));
91
92 size_t sequence = 0;
93 int path_id = 1;
94 int route_id = 1;
95 double route_agg_cost = 0;
96 for (auto &p : paths) {
97 p.recalculate_agg_cost();
98 }
99 for (const auto &path : paths) {
100 if (path.size() > 0) {
101 ::get_path(route_id, path_id, path, tuples, route_agg_cost, sequence);
102 }
103 ++path_id;
104 }
105 (*tuples)[count - 1].edge = -2;
106
107 pgassert(count == sequence);
108
109 return sequence;
110}
