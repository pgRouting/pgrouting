45 size_t check_vertices(
46 std::vector < Basic_vertex > vertices) {
47 auto count(vertices.size());
48 std::stable_sort(
49 vertices.begin(), vertices.end(),
50 [](const Basic_vertex &lhs, const Basic_vertex &rhs)
51 {return lhs.id < rhs.id;});
52 vertices.erase(
53 std::unique(
54 vertices.begin(), vertices.end(),
55 [](const Basic_vertex &lhs, const Basic_vertex &rhs)
56 {return lhs.id == rhs.id;}), vertices.end());
57
58 return vertices.size() - count;
59 }
