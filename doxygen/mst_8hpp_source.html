173 template <typename T>
174 std::vector<MST_rt> get_results(
175 T order,
176 int64_t p_root,
177 const G &graph) {
178 std::vector<MST_rt> results;
179
180 std::vector<double> agg_cost(graph.num_vertices(), 0);
181 std::vector<int64_t> depth(graph.num_vertices(), 0);
182 int64_t root(p_root);
183
184 for (const auto edge : order) {
185 auto u = graph.source(edge);
186 auto v = graph.target(edge);
187 if (depth[u] == 0 && depth[v] != 0) {
188 std::swap(u, v);
189 }
190
191 auto component = m_get_component? m_tree_id[m_components[u]] : 0;
192 if (m_suffix != "" && depth[u] == 0 && depth[v] == 0) {
193 if (!m_roots.empty() && graph[u].id != root) std::swap(u, v);
194 if (m_roots.empty() && graph[u].id != component) std::swap(u, v);
195 if (!p_root && graph[u].id > graph[v].id) std::swap(u, v);
196
197 root = p_root? p_root: graph[u].id;
198 depth[u] = -1;
199 results.push_back({
200 root,
201 0,
202 graph[u].id,
203 graph[u].id,
204 -1,
205 0.0,
206 0.0 });
207 }
208
209 agg_cost[v] = agg_cost[u] + graph[edge].cost;
210 depth[v] = depth[u] == -1? 1 : depth[u] + 1;
211
212 if ((m_suffix == "")
213 || ((m_suffix == "BFS") && (m_max_depth >= depth[v]))
214 || ((m_suffix == "DFS") && m_max_depth >= depth[v])
215 || ((m_suffix == "DD") && m_distance >= agg_cost[v])) {
216 results.push_back({
217 root,
218 m_suffix != ""? depth[v] : 0,
219 graph[u].id,
220 graph[v].id,
221 graph[edge].id,
222 graph[edge].cost,
223 m_suffix != ""? agg_cost[v] : 0.0
224 });
225 }
226 }
227 return results;
228 }

231 void calculate_component(const G &graph) {
232 if (!m_get_component) return;
233
234 m_components.resize(num_vertices(graph.graph));
235
236 /*
237 * Calculate connected components
238 *
239 * Number of components of graph: num_comps components
240 */
241 auto num_comps = boost::connected_components(
242 graph.graph,
243 &m_components[0]);
244
245 m_tree_id.resize(num_comps, 0);
246
247 for (const auto v : boost::make_iterator_range(vertices(graph.graph))) {
248 m_tree_id[m_components[v]] =
249 (m_tree_id[m_components[v]] == 0
250 || m_tree_id[m_components[v]] >= graph[v].id) ?
251 graph[v].id : m_tree_id[m_components[v]];
252 }
253 }

260 std::vector<MST_rt>
261 dfs_forest(const G &graph) {
262 boost::filtered_graph<B_G, InSpanning, boost::keep_all>
263 mstGraph(graph.graph, m_spanning_tree, {});
264 std::vector<E> visited_order;
265
266 using dfs_visitor = visitors::Edges_order_dfs_visitor<E>;
267 /* abort in case of an interruption occurs (e.g. the query is being cancelled) */
268 CHECK_FOR_INTERRUPTS();
269 try {
270 boost::depth_first_search(mstGraph, visitor(dfs_visitor(visited_order)));
271 } catch (boost::exception const& ex) {
272 (void)ex;
273 throw;
274 } catch (std::exception &e) {
275 (void)e;
276 throw;
277 } catch (...) {
278 throw;
279 }
280
281 return get_results(visited_order, 0, graph);
282 }

285 std::vector<MST_rt>
286 dfs_ordering(const G &graph) {
287 boost::filtered_graph<B_G, InSpanning, boost::keep_all>
288 mstGraph(graph.graph, m_spanning_tree, {});
289
290 if (m_roots.empty()) {
291 return dfs_forest(graph);
292 } else {
293 std::vector<MST_rt> results;
294 for (const auto root : m_roots) {
295 using dfs_visitor = visitors::Dfs_visitor_with_root<V, E>;
296 if (graph.has_vertex(root)) {
297 std::vector<E> visited_order;
298 CHECK_FOR_INTERRUPTS();
299 try {
300 boost::depth_first_search(
301 mstGraph,
302 visitor(dfs_visitor(graph.get_V(root), visited_order))
303 .root_vertex(graph.get_V(root)));
304 } catch(found_goals &) {
305 {}
306 } catch (boost::exception const& ex) {
307 (void)ex;
308 throw;
309 } catch (std::exception &e) {
310 (void)e;
311 throw;
312 } catch (...) {
313 throw;
314 }
315 auto result = get_results(visited_order, root, graph);
316 results.insert(results.end(), result.begin(), result.end());
317 } else {
318 results.push_back({root, 0, root, root, -1, 0.0, 0.0});
319 }
320 }
321 return results;
322 }

325 std::vector<MST_rt>
326 bfs_ordering(const G &graph) {
327 std::vector<MST_rt> results;
328 /*
329 * order by bfs
330 */
331 boost::filtered_graph<B_G, InSpanning, boost::keep_all>
332 mst(graph.graph, m_spanning_tree, {});
333
334 calculate_component(graph);
335
336 std::vector<int64_t> roots;
337 if (!m_roots.empty()) {
338 roots = m_roots;
339 } else {
340 roots = m_tree_id;
341 }
342
343 using bfs_visitor = visitors::Edges_order_bfs_visitor<E>;
344 for (auto root : roots) {
345 std::vector<E> visited_order;
346 if (graph.has_vertex(root)) {
347 boost::breadth_first_search(mst,
348 graph.get_V(root),
349 visitor(bfs_visitor(visited_order)));
350
351 auto tree_results = get_results(visited_order, root, graph);
352 results.insert(results.end(), tree_results.begin(), tree_results.end());
353 } else {
354 results.push_back({root, 0, root, root, -1, 0.0, 0.0});
355 }
356 /* abort in case of an interruption occurs (e.g. the query is being cancelled) */
357 CHECK_FOR_INTERRUPTS();
358 }
359
360 return results;
361 }

363 bool no_neg_costs(const G &graph) {
364 E_i ei, ei_end;
365 for (boost::tie(ei, ei_end) = edges(graph.graph); ei != ei_end; ++ei) {
366 pgassert(graph[*ei].cost > 0);
367 }
368 return true;
369 }
