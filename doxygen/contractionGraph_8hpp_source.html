99 get_min_cost_edge(V u, V v) {
100 Identifiers<int64_t> contracted_vertices;
101 double min_cost = (std::numeric_limits<double>::max)();
102 bool found = false;
103 CH_edge edge;
104
105 if (this->is_directed()) {
106 for (const auto &e : boost::make_iterator_range(out_edges(u, this->graph))) {
107 if (target(e, this->graph) == v) {
108 contracted_vertices += this->graph[e].contracted_vertices();
109 if (this->graph[e].cost < min_cost) {
110 min_cost = this->graph[e].cost;
111 edge = this->graph[e];
112 found = true;
113 }
114 }
115 }
116
117 /*
118 To follow the principles presented
119 for linear contraction in "issue_1002.pg" test 3
120 */
121 edge.set_contracted_vertices(contracted_vertices);
122 return std::make_tuple(edge, found);
123 }
124
125 pgassert(this->is_undirected());
126 for (const auto &e : boost::make_iterator_range(out_edges(u, this->graph))) {
127 if (this->adjacent(u, e) == v) {
128 contracted_vertices += this->graph[e].contracted_vertices();
129 if ((this->graph[e]).cost < min_cost) {
130 min_cost = (this->graph[e]).cost;
131 edge = this->graph[e];
132 found = true;
133 }
134 }
135 }
136 // To follow the principles presented
137 // for linear contraction in "issue_1002.pg" test 3
138 edge.set_contracted_vertices(contracted_vertices);
139 return std::make_tuple(edge, found);

222 bool is_shortcut_possible(
223 V u,
224 V v,
225 V w) {
226 if (u == v || v == w || u == w) return false;
227 pgassert(u != v);
228 pgassert(v != w);
229 pgassert(u != w);
230 if (this->is_undirected()) {
231 /*
232 * u - v - w
233 */
234 return has_u_v_w(u, v, w);
235 }
236
237 pgassert(this->is_directed());
238 return
239 /*
240 * u <-> v <-> w
241 */
242 (has_u_v_w(u, v, w) && has_u_v_w(w, v, u))
243 /*
244 * u -> v -> w
245 */
246 ||
247 (has_u_v_w(u, v, w) && !(boost::edge(v, u, this->graph).second || boost::edge(w, v, this->graph).second))
248 /*
249 * u <- v <- w
250 */
251 ||
252 (has_u_v_w(w, v, u) && !(boost::edge(v, w, this->graph).second || boost::edge(u, v, this->graph).second));
253 }

255 bool is_linear(V v) {
256 // Checking adjacent vertices constraint
257 auto adjacent_vertices = find_adjacent_vertices(v);
258
259 if (adjacent_vertices.size() == 2) {
260 // Checking u - v - w
261 V u = adjacent_vertices.front();
262 adjacent_vertices.pop_front();
263 V w = adjacent_vertices.front();
264 adjacent_vertices.pop_front();
265 if (is_shortcut_possible(u, v, w)) {
266 return true;
267 }
268 return false;
269 }
270 return false;
271 }

369 CH_edge process_shortcut(V u, V v, V w) {
370 auto e1 = get_min_cost_edge(u, v);
371 auto e2 = get_min_cost_edge(v, w);
372
373 double cost = std::numeric_limits<double>::max();
374 if (std::get<1>(e1) && std::get<1>(e2)) {
375 cost = std::get<0>(e1).cost + std::get<0>(e2).cost;
376 }
377
378 // Create shortcut
379 CH_edge shortcut(
380 get_next_id(),
381 (this->graph[u]).id,
382 (this->graph[w]).id,
383 cost);
384 shortcut.add_contracted_vertex(this->graph[v]);
385 shortcut.add_contracted_edge_vertices(std::get<0>(e1));
386 shortcut.add_contracted_edge_vertices(std::get<0>(e2));
387
388 // Add shortcut in the current graph (to go on the process)
389 add_shortcut(shortcut, u, w);
390
391 return shortcut;
392 }

436 int64_t compute_pmax(V u, V v, Identifiers<V> out_vertices) {
437 int64_t p_max;
438 E e, f;
439 bool found_e;
440 p_max = 0;
441 boost::tie(e, found_e) = boost::edge(u, v, this->graph);
442
443 if (found_e) {
444 p_max = this->graph[e].cost;
445 for (V w : out_vertices) {
446 bool found_f;
447 boost::tie(f, found_f) = boost::edge(v, w, this->graph);
448 if ((found_f) && (u != w)) {
449 if ((this->graph[e].cost + this->graph[f].cost) > p_max) {
450 p_max = this->graph[e].cost + this->graph[f].cost;
451 }
452 }
453 }
454 }
455 return p_max;
