52void
53Pgr_pickDeliver::solve() {
54 auto initial_sols = solutions;
55
56 if (m_initial_id == 0) {
57 msg.log << "trying all \n";
58 for (int i = 1; i < 7; ++i) {
59 initial_sols.push_back(Initial_solution((Initials_code)i, m_orders.size()));
60 msg.log << "solution " << i << "\n" << initial_sols.back().tau();
61 // TODO(vicky) calculate the time it takes
62 msg.log << "Initial solution " << i
63 << " duration: " << initial_sols.back().duration();
64 }
65 } else {
66 msg.log << "only trying " << m_initial_id << "\n";
67 initial_sols.push_back(Initial_solution((Initials_code)m_initial_id, m_orders.size()));
68 // TODO(vicky) calculate the time it takes
69 msg.log << "Initial solution " << m_initial_id
70 << " duration: " << initial_sols[0].duration();
71 }
72
73
74 /*
75 * Sorting solutions: the best is at the back
76 */
77 pgassert(!initial_sols.empty());
78 std::sort(initial_sols.begin(), initial_sols.end(), []
79 (const Solution &lhs, const Solution &rhs) -> bool {
80 return rhs < lhs;
81 });
82
83 solutions.push_back(Optimize(initial_sols.back(), m_max_cycles));
84 pgassert(!solutions.empty());
85
86 msg.log << "best solution duration = " << solutions.back().duration();
87}

91std::vector< Schedule_rt >
92Pgr_pickDeliver::get_postgres_result() const {
93 auto result = solutions.back().get_postgres_result();
94
95 Schedule_rt aggregates = {
96 /*
97 * Vehicle id = -2 indicates its an aggregate row
98 *
99 * (twv, cv, fleet, wait, duration)
100 */
101 -2, // summary row on vehicle_number
102 solutions.back().twvTot(), // on vehicle_id
103 solutions.back().cvTot(), // on vehicle_seq
104 -1, // on order_id
105 -1, // on stop_id
106 -2, // on stop_type (gets increased later by one so it gets -1)
107 -1, // not accounting total loads
108 solutions.back().total_travel_time(),
109 -1, // not accounting arrival_travel_time
110 solutions.back().wait_time(),
111 solutions.back().total_service_time(),
112 solutions.back().duration(),
113 };
114 result.push_back(aggregates);
115
116
117#ifndef NDEBUG
118 for (const auto &sol : solutions) {
119 msg.log << sol.tau();
120 }
121#endif
122 return result;
123}

133Pgr_pickDeliver::Pgr_pickDeliver(
134 const std::vector<Orders_t> &pd_orders,
135 const std::vector<Vehicle_t> &vehicles,
136 const pgrouting::tsp::Dmatrix &cost_matrix,
137 double factor,
138 size_t p_max_cycles,
139 int initial) :
140 PD_problem(this),
141 m_initial_id(initial),
142 m_max_cycles(p_max_cycles),
143 m_nodes(),
144 m_cost_matrix(cost_matrix),
145 m_orders(pd_orders),
146 m_trucks(vehicles, factor) {
147 ENTERING(msg);
148 pgassert(!pd_orders.empty());
149 pgassert(!vehicles.empty());
150 pgassert(!m_cost_matrix.empty());
151 if (!(m_initial_id > 0 && m_initial_id < OneDepot)) {
152 msg.log << "\n m_initial_id " << m_initial_id;
153 }
154 pgassertwm(m_initial_id > 0 && m_initial_id <= OneDepot, msg.get_log().c_str());
155 pgassert(!m_nodes.empty());
156
157 if (!msg.get_error().empty()) {
158 return;
159 }
160
161
162 msg.log << "\n Checking fleet ...";
163 if (!m_trucks.is_fleet_ok()) {
164 pgassert(msg.get_error().empty());
165 return;
166 }
167 msg.log << "fleet OK \n";
168
169
170 /*
171 * check the (S, P, D, E) order on all vehicles
172 * stop when a feasible truck is found
173 */
174 msg.log << "\n Checking orders";
175 for (const auto &o : m_orders) {
176 if (!m_trucks.is_order_ok(o)) {
177 msg.error << "Order not feasible on any truck was found";
178 msg.log << "The order "
179 << o.id()
180 << " is not feasible on any truck";
181 msg.log << "\n" << o;
182 return;
183 }
184 }
185 msg.log << "orders OK \n";
186
187 m_trucks.set_compatibles(m_orders);
188 EXITING(msg);
189 } // constructor
