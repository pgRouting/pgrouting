48double
49Tw_node::arrival_j_opens_i(const Tw_node &I, double speed) const {
50 if (m_type == kStart) return (std::numeric_limits<double>::max)();
51 return I.opens() + I.service_time() + I.travel_time_to(*this, speed);

58bool
59Tw_node::is_compatible_IJ(const Tw_node &I, double speed) const {
60 /*
61 * I /-> kStart
62 */
63 if (m_type == kStart) return false;
64 /*
65 * kEnd /-> (*this)
66 */
67 if (I.m_type == kEnd) return false;
68
69 return !is_late_arrival(arrival_j_opens_i(I, speed));
70}

150Tw_node::Tw_node(
151 size_t id,
152 Orders_t data,
153 NodeType type) :
154 Dnode(id, data.pick_node_id),
155 m_order(data.id),
156 m_opens(data.pick_open_t),
157 m_closes(data.pick_close_t),
158 m_service_time(data.pick_service_t),
159 m_demand(data.demand),
160 m_type(type) {
161 if (m_type == kDelivery) {
162 reset_id(data.deliver_node_id);
163 m_opens = data.deliver_open_t;
164 m_closes = data.deliver_close_t;
165 m_service_time = data.deliver_service_t;
166 m_demand *= -1;
167 }
168 }
