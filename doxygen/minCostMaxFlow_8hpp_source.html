104 /*
105 * vertices = {sources} U {sink} U {edges.source} U {edge.target}
106 */
107 template <typename T>
108 void AddVertices(
109 const T &edges,
110 const std::set<int64_t> &source_vertices,
111 const std::set<int64_t> &sink_vertices) {
112 std::set<int64_t> vertices(source_vertices);
113 vertices.insert(sink_vertices.begin(), sink_vertices.end());
114
115 for (const auto e : edges) {
116 vertices.insert(e.source);
117 vertices.insert(e.target);
118 }
119
120 for (const auto id : vertices) {
121 V v = add_vertex(graph);
122 idToV.insert(std::pair<int64_t, V>(id, v));
123 vToId.insert(std::pair<V, int64_t>(v, id));
124 }
125
126 SetSupersource(source_vertices);
127 SetSupersink(sink_vertices);
128 }

133 std::map<V, int64_t> vToId;
134 std::map<E, int64_t> eToId;
