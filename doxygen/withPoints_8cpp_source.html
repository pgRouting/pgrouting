112Pg_points_graph::check_points() {
113 log << "original points" << *this;
114 /*
115 * deleting duplicate points
116 */
117 std::sort(m_points.begin(), m_points.end(),
118 [](const Point_on_edge_t &a, const Point_on_edge_t &b)
119 -> bool {
120 if (a.pid != b.pid) return a.pid < b.pid;
121 if (a.edge_id != b.edge_id) return a.edge_id < b.edge_id;
122 if (a.fraction != b.fraction) return a.fraction < b.fraction;
123 return a.side < b.side;
124 });
125 log << "after sorting" << *this;
126 auto last = std::unique(m_points.begin(), m_points.end(),
127 [](const Point_on_edge_t &a, const Point_on_edge_t &b) {
128 return a.pid == b.pid &&
129 a.edge_id == b.edge_id &&
130 a.fraction == b.fraction &&
131 a.side == b.side;
132 });
133 m_points.erase(last, m_points.end());
134 size_t total_points = m_points.size();
135
136 log << "after deleting repetitions" << *this;
137 log << "We have " << total_points << " different points";
138
139 last = std::unique(m_points.begin(), m_points.end(),
140 [](const Point_on_edge_t &a, const Point_on_edge_t &b) {
141 return a.pid == b.pid;
142 });
143 m_points.erase(last, m_points.end());
144 log << "after deleting points with same id" << *this;
145
146 if (m_points.size() != total_points) {
147 error << "Unexpected point(s) with same pid"
148 << " but different edge/fraction/side combination found.";
149 }

208void
209Pg_points_graph::eliminate_details_dd(
210 Path &path) const {
211 /*
212 * There is no path nothing to do
213 */
214 if (path.empty()) return;
215
216 /* TODO maybe look for duplicate edges and remove the ones that node < 0 */
217 /*
218 * do not remove
219 * - first node of path
220 * - node of path that is reachable but not the next node on the edge
221 */
222 Path newPath(path.start_id(), path.end_id());
223 for (const auto &ps : path) {
224 if (ps.node == path.start_id() || (ps.node >= 0)) {
225 newPath.push_back(ps);
226 continue;
227 }
228 auto edge_ptr = get_edge_data(ps.edge);
229 pgassert(edge_ptr);
230
231 auto s_exists = std::find_if(path.begin(), path.end(),
232 [&](const Path_t& x) { return x.node == edge_ptr->source;}) != path.end();
233 auto t_exists = std::find_if(path.begin(), path.end(),
234 [&](const Path_t& x) { return x.node == edge_ptr->target;}) != path.end();
235 if (s_exists && t_exists) continue;
236 newPath.push_back(ps);
237 }
238
239 path = newPath;
240}

244Path
245Pg_points_graph::eliminate_details(
246 Path path) const {
247 /*
248 * There is no path nothing to do
249 */
250 if (path.empty()) return path;
251 path.recalculate_agg_cost();
252
253 Path newPath(path.start_id(), path.end_id());
254
255 double cost = 0.0;
256 auto current_node = path[0].node;
257 auto current_edge = path[0].edge;
258
259 for (const auto &pathstop : path) {
260 if (pathstop.edge == current_edge) {
261 cost += pathstop.cost;
262 continue;
263 }
264 /* TODO(v4) add correct predecessor node */
265 newPath.push_back({current_node, current_edge, cost, 0, 0});
266 cost = pathstop.cost;
267 current_node = pathstop.node;
268 current_edge = pathstop.edge;
269 }
270 /* TODO(v4) add correct predecessor node */
271 newPath.push_back({current_node, current_edge, cost, 0, 0});
272 newPath.recalculate_agg_cost();
273
274 return newPath;
275}

303 * There is no path nothing to do

309 int64_t start_vid = path.start_id();
310 int64_t end_vid = path.end_id();
311
312 int64_t start_pid = 0;
313 int64_t end_pid = 0;
314
315 for (auto &p : points) {
316 if (p.vertex_id == start_vid) {
317 start_pid = -p.pid;
318 }
319 if (p.vertex_id == end_vid) {
320 end_pid = -p.pid;
321 }
322 }
323 adjust_pids(points, start_pid, end_pid, path);
