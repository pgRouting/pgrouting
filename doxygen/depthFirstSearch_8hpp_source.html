115 bool depthFirstSearch_single_vertex(
116 G &graph,
117 V root,
118 std::vector < E > &visited_order,
119 bool directed,
120 int64_t max_depth) {
121 using dfs_visitor = visitors::Dfs_visitor < V, E, G >;
122
123 // Exterior property storage containers
124 std::vector < boost::default_color_type > colors(boost::num_vertices(graph.graph));
125 std::map < E, boost::default_color_type > edge_color;
126
127 auto i_map = boost::get(boost::vertex_index, graph.graph);
128
129 // Iterator property maps which record the color of each vertex and edge, respectively
130 auto vertex_color_map = boost::make_iterator_property_map(colors.begin(), i_map);
131 auto edge_color_map = boost::make_assoc_property_map(edge_color);
132
133 auto vis = dfs_visitor(root, visited_order, max_depth, colors, graph);
134
135 /* abort in case of an interruption occurs (e.g. the query is being cancelled) */
136 CHECK_FOR_INTERRUPTS();
137
138 try {
139 if (directed) {
140 boost::depth_first_search(graph.graph, vis, vertex_color_map, root);
141 } else {
142 boost::undirected_dfs(graph.graph, vis, vertex_color_map, edge_color_map, root);
143 }
144 } catch(found_goals &) {
145 {}
146 } catch (boost::exception const& ex) {
147 (void)ex;
148 throw;
149 } catch (std::exception &e) {
150 (void)e;
151 throw;
152 } catch (...) {
153 throw;
154 }
155 return true;
156 }

171 template < typename T >
172 std::vector < MST_rt > get_results(
173 T visited_order,
174 int64_t root,
175 int64_t max_depth,
176 const G &graph) {
177 std::vector < MST_rt > results;
178
179 std::vector < double > agg_cost(graph.num_vertices(), 0);
180 std::vector < int64_t > depth(graph.num_vertices(), 0);
181
182 for (const auto edge : visited_order) {
183 auto u = graph.source(edge);
184 auto v = graph.target(edge);
185
186 agg_cost[v] = agg_cost[u] + graph[edge].cost;
187 depth[v] = depth[u] + 1;
188
189 if (max_depth >= depth[v]) {
190 results.push_back({
191 root,
192 depth[v],
193 graph[u].id,
194 graph[v].id,
195 graph[edge].id,
196 graph[edge].cost,
197 agg_cost[v]
198 });
199 }
200 }
201 return results;
202 }
