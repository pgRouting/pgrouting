136 /*
137  * .. to ... from ....
138  */
139 bool
140 Optimize::swap_worse(Vehicle_pickDeliver &to, Vehicle_pickDeliver &from) {
141  /*
142  * working on a copy of the vehicles
143  */
144  auto from_truck = from;
145  auto to_truck = to;
146 
147  auto swapped = false;
148 
149  /*
150  * To avoid invalidation of cycles
151  */
152  auto o_from_orders = from_truck.orders_in_vehicle();
153  auto o_to_orders = to_truck.orders_in_vehicle();
154 
155  pgassert((o_from_orders * o_to_orders).empty());
156 
157  for (auto from_orders = from_truck.orders_in_vehicle();
158  !from_orders.empty();
159  from_orders.pop_front()) {
160  auto from_order = from_truck.orders()[from_orders.front()];
161 
162  pgassert(from_truck.has_order(from_order));
163 
164  if (move_order(from_order, from_truck, to_truck)) {
165  pgassert(!from_truck.has_order(from_order));
166  pgassert(to_truck.has_order(from_order));
167  /*
168  * The order could be moved to "to" truck
169  */
170  to = to_truck;
171  from = from_truck;
172  /*
173  * go to next order
174  */
175  pgassert(!from.has_order(from_order));
176  pgassert(to.has_order(from_order));
177  pgassert(!from_truck.has_order(from_order));
178  pgassert(to_truck.has_order(from_order));
179  pgassert(!(from.has_order(from_order) && to.has_order(from_order)));
180  pgassert(from.has_order(from_order) || to.has_order(from_order));
181  continue;
182  }
183 
184  pgassert(from_truck.has_order(from_order));
185  auto curr_from_duration = from_truck.duration();
186 
187  for (auto to_order_id : o_to_orders) {
188  auto to_order = to.orders()[to_order_id];
189  /*
190  * The orders might have being swapped before
191  */
192  if (!to_truck.has_order(to_order)) continue;
193  // if (!from_truck.has_order(from_order)) break; // TODO delete line
194 
195  pgassert(from_truck.has_order(from_order));
196  pgassert(to_truck.has_order(to_order));
197 
198  auto curr_to_duration = to_truck.duration();
199 
200  /*
201  * delete from_order, and to order from their trucks
202  */
203  pgassert(from_truck.has_order(from_order));
204  from_truck.erase(from_order);
205  pgassert(to_truck.has_order(to_order));
206  to_truck.erase(to_order);
207 
208  /*
209  * insert them in the other truck
210  */
211  if (this->get_kind() == OneDepot) {
212  from_truck.semiLIFO(to_order);
213  to_truck.semiLIFO(from_order);
214  } else {
215  from_truck.insert(to_order);
216  to_truck.insert(from_order);
217  }
218 
219  pgassert((from_truck.has_order(to_order) && from_truck.is_feasable()) || !from_truck.has_order(to_order));
220  pgassert((to_truck.has_order(from_order) && to_truck.is_feasable()) || !to_truck.has_order(from_order));
221 
222  if (from_truck.has_order(to_order) && to_truck.has_order(from_order)) {
223  auto new_from_duration = from_truck.duration();
224  auto new_to_duration = to_truck.duration();
225 
226  auto estimated_delta =
227  - (curr_from_duration + curr_to_duration)
228  + (new_to_duration + new_from_duration);
229 
230  auto estimated_duration = duration() + estimated_delta;
231 
232  /*
233  * Can swap when:
234  * - or from_truck duration is reduced
235  * - the total fleet duration is reduced
236  * - or the new fleet duration is better than best_solution duration
237  */
238  if (new_from_duration < curr_from_duration ||
239  estimated_delta < 0 ||
240  estimated_duration < best_solution.duration()) {
241  /*
242  * this acctually makes the swapping
243  */
244  to = to_truck;
245  from = from_truck;
246 
247  pgassert(!(from.has_order(from_order) && to.has_order(from_order)));
248  pgassert(from.has_order(from_order) || to.has_order(from_order));
249  pgassert(!(from.has_order(to_order) && to.has_order(to_order)));
250  pgassert(from.has_order(to_order) || to.has_order(to_order));
251 
252  swapped = true;
253  break;
254  }
255  }
256  /*
257  * Can't swap, restore vehicles
258  */
259  to_truck = to;
260  from_truck = from;
261 
262  pgassert(!(from.has_order(from_order) && to.has_order(from_order)));
263  pgassert(from.has_order(from_order) || to.has_order(from_order));
264  pgassert(!(from.has_order(to_order) && to.has_order(to_order)));
265  pgassert(from.has_order(to_order) || to.has_order(to_order));
266  }
267  }
268 
269  return false && swapped;
270 }

321 /*
322  * from_truck: position of the truck where the order is
323  * to truck: truck to put the order
324  */
325 bool
326 Optimize::swap_order(
327  const Order from_order,
328  Vehicle_pickDeliver &from_truck,
329  const Order to_order,
330  Vehicle_pickDeliver &to_truck) {
331  if (!from_truck.has_order(from_order)
332  || !to_truck.has_order(to_order)) {
333  return false;
334  }
335 
336  pgassert(from_truck.has_order(from_order));
337  pgassert(to_truck.has_order(to_order));
338 
339  from_truck.erase(from_order);
340  to_truck.erase(to_order);
341 
342  /*
343  * insert them in the other truck
344  */
345  if (this->get_kind() == OneDepot) {
346  from_truck.semiLIFO(to_order);
347  to_truck.semiLIFO(from_order);
348  } else {
349  from_truck.insert(to_order);
350  to_truck.insert(from_order);
351  }
352 
353  pgassert((from_truck.has_order(to_order) && from_truck.is_feasable()) || !from_truck.has_order(to_order));
354  pgassert((to_truck.has_order(from_order) && to_truck.is_feasable()) || !to_truck.has_order(from_order));
355 
356  if (!from_truck.has_order(to_order) || !to_truck.has_order(from_order)) {
357  /*
358  * swap generates a violation: restore trucks
359  */
360  if (from_truck.has_order(to_order)) from_truck.erase(to_order);
361  if (to_truck.has_order(from_order)) to_truck.erase(from_order);
362 
363  pgassert(!from_truck.has_order(to_order));
364  pgassert(!from_truck.has_order(from_order));
365  pgassert(!to_truck.has_order(from_order));
366  pgassert(!to_truck.has_order(to_order));
367 
368  /*
369  * insert them again in the truck
370  */
371  if (this->get_kind() == OneDepot) {
372  from_truck.semiLIFO(from_order);
373  to_truck.semiLIFO(to_order);
374  } else {
375  from_truck.insert(from_order);
376  to_truck.insert(to_order);
377  }
378  pgassert((from_truck.has_order(from_order) && from_truck.is_feasable()) || !from_truck.has_order(from_order));
379  pgassert((to_truck.has_order(to_order) && to_truck.is_feasable()) || !to_truck.has_order(to_order));
380 
381  return false;
382  }
383 
384  pgassert(from_truck.has_order(to_order));
385  pgassert(to_truck.has_order(from_order));
386  return true;
387 }

508 /*
509  * from_truck trying to make from_truck's duration smaller
510  * - maybe all orders can be moved
511  * - if that is the case, the from_truck could be removed
512  *
513  * Deleting an order on the from_truck
514  * - number of truck remains the same
515  * - from_truk duration() can not get larger
516  * - the overall duration can get larger
517  *
518  */
519 bool
520 Optimize::move_reduce_cost(
521  Vehicle_pickDeliver &from,
522  Vehicle_pickDeliver &to) {
523  pgassert(&from != &to);
524 
525  auto from_truck = from;
526  auto to_truck = to;
527  /*
528  * don't move to empty truck
529  */
530  if (to_truck.empty()) return false;
531 
532  /*
533  * don't move from a real truck to a phoney truck
534  */
535  if (!from_truck.is_phony() && to_truck.is_phony()) {
536  return false;
537  }
538 
539  auto moved = false;
540 
541  auto from_orders = from_truck.orders_in_vehicle();
542  for (const auto o_id : from_orders) {
543  /*
544  * removing an order decreases the duration
545  */
546  auto order = from_truck.orders()[o_id];
547 
548  auto curr_duration = from_truck.duration() + to_truck.duration();
549  /*
550  * insert it in the "to" truck
551  */
552  pgassert(!to_truck.has_order(order));
553  this->get_kind() == OneDepot?
554  to_truck.semiLIFO(order) :
555  to_truck.insert(order);
556  pgassert((to_truck.has_order(order) && to_truck.is_feasable()) || !to_truck.has_order(order));
557 
558  if (to_truck.has_order(order)) {
559  from_truck.erase(order);
560 
561  auto new_duration = from_truck.duration() + to_truck.duration();
562 
563  /*
564  * cost is reduced
565  */
566  if (new_duration < curr_duration
567  || from_truck.empty()
568  || new_duration < best_solution.duration()) {
569  moved = true;
570  save_if_best();
571  continue;
572  }
573 
574  /*
575  * cost is not reduced
576  * revert changes
577  */
578  to_truck.erase(order);
579  this->get_kind() == OneDepot?
580  from_truck.semiLIFO(order) :
581  from_truck.insert(order);
582  }
583  }
584  return moved;
585 }

602 Optimize::move_order(
603  Order order,
604  Vehicle_pickDeliver &from_truck,
605  Vehicle_pickDeliver &to_truck) {
606  pgassert(from_truck.has_order(order));
607  pgassert(!to_truck.has_order(order));
608  /*
609  * don't move to empty truck
610  */
611  if (to_truck.empty()) return false;
612 
613  /*
614  * don't move from a real truck to a phoney truck
615  */
616  if (!from_truck.is_phony() && to_truck.is_phony()) return false;
617 
618  /*
619  * Don't move from a vehicle with more orders
620  */
621  if (from_truck.size() > to_truck.size()) return false;
622 
623  /*
624  * insert the order
625  */
626  this->get_kind() == OneDepot?
627  to_truck.semiLIFO(order) :
628  to_truck.insert(order);
629 
630  if (to_truck.has_order(order)) {
631  from_truck.erase(order);
632 
633  pgassert(!(from_truck.has_order(order) && to_truck.has_order(order)));
634  pgassert(!from_truck.has_order(order));
635  pgassert(to_truck.has_order(order));
636  return true;
637  }
638 
639  pgassert(!(from_truck.has_order(order) && to_truck.has_order(order)));
640  pgassert(from_truck.has_order(order));
641  pgassert(!to_truck.has_order(order));
642  return false;

662  * Optimize decreasing truck
663  *
664  * - Objective: try to remove truck with less duration
665  * - Secundary objective, acts like a shake operation
666  *
667  */
668 void
669 Optimize::decrease_truck() {
670  bool decreased(false);
671  for (size_t i = 1; i < fleet.size(); ++i) {
672  decreased = decrease_truck(i) || decreased;
673  }
674  if (decreased) {
675  delete_empty_truck();
676  save_if_best();
677  decrease_truck();
678  }
679  save_if_best();
680 }

682 bool
683 Optimize::decrease_truck(size_t cycle) {
684  auto position = cycle;
685  for (auto orders = fleet[position].orders_in_vehicle();
686  !orders.empty();
687  orders.pop_front()) {
688  /* Step 2: grab an order */
689  auto order = fleet[position].orders()[orders.front()];
690  pgassert(order.idx() == orders.front());
691 
692 
693  /* Step 3:
694  * cycle the fleet
695  * insert in first truck possible
696  */
697 
698  for (size_t i = 0; i < position; ++i) {
699  fleet[i].insert(order);
700  pgassert((fleet[i].has_order(order) && fleet[i].is_feasable()) || !fleet[i].has_order(order));
701  if (fleet[i].has_order(order)) {
702  /*
703  * delete the order from the current truck
704  */
705  fleet[position].erase(order);
706  break;
707  }
708  }
709  }
710  return fleet[position].orders_in_vehicle().empty();
711 }
