64void
65Optimize::inter_swap(size_t times) {
66 msg().log << tau("before sort by size");
67 sort_by_size();
68 msg().log << tau("before decrease");
69 decrease_truck();
70 msg().log << tau("after decrease");
71 sort_by_size();
72 msg().log << tau("after sort by size");
73
74 size_t i = 0;
75 while (i++ < times) {
76 msg().log << "\n*************************** CYCLE" << i;
77 inter_swap();
78 msg().log << tau("after inter swap");
79 std::rotate(fleet.begin(), fleet.begin() + 1, fleet.end());
80 msg().log << tau("before next cycle");
81 }
82}

84/*
85 * ordering the trucks by number of orders it has
86 * - the truck with more orders wants to:
87 * - make less time
88 * - do more orders
89 * The inter_swap objective is to make the trucks with more orders
90 * - less time consuming
91 */
92bool
93Optimize::inter_swap() {
94 msg().log
95 << "\n" <<tau("before inter swap");
96 delete_empty_truck();
97 auto swapped_f = false;
98 /*
99 * .. to ... from ....
100 */
101 for (auto &from : fleet) {
102 for (auto &to : fleet) {
103 if (&from == &to) break;
104
105 swap_worse(to, from);
106 move_reduce_cost(from, to);
107 }
108 }
109
110 msg().log
111 << "\n" <<tau("after");
112 delete_empty_truck();
113
114 return swapped_f;
115}

122bool
123Optimize::swap_worse(Vehicle_pickDeliver &to, Vehicle_pickDeliver &from) {
124 /*
125 * working on a copy of the vehicles
126 */
127 auto from_truck = from;
128 auto to_truck = to;
129
130 auto swapped = false;
131
132 /*
133 * To avoid invalidation of cycles
134 */
135 auto o_from_orders = from_truck.orders_in_vehicle();
136 auto o_to_orders = to_truck.orders_in_vehicle();
137
138 pgassert((o_from_orders * o_to_orders).empty());
139
140 for (auto from_orders = from_truck.orders_in_vehicle();
141 !from_orders.empty();
142 from_orders.pop_front()) {
143 auto from_order = from_truck.orders()[from_orders.front()];
144
145 pgassert(from_truck.has_order(from_order));
146
147 if (move_order(from_order, from_truck, to_truck)) {
148 pgassert(!from_truck.has_order(from_order));
149 pgassert(to_truck.has_order(from_order));
150 /*
151 * The order could be moved to "to" truck
152 */
153 to = to_truck;
154 from = from_truck;
155 /*
156 * go to next order
157 */
158 pgassert(!from.has_order(from_order));
159 pgassert(to.has_order(from_order));
160 pgassert(!from_truck.has_order(from_order));
161 pgassert(to_truck.has_order(from_order));
162 pgassert(!(from.has_order(from_order) && to.has_order(from_order)));
163 pgassert(from.has_order(from_order) || to.has_order(from_order));
164 continue;
165 }
166
167 pgassert(from_truck.has_order(from_order));
168 auto curr_from_duration = from_truck.duration();
169
170 for (auto to_order_id : o_to_orders) {
171 auto to_order = to.orders()[to_order_id];
172 /*
173 * The orders might have being swapped before
174 */
175 if (!to_truck.has_order(to_order)) continue;
176 // if (!from_truck.has_order(from_order)) break; // TODO delete line
177
178 pgassert(from_truck.has_order(from_order));
179 pgassert(to_truck.has_order(to_order));
180
181 auto curr_to_duration = to_truck.duration();
182
183 /*
184 * delete from_order, and to order from their trucks
185 */
186 pgassert(from_truck.has_order(from_order));
187 from_truck.erase(from_order);
188 pgassert(to_truck.has_order(to_order));
189 to_truck.erase(to_order);
190
191 /*
192 * insert them in the other truck
193 */
194 if (this->get_kind() == OneDepot) {
195 from_truck.semiLIFO(to_order);
196 to_truck.semiLIFO(from_order);
197 } else {
198 from_truck.insert(to_order);
199 to_truck.insert(from_order);
200 }
201
202 pgassert((from_truck.has_order(to_order) && from_truck.is_feasable()) || !from_truck.has_order(to_order));
203 pgassert((to_truck.has_order(from_order) && to_truck.is_feasable()) || !to_truck.has_order(from_order));
204
205 if (from_truck.has_order(to_order) && to_truck.has_order(from_order)) {
206 auto new_from_duration = from_truck.duration();
207 auto new_to_duration = to_truck.duration();
208
209 auto estimated_delta =
210 - (curr_from_duration + curr_to_duration)
211 + (new_to_duration + new_from_duration);
212
213 auto estimated_duration = duration() + estimated_delta;
214
215 /*
216 * Can swap when:
217 * - or from_truck duration is reduced
218 * - the total fleet duration is reduced
219 * - or the new fleet duration is better than best_solution duration
220 */
221 if (new_from_duration < curr_from_duration ||
222 estimated_delta < 0 ||
223 estimated_duration < best_solution.duration()) {
224 /*
225 * this actually makes the swapping
226 */
227 to = to_truck;
228 from = from_truck;
229
230 pgassert(!(from.has_order(from_order) && to.has_order(from_order)));
231 pgassert(from.has_order(from_order) || to.has_order(from_order));
232 pgassert(!(from.has_order(to_order) && to.has_order(to_order)));
233 pgassert(from.has_order(to_order) || to.has_order(to_order));
234
235 swapped = true;
236 break;
237 }
238 }
239 /*
240 * Can't swap, restore vehicles
241 */
242 to_truck = to;
243 from_truck = from;
244
245 pgassert(!(from.has_order(from_order) && to.has_order(from_order)));
246 pgassert(from.has_order(from_order) || to.has_order(from_order));
247 pgassert(!(from.has_order(to_order) && to.has_order(to_order)));
248 pgassert(from.has_order(to_order) || to.has_order(to_order));
249 }
250 }
251
252 return swapped;
253}

289/*
290 * from_truck trying to make from_truck's duration smaller
291 * - maybe all orders can be moved
292 * - if that is the case, the from_truck could be removed
293 *
294 * Deleting an order on the from_truck
295 * - number of truck remains the same
296 * - from_truk duration() can not get larger
297 * - the overall duration can get larger
298 *
299 */
300bool
301Optimize::move_reduce_cost(
302 Vehicle_pickDeliver &from,
303 Vehicle_pickDeliver &to) {
304 pgassert(&from != &to);
305
306 auto from_truck = from;
307 auto to_truck = to;
308 /*
309 * don't move to empty truck
310 */
311 if (to_truck.empty()) return false;
312
313 /*
314 * don't move from a real truck to a phoney truck
315 */
316 if (!from_truck.is_phony() && to_truck.is_phony()) {
317 return false;
318 }
319
320 auto moved = false;
321
322 auto from_orders = from_truck.orders_in_vehicle();
323 for (const auto o_id : from_orders) {
324 /*
325 * removing an order decreases the duration
326 */
327 auto order = from_truck.orders()[o_id];
328
329 auto curr_duration = from_truck.duration() + to_truck.duration();
330 /*
331 * insert it in the "to" truck
332 */
333 pgassert(!to_truck.has_order(order));
334 this->get_kind() == OneDepot?
335 to_truck.semiLIFO(order) :
336 to_truck.insert(order);
337 pgassert((to_truck.has_order(order) && to_truck.is_feasable()) || !to_truck.has_order(order));
338
339 if (to_truck.has_order(order)) {
340 from_truck.erase(order);
341
342 auto new_duration = from_truck.duration() + to_truck.duration();
343
344 /*
345 * cost is reduced
346 */
347 if (new_duration < curr_duration
348 || from_truck.empty()
349 || new_duration < best_solution.duration()) {
350 moved = true;
351 save_if_best();
352 continue;
353 }
354
355 /*
356 * cost is not reduced
357 * revert changes
358 */
359 to_truck.erase(order);
360 this->get_kind() == OneDepot?
361 from_truck.semiLIFO(order) :
362 from_truck.insert(order);
363 }
364 }
365 return moved;
366}

382bool
383Optimize::move_order(
384 Order order,
385 Vehicle_pickDeliver &from_truck,
386 Vehicle_pickDeliver &to_truck) {
387 pgassert(from_truck.has_order(order));
388 pgassert(!to_truck.has_order(order));
389 /*
390 * don't move to empty truck
391 */
392 if (to_truck.empty()) return false;
393
394 /*
395 * don't move from a real truck to a phoney truck
396 */
397 if (!from_truck.is_phony() && to_truck.is_phony()) return false;
398
399 /*
400 * Don't move from a vehicle with more orders
401 */
402 if (from_truck.size() > to_truck.size()) return false;
403
404 /*
405 * insert the order
406 */
407 this->get_kind() == OneDepot?
408 to_truck.semiLIFO(order) :
409 to_truck.insert(order);
410
411 if (to_truck.has_order(order)) {
412 from_truck.erase(order);
413
414 pgassert(!(from_truck.has_order(order) && to_truck.has_order(order)));
415 pgassert(!from_truck.has_order(order));
416 pgassert(to_truck.has_order(order));
417 return true;
418 }
419
420 pgassert(!(from_truck.has_order(order) && to_truck.has_order(order)));
421 pgassert(from_truck.has_order(order));
422 pgassert(!to_truck.has_order(order));
423 return false;
424}

428/*
429 * Optimize decreasing truck
430 *
431 * - Objective: try to remove truck with less duration
432 * - Secondary objective, acts like a shake operation
433 *
434 */
435void
436Optimize::decrease_truck() {
437 bool decreased(false);
438 for (size_t i = 1; i < fleet.size(); ++i) {
439 decreased = decrease_truck(i) || decreased;
440 }
441 if (decreased) {
442 delete_empty_truck();
443 save_if_best();
444 decrease_truck();
445 }
446 save_if_best();
447}

449bool
450Optimize::decrease_truck(size_t cycle) {
451 auto position = cycle;
452 for (auto orders = fleet[position].orders_in_vehicle();
453 !orders.empty();
454 orders.pop_front()) {
455 /* Step 2: grab an order */
456 auto order = fleet[position].orders()[orders.front()];
457 pgassert(order.idx() == orders.front());
458
459
460 /* Step 3:
461 * cycle the fleet
462 * insert in first truck possible
463 */
464
465 for (size_t i = 0; i < position; ++i) {
466 fleet[i].insert(order);
467 pgassert((fleet[i].has_order(order) && fleet[i].is_feasable()) || !fleet[i].has_order(order));
468 if (fleet[i].has_order(order)) {
469 /*
470 * delete the order from the current truck
471 */
472 fleet[position].erase(order);
473 break;
474 }
475 }
476 }
477 return fleet[position].orders_in_vehicle().empty();
478}
