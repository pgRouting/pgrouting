38void
39pgr_send_error(int errcode) {
40 switch (errcode) {
41 case 1:
42 elog(ERROR, "Unexpected point(s) with same pid but different"
43 " edge/fraction/side combination found.");
44 break;
45 case 2:
46 elog(ERROR, "Internal: Unexpected mismatch "
47 "count and sequence number on results");
48 break;
49 default:
50 elog(ERROR, "Unknown exception");
51 }
52}

56// http://www.postgresql.org/docs/9.4/static/spi-spi-finish.html
57void
58pgr_SPI_finish(void) {
59 int code = SPI_finish();
60 if (code != SPI_OK_FINISH) { // SPI_ERROR_UNCONNECTED
61 elog(ERROR, "There was no connection to SPI");
62 }
63}

65void
66pgr_SPI_connect(void) {
67 int SPIcode;
68 SPIcode = SPI_connect();
69 if (SPIcode != SPI_OK_CONNECT) {
70 elog(ERROR, "Couldn't open a connection to SPI");
71 }
72}

74SPIPlanPtr
75pgr_SPI_prepare(const char* sql) {
76 SPIPlanPtr SPIplan;
77 SPIplan = SPI_prepare(sql, 0, NULL);
78 if (SPIplan == NULL) {
79 elog(ERROR, "Couldn't create query plan via SPI: %s", sql);
80 }
