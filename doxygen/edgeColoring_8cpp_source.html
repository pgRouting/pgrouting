85 * Inserting edges

89 auto v1 = get_boost_vertex(edge.source);

92 // NOLINTNEXTLINE
93 if (e_exists.second) continue;
94

96
97 if (edge.cost < 0 && edge.reverse_cost < 0) continue;
98
99 E e;
100 // NOLINTNEXTLINE
101 boost::tie(e, added) = boost::add_edge(v1, v2, graph);

103 E_to_id.insert(std::make_pair(e, edge.id));
104 }
