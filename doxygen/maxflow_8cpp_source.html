72/* Inserting edges
73 * Push-relabel requires each edge to be mapped to its reverse with capacity 0.
74 */
75void PgrFlowGraph::insert_edges_push_relabel(
76 const std::vector<Edge_t> &edges) {
77 bool added;
78 for (const auto edge : edges) {
79 V v1 = get_boost_vertex(edge.source);
80 V v2 = get_boost_vertex(edge.target);
81 E e1, e1_rev, e2, e2_rev;
82 if (edge.cost > 0) {
83 boost::tie(e1, added) = boost::add_edge(v1, v2, graph);
84 boost::tie(e1_rev, added) =
85 boost::add_edge(v2, v1, graph);
86 E_to_id.insert(std::pair<E, int64_t>(e1, edge.id));
87 E_to_id.insert(std::pair<E, int64_t>(e1_rev, edge.id));
88 capacity[e1] = static_cast<int64_t>(edge.cost);
89 capacity[e1_rev] = 0;
90 rev[e1] = e1_rev;
91 rev[e1_rev] = e1;
92 }
93 if (edge.reverse_cost > 0) {
94 boost::tie(e2, added) = boost::add_edge(v2, v1, graph);
95 boost::tie(e2_rev, added) =
96 boost::add_edge(v1, v2, graph);
97 E_to_id.insert(std::pair<E, int64_t>(e2, edge.id));
98 E_to_id.insert(std::pair<E, int64_t>(e2_rev, edge.id));
99 capacity[e2] = static_cast<int64_t>(edge.reverse_cost);
100 capacity[e2_rev] = 0;
101 rev[e2] = e2_rev;
102 rev[e2_rev] = e2;
103 }
104 }
105}

107/* Inserting edges
108 * The other pgr_maxflow algorithms have no such requirement. (can have have as many edges)
109 */
110void PgrFlowGraph::insert_edges(
111 const std::vector<Edge_t> &edges) {
112 bool added;
113 for (const auto edge : edges) {
114 V v1 = get_boost_vertex(edge.source);
115 V v2 = get_boost_vertex(edge.target);
116 E e, e_rev;
117 boost::tie(e, added) = boost::add_edge(v1, v2, graph);
118 boost::tie(e_rev, added) =
119 boost::add_edge(v2, v1, graph);
120 E_to_id.insert(std::pair<E, int64_t>(e, edge.id));
121 E_to_id.insert(std::pair<E, int64_t>(e_rev, edge.id));
122 capacity[e] = edge.cost > 0 ? static_cast<int64_t>(edge.cost) : 0;
123 capacity[e_rev] = edge.reverse_cost > 0
124 ? static_cast<int64_t>(edge.reverse_cost) : 0;
125 rev[e] = e_rev;
126 rev[e_rev] = e;
127 }
128}

130/* Inserting edges
131 * for the edge_disjoint_paths algorithms
132 */
133void PgrFlowGraph::insert_edges_edge_disjoint(
134 const std::vector<Edge_t> &edges,
135 bool directed) {
136 bool added;
137 for (const auto edge : edges) {
138 V v1 = get_boost_vertex(edge.source);
139 V v2 = get_boost_vertex(edge.target);
140 E e, e_rev;
141 boost::tie(e, added) =
142 boost::add_edge(v1, v2, graph);
143 boost::tie(e_rev, added) =
144 boost::add_edge(v2, v1, graph);
145 E_to_id.insert(std::pair<E, int64_t>(e, edge.id));
146 E_to_id.insert(std::pair<E, int64_t>(e_rev,
147 edge.id));
148 if (directed) {
149 capacity[e] = edge.cost >= 0 ? 1 : 0;
150 capacity[e_rev] = edge.reverse_cost >= 0 ? 1 : 0;
151 } else {
152 if (edge.cost >= 0 || edge.reverse_cost >= 0) {
153 capacity[e] = 1;
154 capacity[e_rev] = 1;
155 }
156 }
157 rev[e] = e_rev;
158 rev[e_rev] = e;
159 }

162void PgrFlowGraph::set_supersource(
163 const std::set<int64_t> &source_vertices) {
164 bool added;
165 supersource = add_vertex(graph);
166 for (int64_t source_id : source_vertices) {
167 V source = get_boost_vertex(source_id);
168 E e, e_rev;
169 boost::tie(e, added) =
170 boost::add_edge(supersource, source, graph);
171 boost::tie(e_rev, added) =
172 boost::add_edge(source, supersource, graph);
173
174 capacity[e] = (std::numeric_limits<int32_t>::max)();
175 /* From sources to supersource has 0 capacity*/
176 capacity[e_rev] = 0;
177 rev[e] = e_rev;
178 rev[e_rev] = e;
179 }
180}

182void PgrFlowGraph::set_supersink(
183 const std::set<int64_t> &sink_vertices) {
184 bool added;
185 supersink = add_vertex(graph);
186 for (int64_t sink_id : sink_vertices) {
187 V sink = get_boost_vertex(sink_id);
188 E e, e_rev;
189 boost::tie(e, added) = boost::add_edge(sink, supersink, graph);
190 boost::tie(e_rev, added) =
191 boost::add_edge(supersink, sink, graph);
192 /*
193 * NOTE: int64_t crashes the server
194 */
195 /* From sinks to supersink has maximum capacity*/
196 capacity[e] = (std::numeric_limits<int32_t>::max)();
197 /* From supersink to sinks has 0 capacity*/
198 capacity[e_rev] = 0;
199 rev[e] = e_rev;
200 rev[e_rev] = e;
201 }

204std::vector<Flow_t>
205PgrFlowGraph::get_flow_edges() const {
206 std::vector<Flow_t> flow_edges;
207 E_it e, e_end;
208 for (boost::tie(e, e_end) = boost::edges(graph); e != e_end;
209 ++e) {
210 // A supersource/supersink is used internally
211 if (((capacity[*e] - residual_capacity[*e]) > 0) &&
212 ((*e).m_source != supersource) &&
213 ((*e).m_target != supersink)) {
214 Flow_t edge = {};
215 edge.edge = get_edge_id(*e);
216 edge.source = get_vertex_id((*e).m_source);
217 edge.target = get_vertex_id((*e).m_target);
218 edge.flow = capacity[*e] - residual_capacity[*e];
219 edge.residual_capacity = residual_capacity[*e];
220 flow_edges.push_back(edge);
221 }
222 }
223 return flow_edges;
224}

228void
229PgrFlowGraph::flow_dfs(V vertex,
230 size_t path_id,
231 std::vector<std::vector<int64_t> > &paths) {
232 Eout_it ei, e_end;
233 if (boost::edge(vertex, supersink, graph).second) {
234 int64_t v_id = get_vertex_id(vertex);
235 paths[path_id].push_back(v_id);
236 } else {
237 for (boost::tie(ei, e_end) =
238 boost::out_edges(vertex, graph);
239 ei != e_end; ++ei) {
240 if (residual_capacity[*ei] < capacity[*ei]) {
241 // exclude this edge from subsequent visits
242 capacity[*ei] = -1;
243 int64_t v_id = get_vertex_id(vertex);
244 paths[path_id].push_back(v_id);
245 flow_dfs((*ei).m_target,
246 path_id,
247 paths);
248 break;
249 }
250 }
251 }

254std::vector<Path_rt>
255PgrFlowGraph::get_edge_disjoint_paths(
256 size_t flow) {
257 std::vector<Path_rt> path_elements;
258
259 std::vector<std::vector<int64_t> > paths(flow, std::vector<int64_t>());
260 size_t path_id = 0;
261 Eout_it ei, e_end, ei2, e2_end;
262 for (boost::tie(ei, e_end) =
263 boost::out_edges(supersource, graph);
264 ei != e_end; ++ei) {
265 if (capacity[*ei] - residual_capacity[*ei] > 0) {
266 for (boost::tie(ei2, e2_end) =
267 boost::out_edges((*ei).m_target, graph);
268 ei2 != e2_end; ++ei2) {
269 if (capacity[*ei2] - residual_capacity[*ei2]
270 > 0) {
271 paths[path_id].push_back(get_vertex_id((*ei2).m_source));
272 flow_dfs((*ei2).m_target, path_id, paths);
273 path_id++;
274 }
275 }
276 }
277 }
278 for (size_t i = 0; i < flow; i++) {
279 size_t size = paths[i].size();
280 E e;
281 bool exists;
282 size_t j;
283 for (j = 0; j < size - 1; j++) {
284 Path_rt edge = {};
285 edge.start_id = paths[i][0];
286 edge.end_id = paths[i][size - 1];
287 edge.node = paths[i][j];
288 boost::tie(e, exists) = boost::edge(get_boost_vertex(paths[i][j]),
289 get_boost_vertex(paths[i][j + 1]),
290 graph);
291 edge.edge = get_edge_id(e);
292 path_elements.push_back(edge);
293 }
294 Path_rt edge = {};
295 edge.start_id = paths[i][0];
296 edge.end_id = paths[i][size - 1];
297 edge.node = paths[i][j];
298 edge.edge = -1;
299 path_elements.push_back(edge);
300 }
301 return path_elements;
302}
