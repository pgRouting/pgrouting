71template < class G >
72void compute_shortcuts(
73 G &graph,
74 typename G::V u,
75 typename G::V v,
76 Identifiers<typename G::V> out_vertices,
77 std::vector<typename G::E> &shortcuts,
78 std::ostringstream &log,
79 std::ostringstream &err
80) {
81 std::vector<typename G::V> predecessors(graph.num_vertices());
82 std::vector<double> distances(graph.num_vertices());
83 typename G::V_i out_i, out_end;
84
85 int64_t p_max = graph.compute_pmax(u, v, out_vertices);
86 log << " p_max = " << p_max << std::endl;
87
88 if ( p_max > 0 ) {
89 // Launch of a shortest paths query from u to all nodes
90 // with distance less than p_max
91 std::set<int64_t> reached_vertices_ids;
92 try {
93 boost::dijkstra_shortest_paths(
94 graph.graph,
95 u,
96 boost::predecessor_map(&predecessors[0])
97 .weight_map(get(&G::G_T_E::cost, graph.graph))
98 .distance_map(&distances[0])
99 .distance_inf(std::numeric_limits<double>::infinity())
100 .visitor(pgrouting::visitors::dijkstra_max_distance_visitor<typename G::V>(
101 p_max, distances, reached_vertices_ids)));
102 }
103 catch ( pgrouting::found_goals & ) {
104 log << " PgRouting exception during labelling!"
105 << std::endl;
106 log << " >>> Labelling interrupted"
107 << " because max distance is reached. " << std::endl;
108 log << " >>> Number of labelled vertices: "
109 << reached_vertices_ids.size() << std::endl;
110 Identifiers<int64_t> r;
111 r.set_ids(reached_vertices_ids);
112 log << " >>> Reached vertices: " << r << std::endl;
113 }
114 catch ( boost::exception const &except ) {
115 err << std::endl << "Boost exception "
116 << "during vertex contraction! "
117 << dynamic_cast<std::exception const &>(except).what()
118 << std::endl;
119 }
120 catch ( ... ) {
121 err << " Unknown exception during labelling!" << std::endl;
122 }
123 /* abort in case of an interruption occurs
124 (e.g. the query is being cancelled) */
125 CHECK_FOR_INTERRUPTS();
126
127 // Create a shortcut, for each w,
128 // when c(u, v) + c(u, w) = cost(shortest path from u to v)
129 for (const auto &w : out_vertices) {
130 if ( u != w ) {
131 typename G::E e, f, g;
132 bool found_e, found_f, found_g;
133 boost::tie(e, found_e) = boost::edge(u, v, graph.graph);
134 boost::tie(f, found_f) = boost::edge(v, w, graph.graph);
135 boost::tie(g, found_g) = boost::edge(u, w, graph.graph);
136 if (found_e && found_f &&
137 (!found_g || (distances[w] < graph[g].cost))) {
138 if ((predecessors[w] == v) && (predecessors[v] == u) &&
139 (distances[w] == graph[e].cost + graph[f].cost)) {
140 graph.process_ch_shortcut(
141 u, v, w,
142 shortcuts, log);
143 }
144 }
145 }
146 }
147 }
148}

161template <class G>
162int64_t vertex_contraction(
163 G &graph,
164 bool directed,
165 typename G::V v,
166 bool simulation,
167 std::vector<CH_edge> &shortcuts,
168 std::ostringstream &log,
169 std::ostringstream &err
170) {
171 Identifiers<typename G::V> adjacent_in_vertices;
172 Identifiers<typename G::V> adjacent_out_vertices;
173 int64_t n_old_edges;
174 std::vector<typename G::E> shortcut_edges;
175
176 if (directed) {
177 adjacent_in_vertices = graph.find_adjacent_in_vertices(v);
178 adjacent_out_vertices = graph.find_adjacent_out_vertices(v);
179 n_old_edges =
180 static_cast<int64_t>(adjacent_in_vertices.size()
181 + adjacent_out_vertices.size());
182 } else {
183 adjacent_in_vertices = graph.find_adjacent_vertices(v);
184 adjacent_out_vertices = adjacent_in_vertices;
185 n_old_edges = static_cast<int64_t>(adjacent_in_vertices.size());
186 }
187
188 log << ">> Contraction of node " << graph[v].id << std::endl
189 << num_vertices(graph.graph) << " vertices and "
190 << num_edges(graph.graph) << " edges " << std::endl;
191
192 for (auto &u : adjacent_in_vertices) {
193 log << " >> from " << graph[u].id << std::endl;
194 compute_shortcuts(
195 graph,
196 u,
197 v,
198 adjacent_out_vertices,
199 shortcut_edges,
200 log,
201 err);
202 }
203 if (!simulation) {
204 for (auto &w : adjacent_out_vertices) {
205 boost::remove_edge(v, w, graph.graph);
206 }
207 for (auto &u : graph.find_adjacent_in_vertices(v)) {
208 boost::remove_edge(u, v, graph.graph);
209 }
210 graph[v].clear_contracted_vertices();
211 for (auto &e : shortcut_edges) {
212 shortcuts.push_back(graph.graph[e]);
213 }
214 } else {
215 for (auto &e : shortcut_edges) {
216 boost::remove_edge(e, graph.graph);
217 }
218 }
219
220 log << " Size of the graph after contraction: "
221 << num_vertices(graph.graph)
222 << " vertices and " << num_edges(graph.graph)
223 << " edges" << std::endl
224 << " " << shortcut_edges.size()
225 << " shortcuts created, " << n_old_edges
226 << " old edges" << std::endl;
227
228 int64_t m;
229 m = static_cast<int64_t>(shortcut_edges.size())
230 - static_cast<int64_t>(n_old_edges);
231 log << " Metric: edge difference = " << shortcut_edges.size()
232 << " - " << n_old_edges << " = " << m << std::endl;
233 return m;
234}

236template < class G >
237void contractionHierarchies(
238 G &graph,
239 bool directed,
240 std::ostringstream &log,
241 std::ostringstream &err
242) {
243 // First iteration over vertices
244 // graph_copy must stay unchanged during this first step,
245 // graph will be modified
246 G graph_copy = graph;
247 std::vector<CH_edge> shortcuts;
248
249 // Fill the priority queue with vertices ordered by their IDs
250 log << "Do contraction" << std::endl;
251 typename G::PQ minPQ, priority_queue;
252 for (const auto &v:
253 boost::make_iterator_range(boost::vertices(graph.graph))) {
254 if (!graph.is_forbidden(v)) {
255 minPQ.push(std::make_pair(graph[v].id, v));
256 }
257 }
258
259 while (!minPQ.empty()) {
260 std::pair< int64_t, typename G::V > ordered_vertex = minPQ.top();
261 minPQ.pop();
262 int64_t corrected_metric =
263 detail::vertex_contraction(
264 graph_copy,
265 directed,
266 ordered_vertex.second,
267 true,
268 shortcuts,
269 log,
270 err);
271 log << " Vertex: " << graph[ordered_vertex.second].id
272 << ", min value of the queue: "
273 << minPQ.top().first << std::endl
274 << " Lazy non-destructive simulation: initial order "
275 << ordered_vertex.first << ", new order "
276 << corrected_metric << std::endl;
277
278 if (minPQ.top().first < corrected_metric) {
279 log << " Vertex reinserted in the queue" << std::endl;
280 minPQ.push(
281 std::make_pair(corrected_metric, ordered_vertex.second));
282 } else {
283 std::pair< int64_t, typename G::V > contracted_vertex;
284 typename G::V u =
285 graph.vertices_map[graph[ordered_vertex.second].id];
286 contracted_vertex.first = detail::vertex_contraction(
287 graph_copy,
288 directed,
289 u,
290 false,
291 shortcuts,
292 log,
293 err);
294 log << " Vertex endly contracted in the queue" << std::endl;
295 contracted_vertex.second = ordered_vertex.second;
296 priority_queue.push(contracted_vertex);
297 }
298 }
299 log << std::endl << "Copy shortcuts" << std::endl;
300 graph.copy_shortcuts(shortcuts, log);
301 log << std::endl << "Priority queue: " << std::endl;
302 graph.set_vertices_metric_and_hierarchy(priority_queue, log);
303}

317template <class G>
318void contractionHierarchies(
319 G& graph,
320 bool directed,
321 const std::vector<int64_t> &forbidden_vertices,
322 std::ostringstream &log,
323 std::ostringstream &err) {
324 // Transform the forbidden vertices IDs vector to a collection of vertices
325 pgrouting::Identifiers<typename G::V> forbid_vertices;
326 for (const auto &vertex : forbidden_vertices) {
327 if (graph.has_vertex(vertex)) {
328 forbid_vertices += graph.get_V(vertex);
329 }
330 }
331 graph.set_forbidden_vertices(forbid_vertices);
332
333 // Execute the contraction
334 try {
335 detail::contractionHierarchies(graph, directed, log, err);
336 }
337 catch ( ... ) {
338 err << "Contractions hierarchy failed" << std::endl;
339 throw;
340 }
341}
