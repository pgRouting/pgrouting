166 template <typename G , typename V> Path(
167 G &graph,
168 int64_t source,
169 double distance,
170 const std::vector<V> &predecessors,
171 const std::vector<double> &distances) :
172 m_start_id(source),
173 m_end_id(source) {
174 for (V i = 0; i < distances.size(); ++i) {
175 if (distances[i] <= distance) {
176 auto cost = distances[i] - distances[predecessors[i]];
177 auto edge_id = graph.get_edge_id(predecessors[i], i, cost);
178 push_back(
179 {graph[i].id,
180 edge_id, cost,
181 distances[i], graph[predecessors[i]].id});
182 }
183 }
184 }

291 template <typename G , typename V> void complete_path(
292 const G &graph,
293 const V v_source,
294 const V v_target,
295 const std::vector<V> &predecessors,
296 const std::vector<double> &distances,
297 bool normal) {
298 // no path was found
299 if (v_target == predecessors[v_target]) {
300 return;
301 }
302
303 /*
304 * set the target
305 */
306 auto target = v_target;
307
308 /*
309 * the last stop is the target
310 */
311 /* TODO(v4) add correct predecessor node */
312 push_front(
313 {graph.graph[target].id, -1,
314 0, distances[target], 0});
315
316 /*
317 * get the path
318 */
319 while (target != v_source) {
320 /*
321 * done when the predecessor of the target is the target
322 */
323 if (target == predecessors[target]) break;
324
325 /*
326 * Inserting values in the path
327 */
328 auto cost = distances[target] - distances[predecessors[target]];
329 auto vertex_id = graph.graph[predecessors[target]].id;
330 auto edge_id = normal?
331 graph.get_edge_id(predecessors[target], target, cost)
332 : graph.get_edge_id(target, predecessors[target], cost);
333
334 /* TODO(v4) add correct predecessor node */
335 push_front({
336 vertex_id,
337 edge_id,
338 cost,
339 distances[target] - cost, 0});
340 target = predecessors[target];
341 }
342
343 return;
344 }
