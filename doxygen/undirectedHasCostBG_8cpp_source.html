45UndirectedHasCostBG::UndirectedHasCostBG(std::vector<IID_t_rt> &distances) {
46 /*
47 * Inserting vertices
48 */
49 Identifiers<int64_t> ids;
50 for (auto &d : distances) {
51 ids += d.from_vid;
52 ids += d.to_vid;
53 /*
54 * Its undirected graph:
55 * keeping from_vid < to_vid
56 */
57 if (d.to_vid > d.from_vid) {
58 std::swap(d.to_vid, d.from_vid);
59 }
60 }
61
62 for (const auto &id : ids) {
63 insert_vertex(id);
64 }
65
66 /*
67 * Inserting edges
68 */
69 for (const auto &edge : distances) {
70 /*
71 * skip loops
72 */
73 if (edge.from_vid == edge.to_vid) continue;
74
75 /*
76 * skip negative costs
77 */
78 if (edge.cost < 0) continue;
79
80 auto v1 = get_boost_vertex(edge.from_vid);
81 auto v2 = get_boost_vertex(edge.to_vid);
82 auto e_exists = boost::edge(v1, v2, m_graph);
83 if (e_exists.second) {
84 auto weight = get(boost::edge_weight_t(), m_graph, e_exists.first);
85 /*
86 * skip duplicated edges with less cost
87 */
88 if (weight < edge.cost) continue;
89 if (edge.cost < weight) {
90 /*
91 * substitute edge with new lesser cost found
92 */
93 boost::put(boost::edge_weight_t(), m_graph, e_exists.first, edge.cost);
94 }
95 continue;
96 }
97
98 auto add_result = boost::add_edge(v1, v2, edge.cost, m_graph);
99 if (!add_result.second) {
100 throw std::make_pair(
101 std::string("INTERNAL: something went wrong adding and edge\n"),
102 std::string(__PGR_PRETTY_FUNCTION__));
103 }
104 }
105}

118UndirectedHasCostBG::UndirectedHasCostBG(const std::vector<Coordinate_t> &coordinates) {
119 /*
120 * Inserting edges
121 */
122 for (size_t i = 0; i < coordinates.size(); ++i) {
123 insert_vertex(coordinates[i].id);
124 auto u = get_boost_vertex(coordinates[i].id);
125 auto ux = coordinates[i].x;
126 auto uy = coordinates[i].y;
127
128 /*
129 * undirected, so only need traverse higher coordinates for connections
130 */
131 for (size_t j = i + 1; j < coordinates.size(); ++j) {
132 insert_vertex(coordinates[j].id);
133 auto v = get_boost_vertex(coordinates[j].id);
134
135 /*
136 * ignoring duplicated coordinates
137 */
138 if (boost::edge(u, v, m_graph).second) continue;
139
140 auto vx = coordinates[j].x;
141 auto vy = coordinates[j].y;
142
143 auto const dx = ux - vx;
144 auto const dy = uy - vy;
145
146 /*
147 * weight is euclidean distance
148 */
149 auto add_result = boost::add_edge(u, v, sqrt(dx * dx + dy * dy), m_graph);
150 if (!add_result.second) {
151 throw std::make_pair(
152 std::string("INTERNAL: something went wrong adding and edge\n"),
153 std::string(__PGR_PRETTY_FUNCTION__));
154 }
155 }
156 }
157}

165void
166UndirectedHasCostBG::insert_vertex(int64_t id) {
167 try {
168 if (has_vertex(id)) return;
169 auto v = add_vertex(m_id_to_V.size(), m_graph);
170 m_id_to_V.insert(std::make_pair(id, v));
171 m_V_to_id.insert(std::make_pair(v, id));
172 } catch (...) {
173 throw std::make_pair(
174 std::string("INTERNAL: something went wrong when inserting a vertex"),
175 std::string(__PGR_PRETTY_FUNCTION__));
176 }
177}
