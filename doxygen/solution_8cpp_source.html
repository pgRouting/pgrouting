130Vehicle::Cost
131Solution::cost() const {
132 double total_duration(0);
133 double total_wait_time(0);
134 int total_twv(0);
135 int total_cv(0);
136 for (const auto &v : fleet) {
137 total_duration += v.duration();
138 total_wait_time += v.total_wait_time();
139 total_twv += v.twvTot();
140 total_cv += v.cvTot();
141 }
142 return std::make_tuple(
143 total_twv, total_cv, fleet.size(),
144 total_wait_time, total_duration);
145}

189bool
190Solution::operator<(const Solution &s_rhs) const {
191 Vehicle::Cost lhs(cost());
192 Vehicle::Cost rhs(s_rhs.cost());
193
194 /*
195 * capacity violations
196 */
197 if (std::get<0>(lhs) < std::get<0>(rhs)) return true;
198 if (std::get<0>(lhs) > std::get<0>(rhs)) return false;
199
200 /*
201 * time window violations
202 */
203 if (std::get<1>(lhs) < std::get<1>(rhs)) return true;
204 if (std::get<1>(lhs) > std::get<1>(rhs)) return false;
205
206 /*
207 * fleet size
208 */
209 if (std::get<2>(lhs) < std::get<2>(rhs)) return true;
210 if (std::get<2>(lhs) > std::get<2>(rhs)) return false;
211
212 /*
213 * waiting time
214 */
215 if (std::get<3>(lhs) < std::get<3>(rhs)) return true;
216 if (std::get<3>(lhs) > std::get<3>(rhs)) return false;
217
218 /*
219 * duration
220 */
221 if (std::get<4>(lhs) < std::get<4>(rhs)) return true;
222 if (std::get<4>(lhs) > std::get<4>(rhs)) return false;
223
224 return false;
225}
