519 int64_t get_edge_id(V from, V to, double &weight) const {
520 EO_i out_i, out_end;
521 double minCost = (std::numeric_limits<double>::max)();
522 int64_t minEdge = -1;
523 for (boost::tie(out_i, out_end) = boost::out_edges(from, graph); out_i != out_end; ++out_i) {
524 auto e = *out_i;
525 auto v_target = target(e);
526 auto v_source = source(e);
527 if ((from == v_source) && (to == v_target) && (weight == graph[e].cost)) return graph[e].id;
528 if ((from == v_source) && (to == v_target) && (minCost > graph[e].cost)) {
529 minCost = graph[e].cost;
530 minEdge = graph[e].id;
531 }
532 }
533 weight = minEdge == -1? 0: minCost;
534 return minEdge;
535 }

567 E get_edge(V from, V to, double &distance) const {
568 E e;
569 EO_i out_i, out_end;
570 V v_source, v_target;
571 double minCost = (std::numeric_limits<double>::max)();
572 E minEdge;
573 bool valid = false;
574 for (boost::tie(out_i, out_end) = boost::out_edges(from, graph);
575 out_i != out_end; ++out_i) {
576 e = *out_i;
577 if (!valid) {
578 minEdge = e;
579 valid = true;
580 }
581 v_target = target(e);
582 v_source = source(e);
583 if ((from == v_source) && (to == v_target)
584 && (distance == graph[e].cost)) {
585 return e;
586 }
587 if ((from == v_source) && (to == v_target)
588 && (minCost > graph[e].cost)) {
589 minCost = graph[e].cost;
590 minEdge = e;
591 }
592 }
593 return minEdge;
594 }

614 void disconnect_edge(int64_t p_from, int64_t p_to) {
615 /* nothing to do, a vertex doesn't exist */
616 if (!has_vertex(p_from) || !has_vertex(p_to)) return;
617
618 EO_i out, out_end;
619 auto u(get_V(p_from));
620 auto v(get_V(p_to));
621
622 for (boost::tie(out, out_end) = out_edges(u, graph); out != out_end; ++out) {
623 if (target(*out) == v) {
624 removed_edges.push_back(get_edge_info(*out));
625 break;
626 }
627 }
628 boost::remove_edge(u, v, graph);
629 }

639 void disconnect_out_going_edge(int64_t vid, int64_t eid) {
640 /* nothing to do, the vertex doesn't exist */
641 if (!has_vertex(vid)) return;
642
643 auto v(get_V(vid));
644 EO_i out, out_end;
645 bool change = true;
646 while (change) {
647 change = false;
648 for (boost::tie(out, out_end) = out_edges(v, graph); out != out_end; ++out) {
649 if (graph[*out].id == eid) {
650 removed_edges.push_back(get_edge_info(*out));
651 boost::remove_edge((*out), graph);
652 change = true;
653 break;
654 }
655 }
656 }

678 void disconnect_vertex(V v) {
679 EO_i out, out_end;
680 for (boost::tie(out, out_end) = out_edges(v, graph); out != out_end; ++out) {
681 removed_edges.push_back(get_edge_info(*out));
682 }
683
684 /* special case */
685 if (m_is_directed) {
686 EI_i in, in_end;
687 for (boost::tie(in, in_end) = in_edges(v, graph); in != in_end; ++in) {
688 removed_edges.push_back(get_edge_info(*in));
689 }
690 }
691
692 /* delete incoming and outgoing edges from the vertex */
693 boost::clear_vertex(v, graph);
694 }

732 template <typename T> void graph_add_edge(const T &edge, bool normal) {
733 bool inserted;
734 E e;
735 if ((edge.cost < 0) && (edge.reverse_cost < 0)) return;
736
737 /*
738 * true: for source
739 * false: for target
740 */
741 auto vm_s = get_V(T_V(edge, true));
742 auto vm_t = get_V(T_V(edge, false));
743
744 pgassert(vertices_map.find(edge.source) != vertices_map.end());
745 pgassert(vertices_map.find(edge.target) != vertices_map.end());
746 if (edge.cost >= 0) {
747 boost::tie(e, inserted) =
748 boost::add_edge(vm_s, vm_t, graph);
749 graph[e].cost = edge.cost;
750 graph[e].id = edge.id;
751 }
752
753 if (edge.reverse_cost >= 0
754 && (m_is_directed || (!m_is_directed && edge.cost != edge.reverse_cost))) {
755 boost::tie(e, inserted) =
756 boost::add_edge(vm_t, vm_s, graph);
757
758 graph[e].cost = edge.reverse_cost;
759 graph[e].id = normal? edge.id : -edge.id;
760 }
761 }

763 template <typename T> void graph_add_min_edge_no_parallel(const T &edge) {
764 bool inserted;
765 E e;
766 if ((edge.cost < 0) && (edge.reverse_cost < 0)) return;
767
768 /*
769 * true: for source
770 * false: for target
771 */
772 auto vm_s = get_V(T_V(edge, true));
773 auto vm_t = get_V(T_V(edge, false));
774
775 pgassert(vertices_map.find(edge.source) != vertices_map.end());
776 pgassert(vertices_map.find(edge.target) != vertices_map.end());
777 if (edge.cost >= 0) {
778 E e1;
779 bool found;
780 boost::tie(e1, found) = boost::edge(vm_s, vm_t, graph);
781 if (found) {
782 if (edge.cost < graph[e1].cost) {
783 graph[e1].cost = edge.cost;
784 graph[e1].id = edge.id;
785 }
786 } else {
787 boost::tie(e, inserted) = boost::add_edge(vm_s, vm_t, graph);
788 graph[e].cost = edge.cost;
789 graph[e].id = edge.id;
790 }
791 }
792
793 if (edge.reverse_cost >= 0
794 && (m_is_directed || (!m_is_directed && edge.cost != edge.reverse_cost))) {
795 E e1;
796 bool found;
797 boost::tie(e1, found) = boost::edge(vm_t, vm_s, graph);
798 if (found) {
799 if (edge.reverse_cost < graph[e1].cost) {

818 template <typename T> void graph_add_neg_edge(const T &edge, bool normal = true) {
819 bool inserted;
820 E e;
821
822 auto vm_s = get_V(T_V(edge, true));
823 auto vm_t = get_V(T_V(edge, false));
824
825 pgassert(vertices_map.find(edge.source) != vertices_map.end());
826 pgassert(vertices_map.find(edge.target) != vertices_map.end());
827
828 boost::tie(e, inserted) = boost::add_edge(vm_s, vm_t, graph);
829 if (edge.cost < 0) {
830 /* reading negative edges as positive */
831 graph[e].cost = (-0.5)*edge.cost;
832 } else {
833 graph[e].cost = edge.cost;
834 }
835 graph[e].id = edge.id;
836
837 if (m_is_directed
838 || (!m_is_directed && edge.cost > edge.reverse_cost)) {
839 boost::tie(e, inserted) =
840 boost::add_edge(vm_t, vm_s, graph);
841 if (edge.reverse_cost < 0) {
842 /* reading negative edges as positive */
843 graph[e].cost = (-0.5)*edge.reverse_cost;
844 } else {
845 graph[e].cost = edge.reverse_cost;
846 }
847
848 graph[e].id = normal? edge.id : -edge.id;
849 }
