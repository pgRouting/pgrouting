56 std::deque<Path> binaryBreadthFirstSearch(
57 G &graph,
58 const std::vector<int64_t> &start_vertex,
59 const std::vector<int64_t> &end_vertex) {
60 std::deque<Path> paths;
61
62 for (const auto &source : start_vertex) {
63 auto result_paths = one_to_many_binaryBreadthFirstSearch(
64 graph,
65 source,
66 end_vertex);
67 std::sort(result_paths.begin(), result_paths.end(),
68 [](const Path &e1, const Path &e2) -> bool {
69 return e1.end_id() < e2.end_id();
70 });
71 paths.insert(
72 paths.begin(),
73 std::make_move_iterator(result_paths.begin()),
74 std::make_move_iterator(result_paths.end()));
75 }
76
77 return paths;
78 }

81 std::deque<Path> binaryBreadthFirstSearch(
82 G &graph,
83 const std::map<int64_t, std::set<int64_t>> &combinations) {
84 std::deque<Path> paths;
85
86 for (const auto &c : combinations) {
87 if (!graph.has_vertex(c.first)) continue;
88
89 std::deque<Path> result_paths = one_to_many_binaryBreadthFirstSearch(
90 graph, c.first, c.second);
91 paths.insert(
92 paths.begin(),
93 std::make_move_iterator(result_paths.begin()),
94 std::make_move_iterator(result_paths.end()));
95 }
96
97 std::sort(paths.begin(), paths.end(),
98 [](const Path &e1, const Path &e2) -> bool {
99 return e1.end_id() < e2.end_id();
100 });
101 std::stable_sort(paths.begin(), paths.end(),
102 [](const Path &e1, const Path &e2) -> bool {
103 return e1.start_id() < e2.start_id();
104 });
105
106 return paths;
107 }

112 std::deque<Path> one_to_many_binaryBreadthFirstSearch(
113 const G &graph,
114 int64_t start_vertex,
115 const std::set<int64_t> &end_vertex) {
116 std::deque<Path> paths;
117
118 if (!graph.has_vertex(start_vertex)) return paths;
119
120 std::vector<double> current_cost(graph.num_vertices(), std::numeric_limits<double>::infinity());
121 std::vector<E> from_edge(graph.num_vertices());
122 std::deque<V> dq;
123 default_edge = from_edge[0];
124
125 auto bgl_start_vertex = graph.get_V(start_vertex);
126
127 current_cost[bgl_start_vertex] = 0;
128 dq.push_front(bgl_start_vertex);
129
130 while (dq.empty() == false) {
131 auto head_vertex = dq.front();
132
133 dq.pop_front();
134
135 updateVertexCosts(graph, current_cost, from_edge, dq, head_vertex);
136 }
137
138 for (auto target_vertex : end_vertex) {
139 if (!graph.has_vertex(target_vertex)) continue;
140
141 auto bgl_target_vertex = graph.get_V(target_vertex);
142
143 if (from_edge[bgl_target_vertex] == default_edge) continue;
144
145 paths.push_front(
146 getPath(graph, bgl_start_vertex, target_vertex, bgl_target_vertex, from_edge, current_cost));
147 }
148
149 return paths;
150 }

152 Path getPath(
153 const G &graph,
154 V bgl_start_vertex,
155 int64_t target,
156 V bgl_target_vertex,
157 std::vector<E> &from_edge,
158 std::vector<double> &current_cost) {
159 auto current_node = bgl_target_vertex;
160
161 Path path = Path(graph[bgl_start_vertex].id, graph[current_node].id);
162
163 /* TODO(v4) add correct predecessor node */
164 path.push_back({target, -1, 0, current_cost[current_node], 0});
165
166 do {
167 E e = from_edge[current_node];
168 auto from = graph.source(e);
169
170 /* TODO(v4) add correct predecessor node */
171 path.push_back({graph[from].id, graph[e].id, graph[e].cost, current_cost[from], 0});
172
173 current_node = from;
174 } while (from_edge[current_node] != default_edge);
175
176 std::reverse(path.begin(), path.end());
177 return path;
178 }

181 void updateVertexCosts(
182 const G &graph,
183 std::vector<double> &current_cost,
184 std::vector<E> &from_edge,
185 std::deque<V> &dq,
186 V &head_vertex) {
187 auto out_edges = boost::out_edges(head_vertex, graph.graph);
188 E e;
189 EO_i out_i;
190 EO_i out_end;
191 V v_source, v_target;
192
193 for (boost::tie(out_i, out_end) = out_edges;
194 out_i != out_end; ++out_i) {
195 e = *out_i;
196 v_target = graph.target(e);
197 v_source = graph.source(e);
198 double edge_cost = graph[e].cost;
199
200 if (std::isinf(current_cost[v_target]) || current_cost[v_source] + edge_cost < current_cost[v_target]) {
201 current_cost[v_target] = current_cost[v_source] + edge_cost;
202
203 from_edge[v_target] = e;
204
205 if (edge_cost != 0) {
206 dq.push_back(v_target);
207 } else {
208 dq.push_front(v_target);
209 }
210 }
211 }
212 }
