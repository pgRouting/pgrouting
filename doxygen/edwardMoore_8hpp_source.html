53 std::deque<Path> edwardMoore(
54 G &graph,
55 const std::map<int64_t, std::set<int64_t>> &combinations) {
56 std::deque<Path> paths;
57
58 for (const auto &comb : combinations) {
59 if (!graph.has_vertex(comb.first)) continue;
60 auto tmp_paths = one_to_many_edwardMoore(graph, comb.first, comb.second);
61 paths.insert(paths.end(), tmp_paths.begin(), tmp_paths.end());
62 }
63 std::sort(paths.begin(), paths.end(),
64 [](const Path &e1, const Path &e2) -> bool {
65 return e1.end_id() < e2.end_id();
66 });
67 std::stable_sort(paths.begin(), paths.end(),
68 [](const Path &e1, const Path &e2) -> bool {
69 return e1.start_id() < e2.start_id();
70 });
71 return paths;
72 }

77 std::deque<Path> one_to_many_edwardMoore(
78 G &graph,
79 int64_t start_vertex,
80 const std::set<int64_t> &end_vertex) {
81 std::deque<Path> paths;
82
83 if (graph.has_vertex(start_vertex) == false) {
84 return paths;
85 }
86
87 std::vector<double> current_cost(graph.num_vertices(), std::numeric_limits<double>::infinity());
88 std::vector<bool> isInQ(graph.num_vertices(), false);
89 std::vector<E> from_edge(graph.num_vertices());
90 std::deque<V> dq;
91 default_edge = from_edge[0];
92
93 auto bgl_start_vertex = graph.get_V(start_vertex);
94
95 current_cost[bgl_start_vertex] = 0;
96 isInQ[bgl_start_vertex] = true;
97 dq.push_front(bgl_start_vertex);
98
99 while (dq.empty() == false) {
100 auto head_vertex = dq.front();
101
102 dq.pop_front();
103 isInQ[head_vertex] = false;
104
105 updateVertexCosts(graph, current_cost, isInQ, from_edge, dq, head_vertex);
106 }
107
108 for (auto target_vertex : end_vertex) {
109 if (graph.has_vertex(target_vertex) == false) {
110 continue;
111 }
112
113 auto bgl_target_vertex = graph.get_V(target_vertex);
114
115 if (from_edge[bgl_target_vertex] == default_edge) {
116 continue;
117 }
118
119 paths.push_front(
120 getPath(graph, bgl_start_vertex, target_vertex, bgl_target_vertex, from_edge, current_cost));
121 }
122
123 return paths;
124 }

126 Path getPath(
127 G &graph,
128 V bgl_start_vertex,
129 int64_t target,
130 V bgl_target_vertex,
131 std::vector<E> &from_edge,
132 std::vector<double> &current_cost) {
133 auto current_node = bgl_target_vertex;
134
135 Path path = Path(graph[bgl_start_vertex].id, graph[current_node].id);
136
137 /* TODO(v4) add correct predecessor node */
138 path.push_back({target, -1, 0, current_cost[current_node], 0});
139
140 do {
141 E e = from_edge[current_node];
142 auto from = graph.source(e);
143
144 /* TODO(v4) add correct predecessor node */
145 path.push_back({graph[from].id, graph[e].id, graph[e].cost, current_cost[from], 0});
146
147 current_node = from;
148 } while (from_edge[current_node] != default_edge);
149
150 std::reverse(path.begin(), path.end());
151 return path;
152 }

154 void updateVertexCosts(
155 G &graph,
156 std::vector<double> &current_cost,
157 std::vector<bool> &isInQ,
158 std::vector<E> &from_edge,
159 std::deque<V> &dq,
160 V &head_vertex) {
161 /* abort in case of an interruption occurs (e.g. the query is being cancelled) */
162 CHECK_FOR_INTERRUPTS();
163
164 auto out_edges = boost::out_edges(head_vertex, graph.graph);
165 E e;
166 EO_i out_i;
167 EO_i out_end;
168 V v_source, v_target;
169
170 for (boost::tie(out_i, out_end) = out_edges;
171 out_i != out_end; ++out_i) {
172 e = *out_i;
173 v_target = graph.target(e);
174 v_source = graph.source(e);
175 double edge_cost = graph[e].cost;
176
177 if (std::isinf(current_cost[v_target]) || current_cost[v_source] + edge_cost < current_cost[v_target]) {
178 current_cost[v_target] = current_cost[v_source] + edge_cost;
179
180 from_edge[v_target] = e;
181
182 if (isInQ[v_target] == false) {
183 dq.push_back(v_target);
184 isInQ[v_target] = true;
185 }
186 }
187 }
188 }
