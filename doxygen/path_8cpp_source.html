72void Path::reverse() {
73 std::swap(m_start_id, m_end_id);
74 if (path.size() <= 1) return;
75 std::deque< Path_t > newpath;
76 for (size_t i = 0; i < path.size(); ++i) {
77 /* TODO(v4) add correct predecessor node */
78 newpath.push_front({
79 path[i].node,
80 (i == 0? -1 : path[i - 1].edge),
81 (i == 0? 0 : path[i - 1].cost),
82 0, 0
83 });
84 }
85 for (size_t i = 0; i < newpath.size(); ++i) {
86 newpath[i].agg_cost = (i == 0)?
87 0 :
88 newpath[i - 1].agg_cost + newpath[i - 1].cost;
89 }
90 path = newpath;
91}

167bool Path::isEqual(const Path &subpath) const {
168 if (subpath.empty()) return true;
169 if (subpath.size() >= path.size()) return false;
170 std::deque< Path_t >::const_iterator i, j;
171 for (i = path.begin(), j = subpath.begin(); j != subpath.end(); ++i, ++j) {
172 if ((*i).node != (*j).node) return false;
173 }
174 return true;

205void Path::append(const Path &other) {
206 pgassert(m_end_id == other.m_start_id);
207 if (other.m_start_id == other.m_end_id) {
208 pgassert(other.path.empty());
209 return;
210 }
211 if (m_start_id == m_end_id) {
212 pgassert(path.empty());
213 *this = other;
214 return;
215 }
216 pgassert(path.back().edge == -1);
217 m_end_id = other.m_end_id;
218
219 auto last = path.back();
220 auto agg_cost = last.agg_cost;
221
222 path.pop_back();
223
224 for (auto item : other.path) {
225 item.agg_cost += agg_cost;
226 push_back(item);
227 }
228}

231void Path::generate_postgres_data(
232 Path_rt **postgres_data,
233 size_t &sequence) const {
234 for (const auto e : path) {
235 auto agg_cost = std::fabs(
236 e.agg_cost - (std::numeric_limits<double>::max)()) < 1?
237 std::numeric_limits<double>::infinity() : e.agg_cost;
238 auto cost = std::fabs(e.cost - (std::numeric_limits<double>::max)()) < 1?
239 std::numeric_limits<double>::infinity() : e.cost;
240
241 (*postgres_data)[sequence] = {start_id(), end_id(), e.node, e.edge, cost, agg_cost};
242 ++sequence;
243 }

339/*
340 * sort the paths by size from greater to smaller
341 * and sort each path by node
342 * all the nodes on p2 are going to be compared
343 * with the nodes of p1
344 *
345 * When both paths reach the node and p1.agg_cost > p2.agg_cost
346 * erase the node of p1
347 * (can't erase from p2 because we loose the iterators
348 * so in a future cycle it will be deleted)
349 *
350 * sort the paths by start_id,
351 */
352
353void
354equi_cost(std::deque< Path > &paths) {
355 /* sort paths by size: largest first */
356 std::sort(paths.begin(), paths.end(),
357 [](const Path &e1, const Path &e2)->bool {
358 return e2.size() < e1.size();
359 });
360
361 /* sort each path by node: smaller id first */
362 for (auto &p : paths) {
363 if (p.size() < 2) continue;
364 std::sort(p.begin(), p.end(),
365 [](const Path_t &e1, const Path_t &e2)->bool {
366 return e1.node < e2.node;
367 });
368 }
369
370 for (auto &p1 : paths) {
371 for (const auto &p2 : paths) {
372 if (p1.start_id() == p2.start_id()) continue;
373 for (const auto &stop : p2.path) {
374 /* find the node of p2 in p1 */
375 auto pos = lower_bound(p1.begin(), p1.end(), stop,
376 [](const Path_t &l, const Path_t &r)->bool {
377 return l.node < r.node;
378 });
379
380 if (pos != p1.end()
381 && (stop.node == pos->node)
382 && (stop.agg_cost < pos->agg_cost)) {
383 /* both share the same node &
384 * the second path has the smallest
385 * So erasing from the first path */
386 p1.erase(pos);
387 }
388 }
389 }
390 }
391 /* sort paths by start_id */
392 std::sort(paths.begin(), paths.end(),
393 [](const Path &e1, const Path &e2)->bool {
394 return e1.start_id() < e2.start_id();
395 });
396
397 /* sort each path by agg_cost, node */
398 for (auto &path : paths) {
399 path.sort_by_node_agg_cost();
400 }
401}
