208  bool is_shortcut_possible(
209  V v,
210  V u,
211  V w) {
212  pgassert(u != v);
213  pgassert(v != w);
214  pgassert(u != w);
215  if (this->is_undirected()) {
216  /*
217  * u - v - w
218  */
219  return has_u_v_w(u, v, w);
220  }
221 
222  pgassert(this->is_directed());
223  return
224  /*
225  * u <-> v <-> w
226  */
227  (has_u_v_w(u, v, w) && has_u_v_w(w, v, u))
228  /*
229  * u -> v -> w
230  */
231  ||
232  (has_u_v_w(u, v, w) && !(boost::edge(v, u, this->graph).second || boost::edge(w, v, this->graph).second))
233  /*
234  * u <- v <- w
235  */
236  ||
237  (has_u_v_w(w, v, u) && !(boost::edge(v, w, this->graph).second || boost::edge(u, v, this->graph).second));
