83 void
84 Pgr_pickDeliver::solve() {
85  auto initial_sols = solutions;
86 
87  if (m_initial_id == 0) {
88  msg.log << "trying all \n";
89  for (int i = 1; i < 7; ++i) {
90  initial_sols.push_back(Initial_solution((Initials_code)i, m_orders.size()));
91  msg.log << "solution " << i << "\n" << initial_sols.back().tau();
92  // TODO(vicky) calculate the time it takes
93  msg.log << "Initial solution " << i
94  << " duration: " << initial_sols.back().duration();
95  }
96  } else {
97  msg.log << "only trying " << m_initial_id << "\n";
98  initial_sols.push_back(Initial_solution((Initials_code)m_initial_id, m_orders.size()));
99  // TODO(vicky) calculate the time it takes
100  msg.log << "Initial solution " << m_initial_id
101  << " duration: " << initial_sols[0].duration();
102  }
103 
104 
105  /*
106  * Sorting solutions: the best is at the back
107  */
108  pgassert(!initial_sols.empty());
109  std::sort(initial_sols.begin(), initial_sols.end(), []
110  (const Solution &lhs, const Solution &rhs) -> bool {
111  return rhs < lhs;
112  });
113 
114 #if 1
115  solutions.push_back(Optimize(initial_sols.back(), m_max_cycles));
116 #else
117  solutions.push_back(initial_sols.back());
118 #endif
119  pgassert(!solutions.empty());
120 
121  msg.log << "best solution duration = " << solutions.back().duration();
122 }

126 std::vector< General_vehicle_orders_t >
127 Pgr_pickDeliver::get_postgres_result() const {
128  auto result = solutions.back().get_postgres_result();
129 
130  General_vehicle_orders_t aggregates = {
131  /*
132  * Vehicle id = -2 indicates its an aggregate row
133  *
134  * (twv, cv, fleet, wait, duration)
135  */
136  -2, // summary row on vehicle_number
137  solutions.back().twvTot(), // on vehicle_id
138  solutions.back().cvTot(), // on vehicle_seq
139  -1, // on order_id
140  -1, // on stop_id
141  -2, // on stop_type (gets increased later by one so it gets -1)
142  -1, // not accounting total loads
143  solutions.back().total_travel_time(),
144  -1, // not accounting arrival_travel_time
145  solutions.back().wait_time(),
146  solutions.back().total_service_time(),
147  solutions.back().duration(),
148  };
149  result.push_back(aggregates);
150 
151 
152 #ifndef NDEBUG
153  for (const auto sol : solutions) {
154  msg.log << sol.tau();
155  }
156 #endif
157  return result;
158 }
