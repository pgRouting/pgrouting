224  void apply_transformation(
225  const pgrouting::DirectedGraph& digraph) {
226  V_i vertexIt, vertexEnd;
227  EO_i e_outIt, e_outEnd;
228  EI_i e_inIt, e_inEnd;
229 
230  // For every vertex in the original graph, create a line graph
231  // using the edges connected to that vertex
232  for (boost::tie(vertexIt, vertexEnd) = boost::vertices(digraph.graph);
233  vertexIt != vertexEnd; vertexIt++) {
234  V vertex = *vertexIt;
235  auto vertex_id = digraph[vertex].id;
236  for (boost::tie(e_outIt, e_outEnd) =
237  boost::out_edges(vertex, digraph.graph);
238  e_outIt != e_outEnd; e_outIt++) {
239  auto out_edge_id = digraph.graph[*e_outIt].id;
240  insert_vertex(vertex_id, out_edge_id);
241  }
242 
243  for (boost::tie(e_inIt, e_inEnd) =
244  boost::in_edges(vertex, digraph.graph);
245  e_inIt != e_inEnd; e_inIt++) {
246  auto in_edge_id = digraph.graph[*e_inIt].id;
247  insert_vertex(vertex_id, in_edge_id);
248 
249  for (boost::tie(e_outIt, e_outEnd) =
250  boost::out_edges(vertex, digraph.graph);
251  e_outIt != e_outEnd; e_outIt++) {
252  auto out_edge_id = digraph.graph[*e_outIt].id;
253 
254  ++m_num_edges;
255 
256  graph_add_edge(

266  // Connect all of the line graphs that were just created using the

269  boost::vertices(digraph.graph);
