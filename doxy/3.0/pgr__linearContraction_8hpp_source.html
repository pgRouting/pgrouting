167  void process_shortcut(G &graph, V u, V v, V w) {
168  auto e1 = graph.get_min_cost_edge(u, v);
169  auto e2 = graph.get_min_cost_edge(v, w);
170 
171  if (std::get<2>(e1) && std::get<2>(e2)) {
172  auto contracted_vertices = std::get<1>(e1) + std::get<1>(e2);
173  double cost = std::get<0>(e1) + std::get<0>(e2);
174  contracted_vertices += graph[v].id;
175  contracted_vertices += graph[v].contracted_vertices();
176 
177  // Create shortcut
178  CH_edge shortcut(
179  get_next_id(),
180  graph[u].id,
181  graph[w].id,
182  cost);
183  shortcut.contracted_vertices() = contracted_vertices;
184 
185  graph.add_shortcut(shortcut, u, w);
186  }
