66  using E = typename G::E;
67  Identifiers<E> eids;
68  for (auto e : boost::make_iterator_range(boost::edges(graph.graph))) {
69  if (graph[e].id < 0) {
70  eids += e;
71  pgassert(!graph[e].contracted_vertices().empty());

73  pgassert(graph[e].contracted_vertices().empty());

75  }
