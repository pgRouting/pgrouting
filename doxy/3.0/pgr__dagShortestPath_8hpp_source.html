58  Path dag(
59  G &graph,
60  int64_t start_vertex,
61  int64_t end_vertex,
62  bool only_cost = false) {
63  clear();
64 
65  // adjust predecessors and distances vectors
66  predecessors.resize(graph.num_vertices());
67  distances.resize(
68  graph.num_vertices(),
69  std::numeric_limits<double>::infinity());
70 
71 
72  if (!graph.has_vertex(start_vertex)
73  || !graph.has_vertex(end_vertex)) {
74  return Path(start_vertex, end_vertex);
75  }
76 
77  // get the graphs source and target
78  auto v_source(graph.get_V(start_vertex));
79  auto v_target(graph.get_V(end_vertex));
80 
81  // perform the algorithm
82  dag_1_to_1(graph, v_source, v_target);
83 
84  // get the results
85  return Path(
86  graph,
87  v_source, v_target,
88  predecessors, distances,
89  only_cost, true);
90  }

94  std::deque<Path> dag(
95  G &graph,
96  int64_t start_vertex,
97  const std::vector< int64_t > &end_vertex,
98  bool only_cost) {
99  // adjust predecessors and distances vectors
100  clear();
101  size_t n_goals = std::numeric_limits<size_t>::max();
102  predecessors.resize(graph.num_vertices());
103  distances.resize(
104  graph.num_vertices(),
105  std::numeric_limits<double>::infinity());
106 
107 
108  // get the graphs source and target
109  if (!graph.has_vertex(start_vertex))
110  return std::deque<Path>();
111  auto v_source(graph.get_V(start_vertex));
112 
113  std::set< V > s_v_targets;
114  for (const auto &vertex : end_vertex) {
115  if (graph.has_vertex(vertex)) {
116  s_v_targets.insert(graph.get_V(vertex));
117  }
118  }
119 
120  std::vector< V > v_targets(s_v_targets.begin(), s_v_targets.end());
121  // perform the algorithm
122  dag_1_to_many(graph, v_source, v_targets, n_goals);
123 
124  std::deque< Path > paths;
125  // get the results // route id are the targets
126  paths = get_paths(graph, v_source, v_targets, only_cost);
127 
128  std::stable_sort(paths.begin(), paths.end(),
129  [](const Path &e1, const Path &e2)->bool {
130  return e1.end_id() < e2.end_id();
131  });
132 
133  return paths;
134  }

136  // preparation for many to 1
137  std::deque<Path> dag(
138  G &graph,
139  const std::vector < int64_t > &start_vertex,
140  int64_t end_vertex,
141  bool only_cost) {
142  std::deque<Path> paths;
143 
144  for (const auto &start : start_vertex) {
145  paths.push_back(
146  dag(graph, start, end_vertex, only_cost));
147  }
148 
149  std::stable_sort(paths.begin(), paths.end(),
150  [](const Path &e1, const Path &e2)->bool {
151  return e1.start_id() < e2.start_id();
152  });
153  return paths;
154  }

157  // preparation for many to many
158  std::deque<Path> dag(
159  G &graph,
160  const std::vector< int64_t > &start_vertex,
161  const std::vector< int64_t > &end_vertex,
162  bool only_cost) {
163  // a call to 1 to many is faster for each of the sources
164  std::deque<Path> paths;
165 
166  for (const auto &start : start_vertex) {
167  auto r_paths = dag(
168  graph,
169  start, end_vertex,
170  only_cost);
171  paths.insert(paths.begin(), r_paths.begin(), r_paths.end());
172  }
173 
174  std::sort(paths.begin(), paths.end(),
175  [](const Path &e1, const Path &e2)->bool {
176  return e1.end_id() < e2.end_id();
177  });
178  std::stable_sort(paths.begin(), paths.end(),
179  [](const Path &e1, const Path &e2)->bool {
180  return e1.start_id() < e2.start_id();
181  });
182  return paths;
183  }

207  } catch (...) {

283  class dijkstra_one_goal_visitor : public boost::default_dijkstra_visitor {
284  public:
285  explicit dijkstra_one_goal_visitor(V goal) : m_goal(goal) {}
286  template <class B_G>
287  void examine_vertex(V &u, B_G &) {
288  if (u == m_goal) throw found_goals();
289  }
290  private:
291  V m_goal;
292  };

295  class dijkstra_many_goal_visitor : public boost::default_dijkstra_visitor {
296  public:
297  explicit dijkstra_many_goal_visitor(
298  const std::vector< V > &goals,
299  size_t n_goals) :
300  m_goals(goals.begin(), goals.end()),
301  m_n_goals(n_goals) {}
302  template <class B_G>
303  void examine_vertex(V u, B_G &) {
304  auto s_it = m_goals.find(u);
305  if (s_it == m_goals.end()) return;
306 
307  // found one more goal
308  m_goals.erase(s_it);
309 
310  // all goals found
311  if (m_goals.size() == 0) throw found_goals();
312 
313  // number of requested goals found
314  --m_n_goals;
315  if (m_n_goals == 0) throw found_goals();
316  }
317 
318  private:
319  std::set< V > m_goals;
320  size_t m_n_goals;
321  };
