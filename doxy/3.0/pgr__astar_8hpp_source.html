72  double factor,
73  double epsilon,

78  distances.resize(graph.num_vertices());
79 
80  if (!graph.has_vertex(start_vertex)

85  auto v_source(graph.get_V(start_vertex));

113  if (!graph.has_vertex(start_vertex)) return std::deque<Path>();

116  std::vector<V> v_targets;

179  // heuristic for one goal
180  class distance_heuristic : public boost::astar_heuristic< B_G, double > {
181  public:
182  distance_heuristic(B_G &g, V goal, int heuristic, double factor)
183  : m_g(g),
184  m_factor(factor),
185  m_heuristic(heuristic) {
186  m_goals.insert(goal);
187  }
188  distance_heuristic(
189  B_G &g,
190  const std::vector< V > &goals,
191  int heuristic,
192  double factor)
193  : m_g(g),
194  m_goals(goals.begin(), goals.end()),
195  m_factor(factor),
196  m_heuristic(heuristic) {}
197 
198  double operator()(V u) {
199  if (m_heuristic == 0) return 0;
200  if (m_goals.empty()) return 0;
201  double best_h((std::numeric_limits<double>::max)());
202  for (auto goal : m_goals) {
203  double current((std::numeric_limits<double>::max)());
204  double dx = m_g[goal].x() - m_g[u].x();
205  double dy = m_g[goal].y() - m_g[u].y();
206  switch (m_heuristic) {
207  case 0:
208  current = 0;
209  break;
210  case 1:
211  current = std::fabs((std::max)(dx, dy)) * m_factor;
212  break;
213  case 2:
214  current = std::fabs((std::min)(dx, dy)) * m_factor;
215  break;
216  case 3:
217  current = (dx * dx + dy * dy) * m_factor * m_factor;
218  break;
219  case 4:
220  current = std::sqrt(dx * dx + dy * dy) * m_factor;
221  break;
222  case 5:
223  current = (std::fabs(dx) + std::fabs(dy)) * m_factor;
224  break;
225  default:
226  current = 0;
227  }
228  if (current < best_h) {
229  best_h = current;
230  }
231  }
232  {
233  auto s_it = m_goals.find(u);
234  if (!(s_it == m_goals.end())) {
235  // found one more goal
236  m_goals.erase(s_it);
237  }
238  }
239  return best_h;
240  }
241 
242  private:
243  B_G &m_g;
244  std::set< V > m_goals;
245  double m_factor;
246  int m_heuristic;
247  }; // class distance_heuristic

289  G &graph,

292  int heuristic,

294  double epsilon) {

299  graph.graph, source,

302  boost::predecessor_map(&predecessors[0])
303  .weight_map(get(&pgrouting::Basic_edge::cost, graph.graph))
304  .distance_map(&distances[0])
305  .visitor(astar_one_goal_visitor(target)));
306  }

308  found = true; // Target vertex found
309  }
310  return found;
