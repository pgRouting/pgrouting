59  Path bellman_ford(
61  G &graph,
62  int64_t start_vertex,
63  int64_t end_vertex,
64  bool only_cost = false) {
65  clear();
66  log << std::string(__FUNCTION__) << "\n";
67 
68  // adjust predecessors and distances vectors
69  predecessors.resize(graph.num_vertices());
70  distances.resize(graph.num_vertices());
71 
72 
73  if (!graph.has_vertex(start_vertex)
74  || !graph.has_vertex(end_vertex)) {
75  return Path(start_vertex, end_vertex);
76  }
77 
78  // get the graphs source and target
79  auto v_source(graph.get_V(start_vertex));
80  auto v_target(graph.get_V(end_vertex));
81 
82  // perform the algorithm
83  bellman_ford_1_to_1(graph, v_source, v_target);
84 
85  // get the results
86  return Path(
87  graph,
88  v_source, v_target,
89  predecessors, distances,
90  only_cost, true);
91  }

95  std::deque<Path> bellman_ford(
96  G &graph,
97  int64_t start_vertex,
98  const std::vector< int64_t > &end_vertex,
99  bool only_cost = false) {
100  // adjust predecessors and distances vectors
101  clear();
102  log << std::string(__FUNCTION__) << "\n";
103  predecessors.resize(graph.num_vertices());
104  distances.resize(graph.num_vertices());
105 
106  // get the graphs source and target
107  if (!graph.has_vertex(start_vertex))
108  return std::deque<Path>();
109  auto v_source(graph.get_V(start_vertex));
110 
111  std::set< V > s_v_targets;
112  for (const auto &vertex : end_vertex) {
113  if (graph.has_vertex(vertex)) {
114  s_v_targets.insert(graph.get_V(vertex));
115  }
116  }
117 
118  std::vector< V > v_targets(s_v_targets.begin(), s_v_targets.end());
119  // perform the algorithm
120  bellman_ford_1_to_many(graph, v_source);
121 
122  std::deque< Path > paths;
123  // get the results // route id are the targets
124  paths = get_paths(graph, v_source, v_targets, only_cost);
125 
126  std::stable_sort(paths.begin(), paths.end(),
127  [](const Path &e1, const Path &e2)->bool {
128  return e1.end_id() < e2.end_id();
129  });
130 
131  return paths;
132  }

134  // BellmanFord many to 1
135  std::deque<Path> bellman_ford(
136  G &graph,
137  const std::vector < int64_t > &start_vertex,
138  int64_t end_vertex,
139  bool only_cost = false) {
140  std::deque<Path> paths;
141  log << std::string(__FUNCTION__) << "\n";
142  for (const auto &start : start_vertex) {
143  paths.push_back(
144  bellman_ford(graph, start, end_vertex, only_cost));
145  }
146 
147  std::stable_sort(paths.begin(), paths.end(),
148  [](const Path &e1, const Path &e2)->bool {
149  return e1.start_id() < e2.start_id();
150  });
151  return paths;
152  }

155  // BellmanFord many to many
156  std::deque<Path> bellman_ford(
157  G &graph,
158  const std::vector< int64_t > &start_vertex,
159  const std::vector< int64_t > &end_vertex,
160  bool only_cost = false) {
161  // a call to 1 to many is faster for each of the sources
162  std::deque<Path> paths;
163  log << std::string(__FUNCTION__) << "\n";
164 
165  for (const auto &start : start_vertex) {
166  auto r_paths = bellman_ford(graph, start, end_vertex, only_cost);
167  paths.insert(paths.begin(), r_paths.begin(), r_paths.end());
168  }
169 
170 
171  std::sort(paths.begin(), paths.end(),
172  [](const Path &e1, const Path &e2)->bool {
173  return e1.end_id() < e2.end_id();
174  });
175  std::stable_sort(paths.begin(), paths.end(),
176  [](const Path &e1, const Path &e2)->bool {
177  return e1.start_id() < e2.start_id();
178  });
179  return paths;
180  }

186  bool bellman_ford_1_to_1(
187  G &graph,
188  V source,
189  V target) {
190  log << std::string(__FUNCTION__) << "\n";
191 
192  try {
193  bool negative = boost::bellman_ford_shortest_paths(
194  graph.graph,
195  static_cast<int>(graph.num_vertices()),
196  boost::predecessor_map(&predecessors[0])
197  .weight_map(get(&G::G_T_E::cost, graph.graph))
198  .distance_map(&distances[0])
199  .root_vertex(source));
200  } catch (boost::exception const& ex) {
201  (void)ex;
202  throw;
203  } catch (std::exception &e) {
204  (void)e;
205  throw;
206  } catch (...) {
207  throw;
208  }
209  return true;
210  }
