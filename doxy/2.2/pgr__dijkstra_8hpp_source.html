271  class dijkstra_many_goal_visitor : public boost::default_dijkstra_visitor {
272  public:
273  explicit dijkstra_many_goal_visitor(std::set< V > goals)
274  :m_goals(goals) {}
275  template <class B_G>
276  void examine_vertex(V u, B_G &g) {
277  typename std::set< V >::iterator s_it;
278  s_it = m_goals.find(u);
279  if (s_it == m_goals.end()) return;
280  // we found one more goal
281  m_goals.erase(s_it);
282  if (m_goals.size() == 0) throw found_goals();
283  num_edges(g);
284  }
285  private:
286  std::set< V > m_goals;
287  };

299  template <class B_G>

320 void
321 Pgr_dijkstra< G >::get_nodesInDistance(
322  G &graph,
323  Path &path,
324  V source,
325  double distance) {
326  path.clear();
327  double cost;
328  int64_t edge_id;
329  Path r_path(graph[source].id, graph[source].id);
330  for (V i = 0; i < distances.size(); ++i) {
331  if (distances[i] <= distance) {
332  cost = distances[i] - distances[predecessors[i]];
333  edge_id = graph.get_edge_id(predecessors[i], i, cost);
334  r_path.push_back(
335  {graph[i].id,
336  edge_id, cost,
337  distances[i]});
338  }
339  }
340  path = r_path;
341 }

360 template < class G >
361 void
362 Pgr_dijkstra< G >::get_path(
363  const G &graph,
364  V source,
365  V target,
366  Path &r_path) const {
367  // backup of the target
368  V target_back = target;
369  uint64_t from(graph.graph[source].id);
370  uint64_t to(graph.graph[target].id);
371 
372  // no path was found
373  if (target == predecessors[target]) {
374  r_path.clear();
375  return;
376  }
377 
378  // findout how large is the path
379  int64_t result_size = 1;
380  while (target != source) {
381  if (target == predecessors[target]) break;
382  result_size++;
383  target = predecessors[target];
384  }
385 
386  // recover the target
387  target = target_back;
388 
389  // variables that are going to be stored
390  int64_t vertex_id;
391  int64_t edge_id;
392  double cost;
393 
394  // working from the last to the beginning
395 
396  // initialize the sequence
397  auto seq = result_size;
398  // the last stop is the target
399  Path path(from, to);
400  path.push_front(
401  {graph.graph[target].id, -1,
402  0, distances[target]});
403 
404  while (target != source) {
405  // we are done when the predecesor of the target is the target
406  if (target == predecessors[target]) break;
407  // values to be inserted in the path
408  --seq;
409  cost = distances[target] - distances[predecessors[target]];
410  vertex_id = graph.graph[predecessors[target]].id;
411  edge_id = graph.get_edge_id(predecessors[target], target, cost);
412 
413  path.push_front({vertex_id, edge_id,
414  cost, (distances[target] - cost)});
415  target = predecessors[target];
416  }
417  r_path = path;
418  return;
419 }

527  clear();
528 
529  // adjust predecessors and distances vectors

531  distances.resize(graph.num_vertices());
532 
533  // get the graphs source and target
534  V v_source;
535  V v_target;
536 
537  if (!graph.get_gVertex(start_vertex, v_source)
538  || !graph.get_gVertex(end_vertex, v_target)) {
539  path.clear();
540  return;
541  }
542 
543  // perform the algorithm
544  dijkstra_1_to_1(graph, v_source, v_target);
545 
546  // get the results
547  if (only_cost) {
548  get_cost(graph, v_source, v_target, path);
549  } else {
550  get_path(graph, v_source, v_target, path);
551  }
552  return;
553 }
