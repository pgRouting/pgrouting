37 int64_t* pgr_get_bigIntArray(size_t *arrlen, ArrayType *input) {
38  int ndims;
39  bool *nulls;
40  Oid i_eltype;
41  int16 i_typlen;
42  bool i_typbyval;
43  char i_typalign;
44  Datum *i_data;
45  int i;
46  int n;
47  int64_t *data;
48 
49  PGR_DBG("Geting integer array");
50  /* get input array element type */
51  i_eltype = ARR_ELEMTYPE(input);
52  get_typlenbyvalalign(i_eltype, &i_typlen, &i_typbyval, &i_typalign);
53 
54 
55  /* validate input data type */
56  switch (i_eltype) {
57  case INT2OID:
58  case INT4OID:
59  case INT8OID:
60  break;
61  default:
62  elog(ERROR, "Expected array of ANY-INTEGER");
63  return (int64_t*) NULL;
64  break;
65  }
66 
67  /* get various pieces of data from the input array */
68  ndims = ARR_NDIM(input);
69  n = (*ARR_DIMS(input));
70  (*arrlen) = (size_t)(n);
71 
72  if (ndims != 1) {
73  elog(ERROR, "One dimension expected");
74  }
75 
76  /* get src data */
77  deconstruct_array(input, i_eltype, i_typlen, i_typbyval, i_typalign,
78  &i_data, &nulls, &n);
79 
80  /* construct a C array */
81  data = (int64_t *) malloc((*arrlen) * sizeof(int64_t));
82 
83  if (!data) {
84  elog(ERROR, "Out of memory!");
85  }
86 
87  PGR_DBG("array size %ld", (*arrlen));
88 
89  for (i = 0; i < (*arrlen); i++) {
90  if (nulls[i]) {
91  data[i] = -1;
92  } else {
93  switch (i_eltype) {
94  case INT2OID:
95  data[i] = (int64_t) DatumGetInt16(i_data[i]);
96  break;
97  case INT4OID:
98  data[i] = (int64_t) DatumGetInt32(i_data[i]);
99  break;
100  case INT8OID:
101  data[i] = DatumGetInt64(i_data[i]);
102  break;
103  }
104  }
105  }
106 
107  pfree(nulls);
108  pfree(i_data);
109 
110  PGR_DBG("Finished processing array");
111  return (int64_t*)data;
112 }
