101 double get_angle(Point p, Point q, Point r)
102 {
103  double m_pi(3.14159265358979323846);
104  Vector v1(q, p);
105  Vector v2(q, r);
106  double cross = v1.x() * v2.y() - v1.y() * v2.x();
107  double dot = v1.x() * v2.x() + v1.y() * v2.y();
108  double angle = atan2(cross, dot);
109  if (angle < 0.0) {
110  angle += 2 * m_pi;
111  }
112  return angle;
113 }

116 void find_next_edge(Segment s, std::vector<Segment>& segments,
117  std::set<int>& unusedIndexes, std::vector<Polygon_2>& rings)
118 {
119  if(unusedIndexes.empty()
120  || prev_size == unusedIndexes.size())
121  {
122  return;
123  }
124 
125  prev_size = unusedIndexes.size();
126 
127  Point start = s.source();
128  Point end = s.target();
129  rings.back().push_back(end);
130 
131  std::vector<int> nextIndexes;
132  for(unsigned int i = 0;i < segments.size(); i++)
133  {
134  if (unusedIndexes.find(i) != unusedIndexes.end())
135  {
136  Point source = segments.at(i).source();
137  if(source == end)
138  {
139  nextIndexes.push_back(i);
140  }
141  }
142  }
143  if (nextIndexes.size() == 1)
144  {
145  int i = nextIndexes.at(0);
146  unusedIndexes.erase(i);
147  find_next_edge(segments.at(i), segments, unusedIndexes, rings);
148  }
149  else if (nextIndexes.size() > 1)
150  {
151  std::vector< std::pair<double, int> > nextAngles;
152  for (unsigned int i = 0; i < nextIndexes.size(); i++)
153  {
154  int j = nextIndexes.at(i);
155  Point target = segments.at(j).target();
156  double angle = get_angle(start, end, target);
157  nextAngles.push_back(std::pair<double, int>(angle, j));
158  }
159  std::sort(nextAngles.begin(), nextAngles.end());
160  int i = nextAngles.begin()->second;
161  unusedIndexes.erase(i);
162  find_next_edge(segments.at(i), segments, unusedIndexes, rings);
163  }
164 
165  if (!unusedIndexes.empty())
166  {
167  for (unsigned int i = 0; i < segments.size(); i++)
168  {
169  if (unusedIndexes.find(i) != unusedIndexes.end())
170  {
171  Polygon_2 ring;
172  ring.push_back(segments.at(i).source());
173  rings.push_back(ring);
174  unusedIndexes.erase(i);
175  find_next_edge(segments.at(i), segments, unusedIndexes, rings);
176  }
