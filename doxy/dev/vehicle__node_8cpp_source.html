66 void
67 Vehicle_node::evaluate(const Vehicle_node &pred, double cargoLimit) {
68  /* time */
69  m_travel_time = pred.travel_time_to(*this);
70  m_arrival_time = pred.departure_time() + travel_time();
71  m_wait_time = is_early_arrival(arrival_time()) ?
72  opens() - m_arrival_time :
73  0;
74  m_departure_time = arrival_time() + wait_time() + service_time();
75 
76  /* time aggregates */
77  m_tot_travel_time = pred.total_travel_time() + travel_time();
78  m_tot_wait_time = pred.total_wait_time() + wait_time();
79  m_tot_service_time = pred.total_service_time() + service_time();
80 
81  /* cargo aggregates */
82  if (is_dump() && pred.cargo() >= 0) {
83  m_demand = -pred.cargo();
84  }
85  m_cargo = pred.cargo() + demand();
86 
87  /* cargo aggregates */
88 
89  m_twvTot = has_twv() ? pred.twvTot() + 1 : pred.twvTot();
90  m_cvTot = has_cv(cargoLimit) ? pred.cvTot() + 1 : pred.cvTot();
91  m_delta_time = departure_time() - pred.departure_time();
92 }
