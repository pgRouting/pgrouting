110 template < class G >
111 std::vector<pgr_componentV_t>
112 Pgr_components< G >::do_connectedComponentsV(
113  G &graph) {
114  // call to boost
115  std::vector< V > components(num_vertices(graph.graph));
116  boost::connected_components(graph.graph, &components[0]);
117 
118  // generate V_to_id
119  generate_map(graph.vertices_map);
120 
121  // generate results
122  auto totalNodes = num_vertices(graph.graph);
123 
124  std::vector< pgr_componentV_t > results;
125  results.resize(totalNodes);
126 
127  std::vector< int64_t > result_comp;
128  result_comp.resize(0);
129  size_t temp_size = 0;
130  for (auto i = 0; i < totalNodes; i++) {
131  results[i].node = V_to_id.find(i)->second;
132  if (components[i] >= temp_size) {
133  result_comp.push_back(results[i].node);
134  temp_size++;
135  } else {
136  result_comp[components[i]] =
137  std::min(results[i].node, result_comp[components[i]]);
138  }
139  results[i].n_seq = -100;
140  }
141 
142  // generate component number
143  for (int i = 0; i < totalNodes; i++) {
144  results[i].component = result_comp[components[i]];
145  }
146 
147  // sort results and generate n_seq
148  std::sort(results.begin(), results.end(), sort_cmp);
149  for (int i = 0; i < totalNodes; i++) {
150  if (i == 0 || results[i].component != results[i - 1].component) {
151  results[i].n_seq = 1;
152  } else {
153  results[i].n_seq = results[i - 1].n_seq + 1;
154  }
155  }
156  return results;
157 }
